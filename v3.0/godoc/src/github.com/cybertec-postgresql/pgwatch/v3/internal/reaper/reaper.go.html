   149  						return metricDefinitionMap.PresetDefs[monitoredDB.PresetMetricsStandby].Metrics
   150  					}
   151  					return nil
   152  				}()

   191  						} else {

   194  							hostLastKnownStatusInRecovery[dbUnique] = false

   197  					}
   198  				}

   200  				if mainLoopCount == 0 && opts.Sources.TryCreateListedExtsIfMissing != "" && !ver.IsInRecovery {
   201  					extsToCreate := strings.Split(opts.Sources.TryCreateListedExtsIfMissing, ",")
   202  					extsCreated := TryCreateMissingExtensions(mainContext, dbUnique, extsToCreate, ver.Extensions)
   203  					logger.Infof("[%s] %d/%d extensions created based on --try-create-listed-exts-if-missing input %v", dbUnique, len(extsCreated), len(extsToCreate), extsCreated)
   204  				}
   205  			}
   206  
   207  			for metricName, interval := range metricConfig {
   208  				metric := metricName
   209  				metricDefOk := false
   210  
   211  				if strings.HasPrefix(metric, recoPrefix) {

   217  					metricDefOk = true
   218  				} else {

   231  						metricCtx, cancelFunc := context.WithCancel(mainContext)

   233

   235  						if _, isSpecialMetric := specialMetrics[metricName]; !isSpecialMetric {

   237  							if err != nil {
   238  								logger.Warning("Failed to determine possible re-routing name, Grafana dashboards with re-routed metrics might not show all hosts")
   239  							} else {
   240  								mvp, err := GetMetricVersionProperties(metricName, vme, nil)
   241  								if err != nil && !strings.Contains(err.Error(), "too old") {

   243  								} else if mvp.StorageName != "" {
   244  									metricNameForStorage = mvp.StorageName
   245  								}
   246  							}
   247  						}
   248  
   249  						if err := measurementsWriter.SyncMetrics(dbUnique, metricNameForStorage, "add"); err != nil {
   250  							logger.Error(err)
   251  						}
   252  
   253  						go r.reapMetricMeasurementsFromSource(metricCtx,
   254  							dbUnique,
   255  							dbUniqueOrig,
   256  							srcType,
   257  							metric,
   258  							metricConfig)
   259  					}
   260  				} else if (!metricDefOk && chOk) || interval <= 0 {
   261  					// metric definition files were recently removed or interval set to zero
   262  					if cancelFunc, isOk := cancelFuncs[dbMetric]; isOk {
   263  						cancelFunc()
   264  					}
   265  					logger.Warning("shutting down metric", metric, "for", monitoredDB.Name)
   266  					delete(cancelFuncs, dbMetric)
   267  				} else if !metricDefOk {
   268  					epoch, ok := lastSQLFetchError.Load(metric)
   269  					if !ok || ((time.Now().Unix() - epoch.(int64)) > 3600) { // complain only 1x per hour
   270  						logger.Warningf("metric definition \"%s\" not found for \"%s\"", metric, dbUnique)
   271  						lastSQLFetchError.Store(metric, time.Now().Unix())
   272  					}
   273  				} else {
   274  					// check if interval has changed
   275  					if hostMetricIntervalMap[dbMetric] != interval {
   276  						logger.Warning("updating interval update for", dbUnique, metric)
   277  						hostMetricIntervalMap[dbMetric] = interval
   278  					}
   279  				}
   280  			}
   281  		}
   282  
   283  		if mainLoopCount == 0 {

   285  		}
   286  
   287  		// loop over existing channels and stop workers if DB or metric removed from config
   288  		// or state change makes it uninteresting
   289  		logger.Debug("checking if any workers need to be shut down...")
   290  		for dbMetric, cancelFunc := range cancelFuncs {
   291  			var currentMetricConfig map[string]float64
   292  			var dbInfo *sources.MonitoredDatabase
   293  			var ok, dbRemovedFromConfig bool
   294  			singleMetricDisabled := false
   295  			splits := strings.Split(dbMetric, dbMetricJoinStr)
   296  			db := splits[0]
   297  			metric := splits[1]
   298  			//log.Debugf("Checking if need to shut down worker for [%s:%s]...", db, metric)
   299  
   300  			_, wholeDbShutDownDueToRoleChange := hostsToShutDownDueToRoleChange[db]
   301  			if !wholeDbShutDownDueToRoleChange {
   302  				monitoredDbCacheLock.RLock()
   303  				dbInfo, ok = monitoredDbCache[db]
   304  				monitoredDbCacheLock.RUnlock()
   305  				if !ok { // normal removing of DB from config
   306  					dbRemovedFromConfig = true
   307  					logger.Debugf("DB %s removed from config, shutting down all metric worker processes...", db)
   308  				}
   309  			}
   310  
   311  			if !(wholeDbShutDownDueToRoleChange || dbRemovedFromConfig) { // maybe some single metric was disabled
   312  				MonitoredDatabasesSettingsLock.RLock()
   313  				verInfo, ok := MonitoredDatabasesSettings[db]
   314  				MonitoredDatabasesSettingsLock.RUnlock()
   315  				if !ok {
   316  					logger.Warningf("Could not find PG version info for DB %s, skipping shutdown check of metric worker process for %s", db, metric)
   317  					continue
   318  				}
   319  				if verInfo.IsInRecovery && dbInfo.PresetMetricsStandby > "" || !verInfo.IsInRecovery && dbInfo.PresetMetrics > "" {
   320  					continue // no need to check presets for single metric disabling
   321  				}
   322  				if verInfo.IsInRecovery && len(dbInfo.MetricsStandby) > 0 {
   323  					currentMetricConfig = dbInfo.MetricsStandby
   324  				} else {
   325  					currentMetricConfig = dbInfo.Metrics
   326  				}
   327  
   328  				interval, isMetricActive := currentMetricConfig[metric]
   329  				if !isMetricActive || interval <= 0 {
   330  					singleMetricDisabled = true
   331  				}
   332  			}
   333  
   334  			if mainContext.Err() != nil || wholeDbShutDownDueToRoleChange || dbRemovedFromConfig || singleMetricDisabled {
   335  				logger.Infof("shutting down gatherer for [%s:%s] ...", db, metric)
   336  				cancelFunc()
   337  				delete(cancelFuncs, dbMetric)
   338  				logger.Debugf("cancel function for [%s:%s] deleted", db, metric)
   339  				gatherersShutDown++
   340  				ClearDBUnreachableStateIfAny(db)
   341  				if err := measurementsWriter.SyncMetrics(db, metric, "remove"); err != nil {
   342  					logger.Error(err)
   343  				}
   344  			}
   345  		}
   346  
   347  		if gatherersShutDown > 0 {
   348  			logger.Warningf("sent STOP message to %d gatherers (it might take some time for them to stop though)", gatherersShutDown)
   349  		}
   350  
   351  		// Destroy conn pools and metric writers
   352  		CloseResourcesForRemovedMonitoredDBs(measurementsWriter, monitoredDbs, prevLoopMonitoredDBs, hostsToShutDownDueToRoleChange)
   353  
   354  	MainLoopSleep:
   355  		mainLoopCount++
   356  		prevLoopMonitoredDBs = slices.Clone(monitoredDbs)
   357  
   358  		logger.Debugf("main sleeping %ds...", opts.Sources.Refresh)
   359  		select {
   360  		case <-time.After(time.Second * time.Duration(opts.Sources.Refresh)):
   361  			// pass
   362  		case <-mainContext.Done():
   363  			return
   364  		}
   365  	}

   369  func (r *Reaper) reapMetricMeasurementsFromSource(ctx context.Context,
   370  	dbUniqueName, dbUniqueNameOrig string,
   371  	srcType sources.Kind,
   372  	metricName string,
   373  	configMap map[string]float64) {
   374  
   375  	hostState := make(map[string]map[string]string)
   376  	var lastUptimeS int64 = -1 // used for "server restarted" event detection
   377  	var lastErrorNotificationTime time.Time
   378  	var vme MonitoredDatabaseSettings
   379  	var mvp metrics.Metric
   380  	var err error
   381  	failedFetches := 0
   382  	lastDBVersionFetchTime := time.Unix(0, 0) // check DB ver. ev. 5 min
   383  
   384  	l := log.GetLogger(ctx).WithField("source", dbUniqueName).WithField("metric", metricName)
   385  	if metricName == specialMetricServerLogEventCounts {
   386  		mdb, err := GetMonitoredDatabaseByUniqueName(dbUniqueName)
   387  		if err != nil {
   388  			return
   389  		}
   390  		MonitoredDatabasesSettingsLock.RLock()
   391  		realDbname := MonitoredDatabasesSettings[dbUniqueName].RealDbname // to manage 2 sets of event counts - monitored DB + global
   392  		MonitoredDatabasesSettingsLock.RUnlock()
   393  		conn := mdb.Conn
   394  		metrics.ParseLogs(ctx, conn, mdb, realDbname, metricName, configMap, r.measurementCh) // no return
   395  		return
   396  	}
   397  
   398  	for {
   399  		interval := configMap[metricName]
   400  		if lastDBVersionFetchTime.Add(time.Minute * time.Duration(5)).Before(time.Now()) {
   401  			vme, err = GetMonitoredDatabaseSettings(ctx, dbUniqueName, srcType, false) // in case of errors just ignore metric "disabled" time ranges
   402  			if err != nil {
   403  				lastDBVersionFetchTime = time.Now()
   404  			}
   405  
   406  			mvp, err = GetMetricVersionProperties(metricName, vme, nil)
   407  			if err != nil {
   408  				l.Errorf("Could not get metric version properties: %v", err)
   409  				return
   410  			}
   411  		}
   412  
   413  		var metricStoreMessages []metrics.MeasurementEnvelope
   414  		var err error
   415  		mfm := MetricFetchConfig{
   416  			DBUniqueName:        dbUniqueName,
   417  			DBUniqueNameOrig:    dbUniqueNameOrig,
   418  			MetricName:          metricName,
   419  			Source:              srcType,
   420  			Interval:            time.Second * time.Duration(interval),
   421  			StmtTimeoutOverride: 0,
   422  		}
   423  
   424  		// 1st try local overrides for some metrics if operating in push mode
   425  		if r.opts.Metrics.DirectOSStats && IsDirectlyFetchableMetric(metricName) {
   426  			metricStoreMessages, err = FetchStatsDirectlyFromOS(ctx, mfm, vme, mvp)
   427  			if err != nil {
   428  				l.WithError(err).Errorf("Could not reader metric directly from OS")
   429  			}
   430  		}
   431  		t1 := time.Now()
   432  		if metricStoreMessages == nil {
   433  			metricStoreMessages, err = FetchMetrics(ctx, mfm, hostState, r.measurementCh, "", r.opts)
   434  		}
   435  		t2 := time.Now()
   436  
   437  		if t2.Sub(t1) > (time.Second * time.Duration(interval)) {
   438  			l.Warningf("Total fetching time of %vs bigger than %vs interval", t2.Sub(t1).Truncate(time.Millisecond*100).Seconds(), interval)
   439  		}
   440  
   441  		if err != nil {
   442  			failedFetches++
   443  			// complain only 1x per 10min per host/metric...
   444  			if lastErrorNotificationTime.IsZero() || lastErrorNotificationTime.Add(time.Second*time.Duration(600)).Before(time.Now()) {
   445  				l.WithError(err).Error("failed to fetch metric data")
   446  				if failedFetches > 1 {
   447  					l.Errorf("Total failed fetches: %d", failedFetches)
   448  				}
   449  				lastErrorNotificationTime = time.Now()
   450  			}
   451  		} else if metricStoreMessages != nil {
   452  			if len(metricStoreMessages[0].Data) > 0 {
   453  
   454  				// pick up "server restarted" events here to avoid doing extra selects from CheckForPGObjectChangesAndStore code
   455  				if metricName == "db_stats" {
   456  					postmasterUptimeS, ok := (metricStoreMessages[0].Data)[0]["postmaster_uptime_s"]
   457  					if ok {
   458  						if lastUptimeS != -1 {
   459  							if postmasterUptimeS.(int64) < lastUptimeS { // restart (or possibly also failover when host is routed) happened
   460  								message := "Detected server restart (or failover) of \"" + dbUniqueName + "\""
   461  								l.Warning(message)
   462  								detectedChangesSummary := make(metrics.Measurements, 0)
   463  								entry := metrics.Measurement{"details": message, "epoch_ns": (metricStoreMessages[0].Data)[0]["epoch_ns"]}
   464  								detectedChangesSummary = append(detectedChangesSummary, entry)
   465  								metricStoreMessages = append(metricStoreMessages,
   466  									metrics.MeasurementEnvelope{
   467  										DBName:     dbUniqueName,
   468  										SourceType: string(srcType),
   469  										MetricName: "object_changes",
   470  										Data:       detectedChangesSummary,
   471  										CustomTags: metricStoreMessages[0].CustomTags,
   472  									})
   473  							}
   474  						}
   475  						lastUptimeS = postmasterUptimeS.(int64)
   476  					}
   477  				}
   478  
   479  				r.measurementCh <- metricStoreMessages
   480  			}
   481  		}
   482  
   483  		select {
   484  		case <-ctx.Done():
   485  			return
   486  		case <-time.After(time.Second * time.Duration(interval)):
   487  			l.Debugf("MetricGathererLoop slept for %s", time.Second*time.Duration(interval))
   488  		}
   489  	}
   490  }

   492  func StoreMetrics(metrics []metrics.MeasurementEnvelope, storageCh chan<- []metrics.MeasurementEnvelope) (int, error) {
   493  	if len(metrics) > 0 {
   494  		storageCh <- metrics

   591  func FetchMetrics(ctx context.Context,
   592  	msg MetricFetchConfig,
   593  	hostState map[string]map[string]string,
   594  	storageCh chan<- []metrics.MeasurementEnvelope,
   595  	context string,
   596  	opts *cmdopts.Options) ([]metrics.MeasurementEnvelope, error) {
   597  
   598  	var dbSettings MonitoredDatabaseSettings
   599  	var dbVersion int
   600  	var err error
   601  	var sql string
   602  	var data, cachedData metrics.Measurements
   603  	var md *sources.MonitoredDatabase
   604  	var fromCache, isCacheable bool
   605  
   606  	md, err = GetMonitoredDatabaseByUniqueName(msg.DBUniqueName)
   607  	if err != nil {
   608  		log.GetLogger(ctx).Errorf("[%s:%s] could not get monitored DB details", msg.DBUniqueName, err)
   609  		return nil, err
   610  	}
   611  
   612  	dbSettings, err = GetMonitoredDatabaseSettings(ctx, msg.DBUniqueName, msg.Source, false)
   613  	if err != nil {
   614  		log.GetLogger(ctx).Error("failed to fetch pg version for ", msg.DBUniqueName, msg.MetricName, err)
   615  		return nil, err
   616  	}
   617  	if msg.MetricName == specialMetricDbSize || msg.MetricName == specialMetricTableStats {
   618  		if dbSettings.ExecEnv == execEnvAzureSingle && dbSettings.ApproxDBSizeB > 1e12 { // 1TB
   619  			subsMetricName := msg.MetricName + "_approx"
   620  			mvpApprox, err := GetMetricVersionProperties(subsMetricName, dbSettings, nil)
   621  			if err == nil && mvpApprox.StorageName == msg.MetricName {
   622  				log.GetLogger(ctx).Infof("[%s:%s] Transparently swapping metric to %s due to hard-coded rules...", msg.DBUniqueName, msg.MetricName, subsMetricName)
   623  				msg.MetricName = subsMetricName
   624  			}
   625  		}
   626  	}
   627  	dbVersion = dbSettings.Version
   628  
   629  	if msg.Source == sources.SourcePgBouncer {
   630  		dbVersion = 0 // version is 0.0 for all pgbouncer sql per convention
   631  	}
   632  
   633  	mvp, err := GetMetricVersionProperties(msg.MetricName, dbSettings, nil)
   634  	if err != nil && msg.MetricName != recoMetricName {
   635  		epoch, ok := lastSQLFetchError.Load(msg.MetricName + dbMetricJoinStr + fmt.Sprintf("%v", dbVersion))
   636  		if !ok || ((time.Now().Unix() - epoch.(int64)) > 3600) { // complain only 1x per hour
   637  			log.GetLogger(ctx).Infof("Failed to get SQL for metric '%s', version '%s': %v", msg.MetricName, dbSettings.VersionStr, err)
   638  			lastSQLFetchError.Store(msg.MetricName+dbMetricJoinStr+fmt.Sprintf("%v", dbVersion), time.Now().Unix())
   639  		}
   640  		if strings.Contains(err.Error(), "too old") {
   641  			return nil, nil
   642  		}
   643  		return nil, err
   644  	}
   645  
   646  	isCacheable = IsCacheableMetric(msg, mvp)
   647  	if isCacheable && opts.Metrics.InstanceLevelCacheMaxSeconds > 0 && msg.Interval.Seconds() > float64(opts.Metrics.InstanceLevelCacheMaxSeconds) {
   648  		cachedData = GetFromInstanceCacheIfNotOlderThanSeconds(msg, opts.Metrics.InstanceLevelCacheMaxSeconds)
   649  		if len(cachedData) > 0 {
   650  			fromCache = true
   651  			goto send_to_storageChannel
   652  		}
   653  	}
   654  
   655  	sql = mvp.GetSQL(dbVersion)
   656  
   657  	if sql == "" && !(msg.MetricName == specialMetricChangeEvents || msg.MetricName == recoMetricName) {
   658  		// let's ignore dummy SQLs
   659  		log.GetLogger(ctx).Debugf("[%s:%s] Ignoring fetch message - got an empty/dummy SQL string", msg.DBUniqueName, msg.MetricName)
   660  		return nil, nil
   661  	}
   662  
   663  	if (mvp.PrimaryOnly() && dbSettings.IsInRecovery) || (mvp.StandbyOnly() && !dbSettings.IsInRecovery) {
   664  		log.GetLogger(ctx).Debugf("[%s:%s] Skipping fetching of  as server not in wanted state (IsInRecovery=%v)", msg.DBUniqueName, msg.MetricName, dbSettings.IsInRecovery)
   665  		return nil, nil
   666  	}
   667  
   668  	if msg.MetricName == specialMetricChangeEvents && context != contextPrometheusScrape { // special handling, multiple queries + stateful
   669  		CheckForPGObjectChangesAndStore(ctx, msg.DBUniqueName, dbSettings, storageCh, hostState) // TODO no hostState for Prometheus currently
   670  	} else if msg.MetricName == recoMetricName && context != contextPrometheusScrape {
   671  		if data, err = GetRecommendations(ctx, msg.DBUniqueName, dbSettings); err != nil {
   672  			return nil, err
   673  		}
   674  	} else if msg.Source == sources.SourcePgPool {
   675  		if data, err = FetchMetricsPgpool(ctx, msg, dbSettings, mvp); err != nil {
   676  			return nil, err
   677  		}
   678  	} else {
   679  		data, err = DBExecReadByDbUniqueName(ctx, msg.DBUniqueName, sql)
   680  
   681  		if err != nil {
   682  			// let's soften errors to "info" from functions that expect the server to be a primary to reduce noise
   683  			if strings.Contains(err.Error(), "recovery is in progress") {
   684  				MonitoredDatabasesSettingsLock.RLock()
   685  				ver := MonitoredDatabasesSettings[msg.DBUniqueName]
   686  				MonitoredDatabasesSettingsLock.RUnlock()
   687  				if ver.IsInRecovery {
   688  					log.GetLogger(ctx).Debugf("[%s:%s] failed to fetch metrics: %s", msg.DBUniqueName, msg.MetricName, err)
   689  					return nil, err
   690  				}
   691  			}
   692  
   693  			if msg.MetricName == specialMetricInstanceUp {
   694  				log.GetLogger(ctx).WithError(err).Debugf("[%s:%s] failed to fetch metrics. marking instance as not up", msg.DBUniqueName, msg.MetricName)
   695  				data = make(metrics.Measurements, 1)
   696  				data[0] = metrics.Measurement{"epoch_ns": time.Now().UnixNano(), "is_up": 0} // should be updated if the "instance_up" metric definition is changed
   697  				goto send_to_storageChannel
   698  			}
   699  
   700  			if strings.Contains(err.Error(), "connection refused") {
   701  				SetDBUnreachableState(msg.DBUniqueName)
   702  			}
   703  
   704  			log.GetLogger(ctx).Infof("[%s:%s] failed to fetch metrics: %s", msg.DBUniqueName, msg.MetricName, err)
   705  
   706  			return nil, err
   707  		}
   708  
   709  		log.GetLogger(ctx).WithFields(map[string]any{"source": msg.DBUniqueName, "metric": msg.MetricName, "rows": len(data)}).Info("measurements fetched")
   710  		if regexIsPgbouncerMetrics.MatchString(msg.MetricName) { // clean unwanted pgbouncer pool stats here as not possible in SQL
   711  			data = FilterPgbouncerData(ctx, data, md.GetDatabaseName(), dbSettings)
   712  		}
   713  
   714  		ClearDBUnreachableStateIfAny(msg.DBUniqueName)
   715  
   716  	}
   717  
   718  	if isCacheable && opts.Metrics.InstanceLevelCacheMaxSeconds > 0 && msg.Interval.Seconds() > float64(opts.Metrics.InstanceLevelCacheMaxSeconds) {
   719  		PutToInstanceCache(msg, data)
   720  	}
   721  
   722  send_to_storageChannel:
   723  
   724  	if (opts.Sinks.RealDbnameField > "" || opts.Sinks.SystemIdentifierField > "") && msg.Source == sources.SourcePostgres {
   725  		MonitoredDatabasesSettingsLock.RLock()
   726  		ver := MonitoredDatabasesSettings[msg.DBUniqueName]
   727  		MonitoredDatabasesSettingsLock.RUnlock()
   728  		data = AddDbnameSysinfoIfNotExistsToQueryResultData(data, ver, opts)
   729  	}
   730  
   731  	if mvp.StorageName != "" {
   732  		log.GetLogger(ctx).Debugf("[%s] rerouting metric %s data to %s based on metric attributes", msg.DBUniqueName, msg.MetricName, mvp.StorageName)
   733  		msg.MetricName = mvp.StorageName
   734  	}
   735  	if fromCache {
   736  		log.GetLogger(ctx).Infof("[%s:%s] loaded %d rows from the instance cache", msg.DBUniqueName, msg.MetricName, len(cachedData))
   737  		return []metrics.MeasurementEnvelope{{DBName: msg.DBUniqueName, MetricName: msg.MetricName, Data: cachedData, CustomTags: md.CustomTags,
   738  			MetricDef: mvp, RealDbname: dbSettings.RealDbname, SystemIdentifier: dbSettings.SystemIdentifier}}, nil
   739  	}
   740  	return []metrics.MeasurementEnvelope{{DBName: msg.DBUniqueName, MetricName: msg.MetricName, Data: data, CustomTags: md.CustomTags,
   741  		MetricDef: mvp, RealDbname: dbSettings.RealDbname, SystemIdentifier: dbSettings.SystemIdentifier}}, nil
   742  
   743  }

   745  var pgBouncerNumericCountersStartVersion = 01_12_00 // pgBouncer changed internal counters data type in v1.12

   747  func FilterPgbouncerData(ctx context.Context, data metrics.Measurements, databaseToKeep string, vme MonitoredDatabaseSettings) metrics.Measurements {
   748  	filteredData := make(metrics.Measurements, 0)
   749  
   750  	for _, dr := range data {
   751  		//log.Debugf("bouncer dr: %+v", dr)
   752  		if _, ok := dr["database"]; !ok {
   753  			log.GetLogger(ctx).Warning("Expected 'database' key not found from pgbouncer_stats, not storing data")
   754  			continue
   755  		}
   756  		if (len(databaseToKeep) > 0 && dr["database"] != databaseToKeep) || dr["database"] == "pgbouncer" { // always ignore the internal 'pgbouncer' DB
   757  			log.GetLogger(ctx).Debugf("Skipping bouncer stats for pool entry %v as not the specified DBName of %s", dr["database"], databaseToKeep)
   758  			continue // and all others also if a DB / pool name was specified in config
   759  		}
   760  
   761  		dr["tag_database"] = dr["database"] // support multiple databases / pools via tags if DbName left empty
   762  		delete(dr, "database")              // remove the original pool name
   763  
   764  		if vme.Version >= pgBouncerNumericCountersStartVersion { // v1.12 counters are of type numeric instead of int64
   765  			for k, v := range dr {
   766  				if k == "tag_database" {
   767  					continue
   768  				}
   769  				decimalCounter, err := decimal.NewFromString(string(v.([]uint8)))
   770  				if err != nil {
   771  					log.GetLogger(ctx).Errorf("Could not parse \"%+v\" to Decimal: %s", string(v.([]uint8)), err)
   772  					return filteredData
   773  				}
   774  				dr[k] = decimalCounter.IntPart() // technically could cause overflow...but highly unlikely for 2^63
   775  			}
   776  		}
   777  		filteredData = append(filteredData, dr)
   778  	}
   779  
   780  	return filteredData
   781  }
