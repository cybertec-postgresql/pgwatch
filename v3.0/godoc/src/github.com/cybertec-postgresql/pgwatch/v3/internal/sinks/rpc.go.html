    11  // RPCWriter is a sink that sends metric measurements to a remote server using the RPC protocol.
    12  // Remote server should implement the Receiver interface. It's up to the implementer to define the
    13  // behavior of the server. It can be a simple logger, external storage, alerting system,
    14  // or an analytics system.
    15  type RPCWriter struct {
    16  	ctx     context.Context
    17  	address string
    18  	client  *rpc.Client
    19  }

    21  func NewRPCWriter(ctx context.Context, address string) (*RPCWriter, error) {
    22  	client, err := rpc.DialHTTP("tcp", address)
    23  	if err != nil {
    24  		return nil, err
    25  	}
    26  
    27  	rw := &RPCWriter{
    28  		ctx:     ctx,
    29  		address: address,
    30  		client:  client,
    31  	}
    32  	go rw.watchCtx()
    33  	return rw, nil
    34  }

    36  // Sends Measurement Message to RPC Sink
    37  func (rw *RPCWriter) Write(msgs []metrics.MeasurementEnvelope) error {
    38  	if rw.ctx.Err() != nil {
    39  		return rw.ctx.Err()
    40  	}
    41  	if len(msgs) == 0 {
    42  		return nil
    43  	}
    44  	l := log.GetLogger(rw.ctx).
    45  		WithField("sink", "rpc").
    46  		WithField("address", rw.address)
    47  	for _, msg := range msgs {
    48  		var logMsg string
    49  		if err := rw.client.Call("Receiver.UpdateMeasurements", &msg, &logMsg); err != nil {
    50  			return err
    51  		}
    52  		if len(logMsg) > 0 {
    53  			l.Info(logMsg)
    54  		}
    55  	}
    56  	return nil
    57  }

    65  func (rw *RPCWriter) SyncMetric(dbUnique string, metricName string, op string) error {
    66  	var logMsg string
    67  	if err := rw.client.Call("Receiver.SyncMetric", &SyncReq{
    68  		Operation:  op,
    69  		DbName:     dbUnique,
    70  		MetricName: metricName,
    71  	}, &logMsg); err != nil {
    72  		return err
    73  	}
    74  	if len(logMsg) > 0 {
    75  		log.GetLogger(rw.ctx).
    76  			WithField("sink", "rpc").
    77  			WithField("address", rw.address).Info(logMsg)
    78  	}
    79  	return nil
    80  }

    83  	<-rw.ctx.Done()
