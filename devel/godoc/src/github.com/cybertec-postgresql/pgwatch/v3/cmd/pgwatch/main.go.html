    20  // setupCloseHandler creates a 'listener' on a new goroutine which will notify the
    21  // program if it receives an interrupt from the OS. We then handle this by calling
    22  // our clean up procedure and exiting the program.
    23  func setupCloseHandler(cancel context.CancelFunc) {
    24  	c := make(chan os.Signal, 2)
    25  	signal.Notify(c, os.Interrupt, syscall.SIGTERM)
    26  	go func() {
    27  		<-c
    28  		log.GetLogger(mainCtx).Debug("SetupCloseHandler received an interrupt from OS. Closing session...")
    29  		cancel()
    30  		exitCode.Store(cmdopts.ExitCodeUserCancel)
    31  	}()
    32  }
