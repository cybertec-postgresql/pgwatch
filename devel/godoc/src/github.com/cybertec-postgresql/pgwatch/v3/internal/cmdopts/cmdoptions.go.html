    63  // New returns a new instance of Options and immediately executes the subcommand if specified.
    64  // Errors are returned for parsing only, if the command line arguments are invalid.
    65  // Subcommands execution errors (if any) do not affect error returned.
    66  // Subcommands are responsible for setting exit code only.
    67  func New(writer io.Writer) (cmdOpts *Options, err error) {
    68  	cmdOpts = new(Options)
    69  	parser := flags.NewParser(cmdOpts, flags.HelpFlag)
    70  	parser.SubcommandsOptional = true // if not command specified, start monitoring
    71  	addCommands(parser, cmdOpts)
    72  	nonParsedArgs, err := parser.Parse() // parse and execute subcommand if any
    73  	if err != nil {
    74  		if flagsErr, ok := err.(*flags.Error); ok && flagsErr.Type == flags.ErrHelp {
    75  			cmdOpts.Help = true
    76  		}
    77  		if !flags.WroteHelp(err) {
    78  			parser.WriteHelp(writer)
    79  		}
    80  		return cmdOpts, err
    81  	}
    82  	if cmdOpts.CommandCompleted { // subcommand executed, nothing to do more
    83  		return
    84  	}
    85  	if len(nonParsedArgs) > 0 { // we don't expect any non-parsed arguments
    86  		return cmdOpts, fmt.Errorf("unknown argument(s): %v", nonParsedArgs)
    87  	}
    88  	err = validateConfig(cmdOpts)
    89  	return
    90  }

   102  func (c *Options) GetConfigKind(arg string) (_ Kind, err error) {
   103  	if arg == "" {
   104  		return Kind(ConfigError), errors.New("no configuration provided")
   105  	}
   106  	if c.IsPgConnStr(arg) {
   107  		return Kind(ConfigPgURL), nil
   108  	}
   109  	var fi os.FileInfo
   110  	if fi, err = os.Stat(arg); err == nil {
   111  		if fi.IsDir() {
   112  			return Kind(ConfigFolder), nil
   113  		}
   114  		return Kind(ConfigFile), nil
   115  	}
   116  	return Kind(ConfigError), err
   117  }

   125  func (c *Options) InitMetricReader(ctx context.Context) (err error) {
   126  	if c.Metrics.Metrics == "" { //if config database is configured, use it for metrics as well
   127  		if c.IsPgConnStr(c.Sources.Sources) {
   128  			c.Metrics.Metrics = c.Sources.Sources
   129  		} else { // otherwise use built-in metrics
   130  			c.MetricsReaderWriter, err = metrics.NewYAMLMetricReaderWriter(ctx, "")
   131  			return err
   132  		}
   133  	}
   134  	if c.IsPgConnStr(c.Metrics.Metrics) {
   135  		c.MetricsReaderWriter, err = metrics.NewPostgresMetricReaderWriter(ctx, c.Metrics.Metrics)
   136  	} else {
   137  		c.MetricsReaderWriter, err = metrics.NewYAMLMetricReaderWriter(ctx, c.Metrics.Metrics)
   138  	}
   139  	return err
   140  }

   142  // InitSourceReader creates a new source reader based on the configuration kind from the options.
   143  func (c *Options) InitSourceReader(ctx context.Context) (err error) {
   144  	var configKind Kind
   145  	if c.Sources.Sources == "" { //if config database is configured, use it for sources as well
   146  		if c.IsPgConnStr(c.Metrics.Metrics) {
   147  			c.Sources.Sources = c.Metrics.Metrics
   148  		}
   149  	}
   150  	if configKind, err = c.GetConfigKind(c.Sources.Sources); err != nil {
   151  		return
   152  	}
   153  	switch configKind {
   154  	case ConfigPgURL:
   155  		c.SourcesReaderWriter, err = sources.NewPostgresSourcesReaderWriter(ctx, c.Sources.Sources)
   156  	default:
   157  		c.SourcesReaderWriter, err = sources.NewYAMLSourcesReaderWriter(ctx, c.Sources.Sources)
   158  	}
   159  	return err
   160  }

   181  func validateConfig(c *Options) error {
   182  	if len(c.Sources.Sources)+len(c.Metrics.Metrics) == 0 {
