    89  func tryDetermineLogFolder(ctx context.Context, conn db.PgxIface) (string, error) {
    90  	sql := `select current_setting('data_directory') as dd, current_setting('log_directory') as ld`
    91  	var dd, ld string
    92  	err := conn.QueryRow(ctx, sql).Scan(&dd, &ld)
    93  	if err != nil {
    94  		return "", err
    95  	}
    96  	if strings.HasPrefix(ld, "/") {
    97  		// we have a full path we can use
    98  		return ld, nil
    99  	}
   100  	return path.Join(dd, ld), nil

   161  func (lp *LogParser) regexMatchesToMap(matches []string) map[string]string {
   162  	result := make(map[string]string)
   163  	if len(matches) == 0 || lp.LogsMatchRegex == nil {

   174  // GetMeasurementEnvelope converts current event counts to a MeasurementEnvelope
   175  func (lp *LogParser) GetMeasurementEnvelope() metrics.MeasurementEnvelope {
   176  	allSeverityCounts := metrics.NewMeasurement(time.Now().UnixNano())
   177  	for _, s := range pgSeverities {
   178  		parsedCount, ok := lp.eventCounts[s]
   179  		if ok {
   180  			allSeverityCounts[strings.ToLower(s)] = parsedCount
   181  		} else {
   182  			allSeverityCounts[strings.ToLower(s)] = int64(0)
   183  		}
   184  		parsedCount, ok = lp.eventCountsTotal[s]
   185  		if ok {
   186  			allSeverityCounts[strings.ToLower(s)+"_total"] = parsedCount
   187  		} else {
   188  			allSeverityCounts[strings.ToLower(s)+"_total"] = int64(0)
   189  		}
   190  	}
   191  	return metrics.MeasurementEnvelope{
   192  		DBName:     lp.SourceConn.Name,
   193  		MetricName: specialMetricServerLogEventCounts,
   194  		Data:       metrics.Measurements{allSeverityCounts},
   195  		CustomTags: lp.SourceConn.CustomTags,
   196  	}
   197  }
