    29  	resolvedDbs := make(SourceConns, 0, len(srcs))
    30  	for _, s := range srcs {
    31  		if !s.IsEnabled {
    32  			continue
    33  		}
    34  		dbs, e := s.ResolveDatabases()
    35  		err = errors.Join(err, e)
    36  		resolvedDbs = append(resolvedDbs, dbs...)
    37  	}
    38  	return resolvedDbs, err

    87  func getTransport(conf HostConfigAttrs) (*tls.Config, error) {
    88  	var caCertPool *x509.CertPool
    89  
    90  	// create valid CertPool only if the ca certificate file exists
    91  	if conf.CAFile != "" {
    92  		caCert, err := os.ReadFile(conf.CAFile)
    93  		if err != nil {
    94  			return nil, fmt.Errorf("cannot load CA file: %s", err)
    95  		}
    96  
    97  		caCertPool = x509.NewCertPool()
    98  		caCertPool.AppendCertsFromPEM(caCert)
    99  	}
   100  
   101  	var certificates []tls.Certificate
   102  
   103  	// create valid []Certificate only if the client cert and key files exists
   104  	if conf.CertFile != "" && conf.KeyFile != "" {
   105  		cert, err := tls.LoadX509KeyPair(conf.CertFile, conf.KeyFile)
   106  		if err != nil {
   107  			return nil, fmt.Errorf("cannot load client cert or key file: %s", err)
   108  		}
   109  
   110  		certificates = []tls.Certificate{cert}
   111  	}
   112  
   113  	tlsClientConfig := new(tls.Config)
   114  
   115  	if caCertPool != nil {
   116  		tlsClientConfig.RootCAs = caCertPool
   117  		if certificates != nil {
   118  			tlsClientConfig.Certificates = certificates
   119  		}
   120  	}
   121  
   122  	return tlsClientConfig, nil
   123  }

   125  func getEtcdClusterMembers(s Source) ([]PatroniClusterMember, error) {
   126  	var ret = make([]PatroniClusterMember, 0)
   127  	var cfg client.Config
   128  
   129  	if len(s.HostConfig.DcsEndpoints) == 0 {
   130  		return ret, errors.New("missing ETCD connect info, make sure host config has a 'dcs_endpoints' key")
   131  	}
   132  
   133  	tlsConfig, err := getTransport(s.HostConfig)
   134  	if err != nil {
   135  		return nil, err
   136  	}
   137  	cfg = client.Config{
   138  		Endpoints:            s.HostConfig.DcsEndpoints,
   139  		TLS:                  tlsConfig,
   140  		DialKeepAliveTimeout: time.Second,
   141  		Username:             s.HostConfig.Username,
   142  		Password:             s.HostConfig.Password,
   143  		DialTimeout:          5 * time.Second,
   144  		Logger:               zap.NewNop(),
   145  	}
   146  
   147  	c, err := client.New(cfg)
   148  	if err != nil {
   149  		return ret, err
   150  	}
   151  	defer c.Close()
   152  
   153  	ctx, cancel := context.WithTimeoutCause(context.Background(), 5*time.Second, errors.New("etcd client timeout"))
   154  	defer cancel()
   155  
   156  	// etcd3 does not have a dir node.
   157  	// Key="/namespace/scope/leader", e.g. "/service/batman/leader"
   158  	// Key="/namespace/scope/members/node", e.g. "/service/batman/members/pg1"
   159  
   160  	p := path.Join(cmp.Or(s.HostConfig.Namespace, "/service"), s.HostConfig.Scope)
   161  	resp, err := c.Get(ctx, p, client.WithPrefix())
   162  	if err != nil {
   163  		return ret, cmp.Or(context.Cause(ctx), err)
   164  	}
   165  
   166  	for _, node := range resp.Kvs {
   167  		nodeData, err := jsonTextToStringMap(string(node.Value))
   168  		if err != nil {
   169  			logger.Errorf("Could not parse ETCD node data for node \"%s\": %s", node, err)
   170  			continue
   171  		}
   172  		// remove leading slash and split by "/"
   173  		parts := strings.Split(strings.TrimPrefix(string(node.Key), "/"), "/")
   174  		if len(parts) < 3 {
   175  			return nil, errors.New("invalid ETCD key format")
   176  		}
   177  		role := nodeData["role"]
   178  		connURL := nodeData["conn_url"]
   179  		scope := parts[1]
   180  		name := parts[3]
   181  		ret = append(ret, PatroniClusterMember{Scope: scope, ConnURL: connURL, Role: role, Name: name})
   182  	}
   183  
   184  	lastFoundClusterMembers[s.Name] = ret
   185  	return ret, nil
   186  }

   194  func ResolveDatabasesFromPatroni(source Source) (SourceConns, error) {
   195  	var mds []*SourceConn
   196  	var clusterMembers []PatroniClusterMember
   197  	var err error
   198  	var ok bool
   199  
   200  	switch source.HostConfig.DcsType {
   201  	case dcsTypeEtcd:
   202  		clusterMembers, err = getEtcdClusterMembers(source)
   203  	case dcsTypeZookeeper:
   204  		clusterMembers, err = getZookeeperClusterMembers(source)
   205  	case dcsTypeConsul:
   206  		clusterMembers, err = getConsulClusterMembers(source)
   207  	default:
   208  		return nil, errors.New("unknown DCS")
   209  	}
   210  	logger := logger.WithField("sorce", source.Name)
   211  	if err != nil {
   212  		if errors.Is(err, errors.ErrUnsupported) {
   213  			return nil, err
   214  		}
   215  		logger.Debug("Failed to get info from DCS, using previous member info if any")
   216  		if clusterMembers, ok = lastFoundClusterMembers[source.Name]; ok { // mask error from main loop not to remove monitored DBs due to "jitter"
   217  			err = nil
   218  		}
   219  	} else {
   220  		lastFoundClusterMembers[source.Name] = clusterMembers
   221  	}
   222  	if len(clusterMembers) == 0 {
   223  		return mds, err
   224  	}
   225  
   226  	for _, patroniMember := range clusterMembers {
   227  		logger.Info("Processing Patroni cluster member: ", patroniMember.Name)
   228  		if source.OnlyIfMaster && patroniMember.Role != "master" {
   229  			continue
   230  		}
   231  		src := *source.Clone()
   232  		src.ConnStr = patroniMember.ConnURL
   233  		if source.Kind == SourcePatroniNamespace {
   234  			src.Name += "_" + patroniMember.Scope
   235  		}
   236  		src.Name += "_" + patroniMember.Name
   237  		if dbs, err := ResolveDatabasesFromPostgres(src); err == nil {
   238  			mds = append(mds, dbs...)
   239  		} else {
   240  			logger.WithError(err).Error("Failed to resolve databases for Patroni member: ", patroniMember.Name)
   241  		}
   242  	}
   243  	return mds, err
   244  }

   246  // ResolveDatabasesFromPostgres reads all the databases from the given cluster,
   247  // additionally matching/not matching specified regex patterns
   248  func ResolveDatabasesFromPostgres(s Source) (resolvedDbs SourceConns, err error) {
   249  	var (
   250  		c      db.PgxPoolIface
   251  		dbname string
   252  		rows   pgx.Rows
   253  	)
   254  	c, err = NewConn(context.TODO(), s.ConnStr)
   255  	if err != nil {
   256  		return
   257  	}
   258  	defer c.Close()
   259  
   260  	sql := `select /* pgwatch_generated */
   261  	datname
   262  	from pg_database
   263  	where not datistemplate
   264  	and datallowconn
   265  	and has_database_privilege (datname, 'CONNECT')
   266  	and case when length(trim($1)) > 0 then datname ~ $1 else true end
   267  	and case when length(trim($2)) > 0 then not datname ~ $2 else true end`
   268  
   269  	if rows, err = c.Query(context.TODO(), sql, s.IncludePattern, s.ExcludePattern); err != nil {
   270  		return nil, err
   271  	}
   272  	for rows.Next() {
   273  		if err = rows.Scan(&dbname); err != nil {
   274  			return nil, err
   275  		}
   276  		rdb := &SourceConn{Source: *s.Clone()}
   277  		rdb.Name += "_" + dbname
   278  		rdb.SetDatabaseName(dbname)
   279  		resolvedDbs = append(resolvedDbs, rdb)
   280  	}
   281  
   282  	if err := rows.Err(); err != nil {
   283  		return nil, err
   284  	}
   285  	return
   286  }
