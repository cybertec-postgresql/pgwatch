    97  	if err != nil {

    99  	}
   100  
   101  	addresses := [2]string{PlainServerAddress, TLSServerAddress}
   102  	for _, address := range addresses {
   103  		lis, err := net.Listen("tcp", address)
   104  		if err != nil {
   105  			panic(err)
   106  		}
   107  
   108  		var creds credentials.TransportCredentials
   109  		if address == TLSServerAddress {
   110  			creds, err = LoadServerTLSCredentials()
   111  			if err != nil {
   112  				panic(err)
   113  			}
   114  		}
   115  
   116  		server := grpc.NewServer(
   117  			grpc.UnaryInterceptor(AuthInterceptor),
   118  			grpc.Creds(creds),
   119  		)
   120  
   121  		recv := new(Receiver)
   122  		pb.RegisterReceiverServer(server, recv)
   123  
   124  		go func() {
   125  			if err := server.Serve(lis); err != nil {
   126  				panic(err)
   127  			}
   128  		}()
   129  	}
   130  	// wait a little for servers start
   131  	time.Sleep(time.Second)
   132  
   133  	exitCode := m.Run()
   134  	_ = os.Remove(CAFile)
   135  
   136  	os.Exit(exitCode)
   137  }
