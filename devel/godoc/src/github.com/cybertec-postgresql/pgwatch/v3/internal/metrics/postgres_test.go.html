   165  	a := assert.New(t)

   171  
   172  	readerWriter, err := metrics.NewPostgresMetricReaderWriterConn(ctx, conn)
   173  	a.NoError(err)

   175  
   176  	metricsRows := func() *pgxmock.Rows {
   177  		return pgxmock.NewRows([]string{"name", "sqls", "init_sql", "description", "node_status", "gauges", "is_instance_level", "storage_name"}).
   178  			AddRow("test", metrics.SQLs{11: "select"}, "init", "desc", "primary", []string{"*"}, true, "storage")
   179  	}
   180  	presetRows := func() *pgxmock.Rows {
   181  		return pgxmock.NewRows([]string{"name", "description", "metrics"}).
   182  			AddRow("test", "desc", map[string]float64{"metric": 30})
   183  	}
   184  
   185  	t.Run("GetMetrics", func(*testing.T) {
   186  		conn.ExpectQuery(`SELECT.+FROM.+metric`).WillReturnRows(metricsRows())
   187  		conn.ExpectQuery(`SELECT.+FROM.+preset`).WillReturnRows(presetRows())

   189  		m, err := readerWriter.GetMetrics()
