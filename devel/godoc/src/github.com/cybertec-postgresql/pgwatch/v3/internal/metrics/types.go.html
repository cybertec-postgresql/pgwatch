    52  func (m Metric) GetSQL(version int) string {
    53  	// Check if there's an exact match for i
    54  	if val, ok := m.SQLs[version]; ok {
    55  		return val
    56  	}
    57  
    58  	// Find the closest value less than version
    59  	var closestVersion int
    60  	for v := range m.SQLs {
    61  		if v < version && (closestVersion == 0 || v > closestVersion) {
    62  			closestVersion = v
    63  		}
    64  	}
    65  	return m.SQLs[closestVersion]
