    39  	// Source represents a configuration how to get databases to monitor. It can be a single database,
    40  	// a group of databases in postgres cluster, a group of databases in HA patroni cluster.
    41  	// pgbouncer and pgpool kinds are purely to indicate that the monitored database connection is made
    42  	// through a connection pooler, which supports its own additional metrics. If one is not interested in
    43  	// those additional metrics, it is ok to specify the connection details as a regular postgres source.
    44  	Source struct {
    45  		Name                 string             `yaml:"name" db:"name"`
    46  		Group                string             `yaml:"group" db:"group"`

    64  func (srcs Sources) Validate() (Sources, error) {
    65  	names := map[string]any{}
    66  	for _, src := range srcs {
    67  		if _, ok := names[src.Name]; ok {
    68  			return nil, fmt.Errorf("duplicate source with name '%s' found", src.Name)
    69  		}
    70  		names[src.Name] = nil
    71  		if src.Kind == "" {
    72  			src.Kind = SourcePostgres
    73  		}
    74  	}
    75  	return srcs, nil
    76  }
