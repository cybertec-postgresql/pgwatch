    69  func (srcs Sources) Validate() (Sources, error) {
    70  	names := map[string]any{}
    71  	for _, src := range srcs {
    72  		if _, ok := names[src.Name]; ok {
    73  			return nil, fmt.Errorf("duplicate source with name '%s' found", src.Name)
    74  		}
    75  		names[src.Name] = nil
    76  		if src.Kind == "" {
    77  			src.Kind = SourcePostgres
    78  		}
    79  	}
    80  	return srcs, nil

   132  // Connect will establish a connection to the database if it's not already connected.
   133  // If the connection is already established, it pings the server to ensure it's still alive.
   134  func (md *MonitoredDatabase) Connect(ctx context.Context, opts CmdOpts) (err error) {
   135  	if md.Conn == nil {
   136  		if err = md.ParseConfig(); err != nil {
   137  			return err
   138  		}
   139  		if md.Kind == SourcePgBouncer {
   140  			md.ConnConfig.ConnConfig.DefaultQueryExecMode = pgx.QueryExecModeSimpleProtocol
   141  		}
   142  		if opts.MaxParallelConnectionsPerDb > 0 {
   143  			md.ConnConfig.MaxConns = int32(opts.MaxParallelConnectionsPerDb)
   144  		}
   145  		md.Conn, err = db.NewWithConfig(ctx, md.ConnConfig)
   146  		if err != nil {
   147  			return err
   148  		}
   149  	}
   150  	return md.ping(ctx)
   151  }

   192  // SyncFromReader will update the monitored databases with the latest configuration from the reader.
   193  // Any resolution errors will be returned, e.g. etcd unavailability.
   194  // It's up to the caller to proceed with the databases available or stop the execution due to errors.
   195  func (mds MonitoredDatabases) SyncFromReader(r Reader) (newmds MonitoredDatabases, err error) {
   196  	srcs, err := r.GetSources()
   197  	if err != nil {
   198  		return nil, err
   199  	}
   200  	newmds, err = srcs.ResolveDatabases()
   201  	for _, newMD := range newmds {
   202  		md := mds.GetMonitoredDatabase(newMD.Name)
   203  		if md == nil {
   204  			continue
   205  		}
   206  		if reflect.DeepEqual(md.Source, newMD.Source) {
   207  			// keep the existing connection if the source is the same
   208  			newMD.Conn = md.Conn
   209  			newMD.ConnConfig = md.ConnConfig
   210  			continue
   211  		}
   212  		if md.Conn != nil {
   213  			md.Conn.Close()
   214  		}
   215  	}
   216  	return newmds, err
   217  }
