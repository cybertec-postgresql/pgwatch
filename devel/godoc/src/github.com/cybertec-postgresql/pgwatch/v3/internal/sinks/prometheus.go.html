    34  func NewPrometheusWriter(ctx context.Context, connstr string) (promw *PrometheusWriter, err error) {
    35  	addr, namespace, found := strings.Cut(connstr, "/")
    36  	if !found {
    37  		namespace = "pgwatch"
    38  	}
    39  	l := log.GetLogger(ctx).WithField("sink", "prometheus").WithField("address", addr)
    40  	ctx = log.WithLogger(ctx, l)
    41  	promw = &PrometheusWriter{
    42  		ctx:                 ctx,
    43  		PrometheusNamespace: namespace,
    44  		lastScrapeErrors: prometheus.NewGauge(prometheus.GaugeOpts{
    45  			Namespace: namespace,
    46  			Name:      "exporter_last_scrape_errors",
    47  			Help:      "Last scrape error count for all monitored hosts / metrics",
    48  		}),
    49  		totalScrapes: prometheus.NewCounter(prometheus.CounterOpts{
    50  			Namespace: namespace,
    51  			Name:      "exporter_total_scrapes",
    52  			Help:      "Total scrape attempts.",
    53  		}),
    54  		totalScrapeFailures: prometheus.NewCounter(prometheus.CounterOpts{
    55  			Namespace: namespace,
    56  			Name:      "exporter_total_scrape_failures",
    57  			Help:      "Number of errors while executing metric queries",
    58  		}),
    59  	}
    60  
    61  	if err = prometheus.Register(promw); err != nil {
    62  		return
    63  	}
    64  	promServer := &http.Server{
    65  		Addr:    addr,
    66  		Handler: promhttp.Handler(),
    67  	}
    68  
    69  	ln, err := net.Listen("tcp", promServer.Addr)
    70  	if err != nil {
    71  		return nil, err
    72  	}
    73  
    74  	go func() { log.GetLogger(ctx).Error(promServer.Serve(ln)) }()
    75  
    76  	l.Info(`measurements sink is activated`)
    77  	return
    78  }

    81  	if len(msg.Data) == 0 { // no batching in async prom mode, so using 0 indexing ok

   133  func (promw *PrometheusWriter) Collect(ch chan<- prometheus.Metric) {
   134  	var lastScrapeErrors float64
   135  	logger := log.GetLogger(promw.ctx)
   136  	promw.totalScrapes.Add(1)
   137  	ch <- promw.totalScrapes
   138  
   139  	if len(promAsyncMetricCache) == 0 {
   140  		logger.Warning("No dbs configured for monitoring. Check config")
   141  		ch <- promw.totalScrapeFailures
   142  		promw.lastScrapeErrors.Set(0)
   143  		ch <- promw.lastScrapeErrors
   144  		return
   145  	}
   146  
   147  	for dbname, metricsMessages := range promAsyncMetricCache {
   148  		promw.setInstanceUpDownState(ch, dbname)
   149  		for metric, metricMessages := range metricsMessages {
   150  			if metric == "change_events" {
   151  				continue // not supported
   152  			}
   153  			promMetrics := promw.MetricStoreMessageToPromMetrics(metricMessages)
   154  			for _, pm := range promMetrics { // collect & send later in batch? capMetricChan = 1000 limit in prometheus code
   155  				ch <- pm
   156  			}
   157  		}
   158  
   159  	}
   160  
   161  	ch <- promw.totalScrapeFailures
   162  	promw.lastScrapeErrors.Set(lastScrapeErrors)
   163  	ch <- promw.lastScrapeErrors
   164  }

   166  func (promw *PrometheusWriter) setInstanceUpDownState(ch chan<- prometheus.Metric, dbName string) {
   167  	logger := log.GetLogger(promw.ctx)
   168  	data := metrics.NewMeasurement(time.Now().UnixNano())
   169  	data[promInstanceUpStateMetric] = 1
   170  
   171  	pm := promw.MetricStoreMessageToPromMetrics(metrics.MeasurementEnvelope{
   172  		DBName:           dbName,
   173  		SourceType:       "postgres",
   174  		MetricName:       promInstanceUpStateMetric,
   175  		CustomTags:       nil, //md.CustomTags,
   176  		Data:             metrics.Measurements{data},
   177  		RealDbname:       dbName, //vme.RealDbname,
   178  		SystemIdentifier: dbName, //vme.SystemIdentifier,
   179  	})
   180  
   181  	if len(pm) > 0 {
   182  		ch <- pm[0]
   183  	} else {
   184  		logger.Error("Could not formulate an instance state report - should not happen")
   185  	}
   186  }

   188  func (promw *PrometheusWriter) MetricStoreMessageToPromMetrics(msg metrics.MeasurementEnvelope) []prometheus.Metric {
   189  	promMetrics := make([]prometheus.Metric, 0)
   190  	logger := log.GetLogger(promw.ctx)
   191  	var epochTime time.Time
   192  
   193  	if len(msg.Data) == 0 {
   194  		return promMetrics
   195  	}
   196  
   197  	epochTime = time.Unix(0, msg.Data.GetEpoch())
   198  
   199  	if epochTime.Before(time.Now().Add(-promScrapingStalenessHardDropLimit)) {
   200  		logger.Warningf("[%s][%s] Dropping metric set due to staleness (>%v) ...", msg.DBName, msg.MetricName, promScrapingStalenessHardDropLimit)
   201  		promw.PurgeMetricsFromPromAsyncCacheIfAny(msg.DBName, msg.MetricName)
   202  		return promMetrics
   203  	}
   204  
   205  	for _, dr := range msg.Data {
   206  		labels := make(map[string]string)
   207  		fields := make(map[string]float64)
   208  		labels["dbname"] = msg.DBName
   209  
   210  		for k, v := range dr {
   211  			if v == nil || v == "" || k == metrics.EpochColumnName {
   212  				continue // not storing NULLs. epoch checked/assigned once
   213  			}
   214  
   215  			if strings.HasPrefix(k, "tag_") {
   216  				tag := k[4:]
   217  				labels[tag] = fmt.Sprintf("%v", v)
   218  			} else {
   219  				switch t := v.(type) {
   220  				case int, int32, int64, float32, float64:
   221  					f, err := strconv.ParseFloat(fmt.Sprintf("%v", v), 64)
   222  					if err != nil {
   223  						logger.Debugf("skipping scraping column %s of [%s:%s]: %v", k, msg.DBName, msg.MetricName, err)
   224  					}
   225  					fields[k] = f
   226  				case bool:
   227  					fields[k] = map[bool]float64{true: 1, false: 0}[t]
   228  				default:
   229  					logger.Debugf("skipping scraping column %s of [%s:%s], unsupported datatype: %v", k, msg.DBName, msg.MetricName, t)
   230  					continue
   231  				}
   232  			}
   233  		}
   234  		if msg.CustomTags != nil {
   235  			for k, v := range msg.CustomTags {
   236  				labels[k] = fmt.Sprintf("%v", v)
   237  			}
   238  		}
   239  
   240  		labelKeys := make([]string, 0)
   241  		labelValues := make([]string, 0)
   242  		for k, v := range labels {
   243  			labelKeys = append(labelKeys, k)
   244  			labelValues = append(labelValues, v)
   245  		}
   246  
   247  		for field, value := range fields {
   248  			fieldPromDataType := prometheus.CounterValue
   249  			if msg.MetricName == promInstanceUpStateMetric ||
   250  				len(msg.MetricDef.Gauges) > 0 &&
   251  					(msg.MetricDef.Gauges[0] == "*" || slices.Contains(msg.MetricDef.Gauges, field)) {
   252  				fieldPromDataType = prometheus.GaugeValue
   253  			}
   254  			var desc *prometheus.Desc
   255  			if promw.PrometheusNamespace != "" {
   256  				if msg.MetricName == promInstanceUpStateMetric { // handle the special "instance_up" check
   257  					desc = prometheus.NewDesc(fmt.Sprintf("%s_%s", promw.PrometheusNamespace, msg.MetricName),
   258  						msg.MetricName, labelKeys, nil)
   259  				} else {
   260  					desc = prometheus.NewDesc(fmt.Sprintf("%s_%s_%s", promw.PrometheusNamespace, msg.MetricName, field),
   261  						msg.MetricName, labelKeys, nil)
   262  				}
   263  			} else {
   264  				if msg.MetricName == promInstanceUpStateMetric { // handle the special "instance_up" check
   265  					desc = prometheus.NewDesc(field, msg.MetricName, labelKeys, nil)
   266  				} else {
   267  					desc = prometheus.NewDesc(fmt.Sprintf("%s_%s", msg.MetricName, field), msg.MetricName, labelKeys, nil)
   268  				}
   269  			}
   270  			m := prometheus.MustNewConstMetric(desc, fieldPromDataType, value, labelValues...)
   271  			promMetrics = append(promMetrics, prometheus.NewMetricWithTimestamp(epochTime, m))
   272  		}
   273  	}
   274  	return promMetrics
   275  }
