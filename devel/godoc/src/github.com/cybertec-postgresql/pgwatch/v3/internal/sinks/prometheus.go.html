    34  func NewPrometheusWriter(ctx context.Context, connstr string) (promw *PrometheusWriter, err error) {
    35  	addr, namespace, found := strings.Cut(connstr, "/")
    36  	if !found {
    37  		namespace = "pgwatch"
    38  	}
    39  	l := log.GetLogger(ctx).WithField("sink", "prometheus").WithField("address", addr)
    40  	ctx = log.WithLogger(ctx, l)
    41  	promw = &PrometheusWriter{
    42  		ctx:                 ctx,
    43  		PrometheusNamespace: namespace,
    44  		lastScrapeErrors: prometheus.NewGauge(prometheus.GaugeOpts{
    45  			Namespace: namespace,
    46  			Name:      "exporter_last_scrape_errors",
    47  			Help:      "Last scrape error count for all monitored hosts / metrics",
    48  		}),
    49  		totalScrapes: prometheus.NewCounter(prometheus.CounterOpts{
    50  			Namespace: namespace,
    51  			Name:      "exporter_total_scrapes",
    52  			Help:      "Total scrape attempts.",
    53  		}),
    54  		totalScrapeFailures: prometheus.NewCounter(prometheus.CounterOpts{
    55  			Namespace: namespace,
    56  			Name:      "exporter_total_scrape_failures",
    57  			Help:      "Number of errors while executing metric queries",
    58  		}),
    59  	}
    60  
    61  	if err = prometheus.Register(promw); err != nil {
    62  		return
    63  	}
    64  	promServer := &http.Server{
    65  		Addr:    addr,
    66  		Handler: promhttp.Handler(),
    67  	}
    68  
    69  	ln, err := net.Listen("tcp", promServer.Addr)
    70  	if err != nil {
    71  		return nil, err
    72  	}
    73  
    74  	go func() { log.GetLogger(ctx).Error(promServer.Serve(ln)) }()
    75  
    76  	l.Info(`measurements sink is activated`)
    77  	return
    78  }

    80  func (promw *PrometheusWriter) Write(msgs []metrics.MeasurementEnvelope) error {
    81  	if len(msgs) == 0 || len(msgs[0].Data) == 0 { // no batching in async prom mode, so using 0 indexing ok
    82  		return nil
    83  	}
    84  	msg := msgs[0]
    85  	promw.PromAsyncCacheAddMetricData(msg.DBName, msg.MetricName, msgs)
    86  	return nil

   134  func (promw *PrometheusWriter) Collect(ch chan<- prometheus.Metric) {
   135  	var lastScrapeErrors float64
   136  	logger := log.GetLogger(promw.ctx)
   137  	promw.totalScrapes.Add(1)
   138  	ch <- promw.totalScrapes
   139  
   140  	if len(promAsyncMetricCache) == 0 {
   141  		logger.Warning("No dbs configured for monitoring. Check config")
   142  		ch <- promw.totalScrapeFailures
   143  		promw.lastScrapeErrors.Set(0)
   144  		ch <- promw.lastScrapeErrors
   145  		return
   146  	}
   147  
   148  	for dbname, metricsMessages := range promAsyncMetricCache {
   149  		promw.setInstanceUpDownState(ch, dbname)
   150  		for metric, metricMessages := range metricsMessages {
   151  			if metric == "change_events" {
   152  				continue // not supported
   153  			}
   154  			if len(metricMessages) > 0 {
   155  				promMetrics := promw.MetricStoreMessageToPromMetrics(metricMessages[0])
   156  				for _, pm := range promMetrics { // collect & send later in batch? capMetricChan = 1000 limit in prometheus code
   157  					ch <- pm
   158  				}
   159  			}
   160  		}
   161  
   162  	}
   163  
   164  	ch <- promw.totalScrapeFailures
   165  	promw.lastScrapeErrors.Set(lastScrapeErrors)
   166  	ch <- promw.lastScrapeErrors
   167  
   168  	// atomic.StoreInt64(&lastSuccessfulDatastoreWriteTimeEpoch, time.Now().Unix())
   169  }

   171  func (promw *PrometheusWriter) setInstanceUpDownState(ch chan<- prometheus.Metric, dbName string) {
   172  	logger := log.GetLogger(promw.ctx)
   173  	data := make(metrics.Measurement)
   174  	data[promInstanceUpStateMetric] = 1
   175  	data[epochColumnName] = time.Now().UnixNano()
   176  
   177  	pm := promw.MetricStoreMessageToPromMetrics(metrics.MeasurementEnvelope{
   178  		DBName:           dbName,
   179  		SourceType:       "postgres",
   180  		MetricName:       promInstanceUpStateMetric,
   181  		CustomTags:       nil, //md.CustomTags,
   182  		Data:             metrics.Measurements{data},
   183  		RealDbname:       dbName, //vme.RealDbname,
   184  		SystemIdentifier: dbName, //vme.SystemIdentifier,
   185  	})
   186  
   187  	if len(pm) > 0 {
   188  		ch <- pm[0]
   189  	} else {
   190  		logger.Error("Could not formulate an instance state report - should not happen")
   191  	}

   194  func (promw *PrometheusWriter) MetricStoreMessageToPromMetrics(msg metrics.MeasurementEnvelope) []prometheus.Metric {
   195  	promMetrics := make([]prometheus.Metric, 0)
   196  	logger := log.GetLogger(promw.ctx)
   197  	var epochTime time.Time
   198  	var epochNs int64
   199  	var epochNow = time.Now()
   200  
   201  	if len(msg.Data) == 0 {
   202  		return promMetrics
   203  	}
   204  
   205  	epochNs, ok := (msg.Data[0][epochColumnName]).(int64)
   206  	if !ok {
   207  		epochTime = time.Now()
   208  	} else {
   209  		epochTime = time.Unix(0, epochNs)
   210  
   211  		if epochTime.Before(epochNow.Add(-1 * promScrapingStalenessHardDropLimit)) {
   212  			logger.Warningf("[%s][%s] Dropping metric set due to staleness (>%v) ...", msg.DBName, msg.MetricName, promScrapingStalenessHardDropLimit)
   213  			promw.PurgeMetricsFromPromAsyncCacheIfAny(msg.DBName, msg.MetricName)
   214  			return promMetrics
   215  		}
   216  	}
   217  
   218  	for _, dr := range msg.Data {
   219  		labels := make(map[string]string)
   220  		fields := make(map[string]float64)
   221  		labels["dbname"] = msg.DBName
   222  
   223  		for k, v := range dr {
   224  			if v == nil || v == "" || k == epochColumnName {
   225  				continue // not storing NULLs. epoch checked/assigned once
   226  			}
   227  
   228  			if strings.HasPrefix(k, "tag_") {
   229  				tag := k[4:]
   230  				labels[tag] = fmt.Sprintf("%v", v)
   231  			} else {
   232  				switch t := v.(type) {
   233  				case int, int32, int64, float32, float64:
   234  					f, err := strconv.ParseFloat(fmt.Sprintf("%v", v), 64)
   235  					if err != nil {
   236  						logger.Debugf("skipping scraping column %s of [%s:%s]: %v", k, msg.DBName, msg.MetricName, err)
   237  					}
   238  					fields[k] = f
   239  				case bool:
   240  					fields[k] = map[bool]float64{true: 1, false: 0}[t]
   241  				default:
   242  					logger.Debugf("skipping scraping column %s of [%s:%s], unsupported datatype: %v", k, msg.DBName, msg.MetricName, t)
   243  					continue
   244  				}
   245  			}
   246  		}
   247  		if msg.CustomTags != nil {
   248  			for k, v := range msg.CustomTags {
   249  				labels[k] = fmt.Sprintf("%v", v)
   250  			}
   251  		}
   252  
   253  		labelKeys := make([]string, 0)
   254  		labelValues := make([]string, 0)
   255  		for k, v := range labels {
   256  			labelKeys = append(labelKeys, k)
   257  			labelValues = append(labelValues, v)
   258  		}
   259  
   260  		for field, value := range fields {
   261  			fieldPromDataType := prometheus.CounterValue
   262  			if msg.MetricName == promInstanceUpStateMetric ||
   263  				len(msg.MetricDef.Gauges) > 0 &&
   264  					(msg.MetricDef.Gauges[0] == "*" || slices.Contains(msg.MetricDef.Gauges, field)) {
   265  				fieldPromDataType = prometheus.GaugeValue
   266  			}
   267  			var desc *prometheus.Desc
   268  			if promw.PrometheusNamespace != "" {
   269  				if msg.MetricName == promInstanceUpStateMetric { // handle the special "instance_up" check
   270  					desc = prometheus.NewDesc(fmt.Sprintf("%s_%s", promw.PrometheusNamespace, msg.MetricName),
   271  						msg.MetricName, labelKeys, nil)
   272  				} else {
   273  					desc = prometheus.NewDesc(fmt.Sprintf("%s_%s_%s", promw.PrometheusNamespace, msg.MetricName, field),
   274  						msg.MetricName, labelKeys, nil)
   275  				}
   276  			} else {
   277  				if msg.MetricName == promInstanceUpStateMetric { // handle the special "instance_up" check
   278  					desc = prometheus.NewDesc(field, msg.MetricName, labelKeys, nil)
   279  				} else {
   280  					desc = prometheus.NewDesc(fmt.Sprintf("%s_%s", msg.MetricName, field), msg.MetricName, labelKeys, nil)
   281  				}
   282  			}
   283  			m := prometheus.MustNewConstMetric(desc, fieldPromDataType, value, labelValues...)
   284  			promMetrics = append(promMetrics, prometheus.NewMetricWithTimestamp(epochTime, m))
   285  		}
   286  	}
   287  	return promMetrics
   288  }
