   187  		return pgw.ctx.Err()

   190  	case pgw.input <- msgs:
   191  		// msgs sent

   194  	}

   196  	case err := <-pgw.lastError:
   197  		return err
   198  	default:
   199  		return nil

   203  // poll is the main loop that reads from the input channel and flushes the data to the database
   204  func (pgw *PostgresWriter) poll() {
   205  	cache := make([]metrics.MeasurementEnvelope, 0, cacheLimit)
   206  	cacheTimeout := pgw.opts.BatchingDelay
   207  	tick := time.NewTicker(cacheTimeout)
   208  	for {
   209  		select {
   210  		case <-pgw.ctx.Done(): //check context with high priority
   211  			return
   212  		default:
   213  			select {
   214  			case entry := <-pgw.input:
   215  				cache = append(cache, entry...)
   216  				if len(cache) < cacheLimit {
   217  					break
   218  				}
   219  				tick.Stop()
   220  				pgw.flush(cache)
   221  				cache = cache[:0]
   222  				tick = time.NewTicker(cacheTimeout)
   223  			case <-tick.C:
   224  				pgw.flush(cache)
   225  				cache = cache[:0]
   226  			case <-pgw.ctx.Done():
   227  				return
   228  			}
   229  		}
   230  	}
   231  }

   233  // flush sends the cached measurements to the database
   234  func (pgw *PostgresWriter) flush(msgs []metrics.MeasurementEnvelope) {
   235  	if len(msgs) == 0 {
   236  		return
   237  	}
   238  	logger := log.GetLogger(pgw.ctx)
   239  	tsWarningPrinted := false
   240  	metricsToStorePerMetric := make(map[string][]MeasurementMessagePostgres)
   241  	rowsBatched := 0
   242  	totalRows := 0
   243  	pgPartBounds := make(map[string]ExistingPartitionInfo)                  // metric=min/max
   244  	pgPartBoundsDbName := make(map[string]map[string]ExistingPartitionInfo) // metric=[dbname=min/max]
   245  	var err error
   246  
   247  	for _, msg := range msgs {
   248  		if len(msg.Data) == 0 {
   249  			continue
   250  		}
   251  		logger.WithField("data", msg.Data).WithField("len", len(msg.Data)).Debug("sending to postgres")
   252  
   253  		for _, dataRow := range msg.Data {
   254  			var epochTime time.Time
   255  			var epochNs int64
   256  
   257  			tags := make(map[string]any)
   258  			fields := make(map[string]any)
   259  
   260  			totalRows++
   261  
   262  			if msg.CustomTags != nil {
   263  				for k, v := range msg.CustomTags {
   264  					tags[k] = fmt.Sprintf("%v", v)
   265  				}
   266  			}
   267  
   268  			for k, v := range dataRow {
   269  				if v == nil || v == "" {
   270  					continue // not storing NULLs
   271  				}
   272  				if k == epochColumnName {
   273  					epochNs = v.(int64)
   274  				} else if strings.HasPrefix(k, tagPrefix) {
   275  					tag := k[4:]
   276  					tags[tag] = fmt.Sprintf("%v", v)
   277  				} else {
   278  					fields[k] = v
   279  				}
   280  			}
   281  
   282  			if epochNs == 0 {
   283  				if !tsWarningPrinted && !regexIsPgbouncerMetrics.MatchString(msg.MetricName) {
   284  					logger.Warning("No timestamp_ns found, server time will be used. measurement:", msg.MetricName)
   285  					tsWarningPrinted = true
   286  				}
   287  				epochTime = time.Now()
   288  			} else {
   289  				epochTime = time.Unix(0, epochNs)
   290  			}
   291  
   292  			var metricsArr []MeasurementMessagePostgres
   293  			var ok bool
   294  
   295  			metricNameTemp := msg.MetricName
   296  
   297  			metricsArr, ok = metricsToStorePerMetric[metricNameTemp]
   298  			if !ok {
   299  				metricsToStorePerMetric[metricNameTemp] = make([]MeasurementMessagePostgres, 0)
   300  			}
   301  			metricsArr = append(metricsArr, MeasurementMessagePostgres{Time: epochTime, DBName: msg.DBName,
   302  				Metric: msg.MetricName, Data: fields, TagData: tags})
   303  			metricsToStorePerMetric[metricNameTemp] = metricsArr
   304  
   305  			rowsBatched++
   306  
   307  			if pgw.metricSchema == DbStorageSchemaTimescale {
   308  				// set min/max timestamps to check/create partitions
   309  				bounds, ok := pgPartBounds[msg.MetricName]
   310  				if !ok || (ok && epochTime.Before(bounds.StartTime)) {
   311  					bounds.StartTime = epochTime
   312  					pgPartBounds[msg.MetricName] = bounds
   313  				}
   314  				if !ok || (ok && epochTime.After(bounds.EndTime)) {
   315  					bounds.EndTime = epochTime
   316  					pgPartBounds[msg.MetricName] = bounds
   317  				}
   318  			} else if pgw.metricSchema == DbStorageSchemaPostgres {
   319  				_, ok := pgPartBoundsDbName[msg.MetricName]
   320  				if !ok {
   321  					pgPartBoundsDbName[msg.MetricName] = make(map[string]ExistingPartitionInfo)
   322  				}
   323  				bounds, ok := pgPartBoundsDbName[msg.MetricName][msg.DBName]
   324  				if !ok || (ok && epochTime.Before(bounds.StartTime)) {
   325  					bounds.StartTime = epochTime
   326  					pgPartBoundsDbName[msg.MetricName][msg.DBName] = bounds
   327  				}
   328  				if !ok || (ok && epochTime.After(bounds.EndTime)) {
   329  					bounds.EndTime = epochTime
   330  					pgPartBoundsDbName[msg.MetricName][msg.DBName] = bounds
   331  				}
   332  			}
   333  		}
   334  	}
   335  
   336  	switch pgw.metricSchema {
   337  	case DbStorageSchemaPostgres:
   338  		err = pgw.EnsureMetricDbnameTime(pgPartBoundsDbName, forceRecreatePartitions)
   339  	case DbStorageSchemaTimescale:
   340  		err = pgw.EnsureMetricTimescale(pgPartBounds, forceRecreatePartitions)
   341  	default:
   342  		logger.Fatal("unknown storage schema...")
   343  	}
   344  	if forceRecreatePartitions {
   345  		forceRecreatePartitions = false
   346  	}
   347  	if err != nil {
   348  		pgw.lastError <- err
   349  	}
   350  
   351  	// send data to PG, with a separate COPY for all metrics
   352  	logger.Debugf("COPY-ing %d metrics to Postgres metricsDB...", rowsBatched)
   353  	t1 := time.Now()
   354  
   355  	for metricName, metrics := range metricsToStorePerMetric {
   356  
   357  		getTargetTable := func() pgx.Identifier {
   358  			return pgx.Identifier{metricName}
   359  		}
   360  
   361  		getTargetColumns := func() []string {
   362  			return []string{"time", "dbname", "data", "tag_data"}
   363  		}
   364  
   365  		for _, m := range metrics {
   366  			l := logger.WithField("db", m.DBName).WithField("metric", m.Metric)
   367  			jsonBytes, err := json.Marshal(m.Data)
   368  			if err != nil {
   369  				logger.Errorf("Skipping 1 metric for [%s:%s] due to JSON conversion error: %s", m.DBName, m.Metric, err)
   370  				continue
   371  			}
   372  
   373  			getTagData := func() any {
   374  				if len(m.TagData) > 0 {
   375  					jsonBytesTags, err := json.Marshal(m.TagData)
   376  					if err != nil {
   377  						l.Error(err)
   378  						return nil
   379  					}
   380  					return string(jsonBytesTags)
   381  				}
   382  				return nil
   383  			}
   384  
   385  			rows := [][]any{{m.Time, m.DBName, string(jsonBytes), getTagData()}}
   386  
   387  			if _, err = pgw.sinkDb.CopyFrom(context.Background(), getTargetTable(), getTargetColumns(), pgx.CopyFromRows(rows)); err != nil {
   388  				l.Error(err)
   389  				forceRecreatePartitions = strings.Contains(err.Error(), "no partition")
   390  				if forceRecreatePartitions {
   391  					logger.Warning("Some metric partitions might have been removed, halting all metric storage. Trying to re-create all needed partitions on next run")
   392  				}
   393  			}
   394  		}
   395  	}
   396  
   397  	diff := time.Since(t1)
   398  	if err == nil {
   399  		logger.WithField("rows", rowsBatched).WithField("elapsed", diff).Info("measurements written")
   400  		return
   401  	}
   402  	pgw.lastError <- err
   403  }

   405  // EnsureMetricTime creates special partitions if Timescale used for realtime metrics
   406  func (pgw *PostgresWriter) EnsureMetricTime(pgPartBounds map[string]ExistingPartitionInfo, force bool) error {
   407  	logger := log.GetLogger(pgw.ctx)
   408  	sqlEnsure := `select part_available_from, part_available_to from admin.ensure_partition_metric_time($1, $2)`
   409  	for metric, pb := range pgPartBounds {
   410  		if !strings.HasSuffix(metric, "_realtime") {
   411  			continue
   412  		}
   413  		if pb.StartTime.IsZero() || pb.EndTime.IsZero() {
   414  			return fmt.Errorf("zero StartTime/EndTime in partitioning request: [%s:%v]", metric, pb)
   415  		}
   416  
   417  		partInfo, ok := partitionMapMetric[metric]
   418  		if !ok || (ok && (pb.StartTime.Before(partInfo.StartTime))) || force {
   419  			err := pgw.sinkDb.QueryRow(pgw.ctx, sqlEnsure, metric, pb.StartTime).
   420  				Scan(&partInfo.StartTime, &partInfo.EndTime)
   421  			if err != nil {
   422  				logger.Error("Failed to create partition on 'metrics': ", err)
   423  				return err
   424  			}
   425  			partitionMapMetric[metric] = partInfo
   426  		}
   427  		if pb.EndTime.After(partInfo.EndTime) || force {
   428  			err := pgw.sinkDb.QueryRow(pgw.ctx, sqlEnsure, metric, pb.EndTime).Scan(nil, &partInfo.EndTime)
   429  			if err != nil {
   430  				logger.Error("Failed to create partition on 'metrics': ", err)
   431  				return err
   432  			}
   433  			partitionMapMetric[metric] = partInfo
   434  		}
   435  	}
   436  	return nil
   437  }

   439  func (pgw *PostgresWriter) EnsureMetricTimescale(pgPartBounds map[string]ExistingPartitionInfo, force bool) (err error) {
   440  	logger := log.GetLogger(pgw.ctx)
   441  	sqlEnsure := `select * from admin.ensure_partition_timescale($1)`
   442  	for metric := range pgPartBounds {
   443  		if strings.HasSuffix(metric, "_realtime") {
   444  			continue
   445  		}
   446  		if _, ok := partitionMapMetric[metric]; !ok {
   447  			if _, err = pgw.sinkDb.Exec(pgw.ctx, sqlEnsure, metric); err != nil {
   448  				logger.Errorf("Failed to create a TimescaleDB table for metric '%s': %v", metric, err)
   449  				return err
   450  			}
   451  			partitionMapMetric[metric] = ExistingPartitionInfo{}
   452  		}
   453  	}
   454  	return pgw.EnsureMetricTime(pgPartBounds, force)

   457  func (pgw *PostgresWriter) EnsureMetricDbnameTime(metricDbnamePartBounds map[string]map[string]ExistingPartitionInfo, force bool) (err error) {
   458  	var rows pgx.Rows
   459  	sqlEnsure := `select * from admin.ensure_partition_metric_dbname_time($1, $2, $3)`
   460  	for metric, dbnameTimestampMap := range metricDbnamePartBounds {
   461  		_, ok := partitionMapMetricDbname[metric]
   462  		if !ok {
   463  			partitionMapMetricDbname[metric] = make(map[string]ExistingPartitionInfo)
   464  		}
   465  
   466  		for dbname, pb := range dbnameTimestampMap {
   467  			if pb.StartTime.IsZero() || pb.EndTime.IsZero() {
   468  				return fmt.Errorf("zero StartTime/EndTime in partitioning request: [%s:%v]", metric, pb)
   469  			}
   470  			partInfo, ok := partitionMapMetricDbname[metric][dbname]
   471  			if !ok || (ok && (pb.StartTime.Before(partInfo.StartTime))) || force {
   472  				if rows, err = pgw.sinkDb.Query(pgw.ctx, sqlEnsure, metric, dbname, pb.StartTime); err != nil {
   473  					return
   474  				}
   475  				if partInfo, err = pgx.CollectOneRow(rows, pgx.RowToStructByPos[ExistingPartitionInfo]); err != nil {
   476  					return err
   477  				}
   478  				partitionMapMetricDbname[metric][dbname] = partInfo
   479  			}
   480  			if pb.EndTime.After(partInfo.EndTime) || pb.EndTime.Equal(partInfo.EndTime) || force {
   481  				if rows, err = pgw.sinkDb.Query(pgw.ctx, sqlEnsure, metric, dbname, pb.StartTime); err != nil {
   482  					return
   483  				}
   484  				if partInfo, err = pgx.CollectOneRow(rows, pgx.RowToStructByPos[ExistingPartitionInfo]); err != nil {
   485  					return err
   486  				}
   487  				partitionMapMetricDbname[metric][dbname] = partInfo
   488  			}
   489  		}
   490  	}
   491  	return nil

   494  // deleteOldPartitions is a background task that deletes old partitions from the measurements DB
   495  func (pgw *PostgresWriter) deleteOldPartitions(delay time.Duration) {
   496  	metricAgeDaysThreshold := pgw.opts.Retention
   497  	if metricAgeDaysThreshold <= 0 {
   498  		return
   499  	}
   500  	logger := log.GetLogger(pgw.ctx)
   501  	select {
   502  	case <-pgw.ctx.Done():
   503  		return
   504  	case <-time.After(delay):
   505  		// to reduce distracting log messages at startup
   506  	}
   507  
   508  	for {
   509  		if pgw.metricSchema == DbStorageSchemaTimescale {
   510  			partsDropped, err := pgw.DropOldTimePartitions(metricAgeDaysThreshold)
   511  			if err != nil {
   512  				logger.Errorf("Failed to drop old partitions (>%d days) from Postgres: %v", metricAgeDaysThreshold, err)
   513  				continue
   514  			}
   515  			logger.Infof("Dropped %d old metric partitions...", partsDropped)
   516  		} else if pgw.metricSchema == DbStorageSchemaPostgres {
   517  			partsToDrop, err := pgw.GetOldTimePartitions(metricAgeDaysThreshold)
   518  			if err != nil {
   519  				logger.Errorf("Failed to get a listing of old (>%d days) time partitions from Postgres metrics DB - check that the admin.get_old_time_partitions() function is rolled out: %v", metricAgeDaysThreshold, err)
   520  				time.Sleep(time.Second * 300)
   521  				continue
   522  			}
   523  			if len(partsToDrop) > 0 {
   524  				logger.Infof("Dropping %d old metric partitions one by one...", len(partsToDrop))
   525  				for _, toDrop := range partsToDrop {
   526  					sqlDropTable := `DROP TABLE IF EXISTS ` + toDrop
   527  					logger.Debugf("Dropping old metric data partition: %s", toDrop)
   528  
   529  					if _, err := pgw.sinkDb.Exec(pgw.ctx, sqlDropTable); err != nil {
   530  						logger.Errorf("Failed to drop old partition %s from Postgres metrics DB: %w", toDrop, err)
   531  						time.Sleep(time.Second * 300)
   532  					} else {
   533  						time.Sleep(time.Second * 5)
   534  					}
   535  				}
   536  			} else {

   548  // maintainUniqueSources is a background task that maintains a listing of unique sources for each metric.
   549  // This is used to avoid listing the same source multiple times in Grafana dropdowns.
   550  func (pgw *PostgresWriter) maintainUniqueSources() {
   551  	logger := log.GetLogger(pgw.ctx)
   552  	// due to metrics deletion the listing can go out of sync (a trigger not really wanted)
   553  	sqlGetAdvisoryLock := `SELECT pg_try_advisory_lock(1571543679778230000) AS have_lock` // 1571543679778230000 is just a random bigint
   554  	sqlTopLevelMetrics := `SELECT table_name FROM admin.get_top_level_metric_tables()`
   555  	sqlDistinct := `
   556  	WITH RECURSIVE t(dbname) AS (
   557  		SELECT MIN(dbname) AS dbname FROM %s
   558  		UNION
   559  		SELECT (SELECT MIN(dbname) FROM %s WHERE dbname > t.dbname) FROM t )
   560  	SELECT dbname FROM t WHERE dbname NOTNULL ORDER BY 1`
   561  	sqlDelete := `DELETE FROM admin.all_distinct_dbname_metrics WHERE NOT dbname = ANY($1) and metric = $2`
   562  	sqlDeleteAll := `DELETE FROM admin.all_distinct_dbname_metrics WHERE metric = $1`
   563  	sqlAdd := `
   564  		INSERT INTO admin.all_distinct_dbname_metrics SELECT u, $2 FROM (select unnest($1::text[]) as u) x
   565  		WHERE NOT EXISTS (select * from admin.all_distinct_dbname_metrics where dbname = u and metric = $2)
   566  		RETURNING *`
   567  
   568  	for {
   569  		select {
   570  		case <-pgw.ctx.Done():
   571  			return
   572  		case <-time.After(time.Hour * 24):
   573  		}
   574  		var lock bool
   575  		logger.Infof("Trying to get metricsDb listing maintainer advisory lock...") // to only have one "maintainer" in case of a "push" setup, as can get costly
   576  		if err := pgw.sinkDb.QueryRow(pgw.ctx, sqlGetAdvisoryLock).Scan(&lock); err != nil {
   577  			logger.Error("Getting metricsDb listing maintainer advisory lock failed:", err)
   578  			continue
   579  		}
   580  		if !lock {
   581  			logger.Info("Skipping admin.all_distinct_dbname_metrics maintenance as another instance has the advisory lock...")
   582  			continue
   583  		}
   584  
   585  		logger.Info("Refreshing admin.all_distinct_dbname_metrics listing table...")
   586  		rows, _ := pgw.sinkDb.Query(pgw.ctx, sqlTopLevelMetrics)
   587  		allDistinctMetricTables, err := pgx.CollectRows(rows, pgx.RowTo[string])
   588  		if err != nil {
   589  			logger.Error(err)
   590  			continue
   591  		}
   592  
   593  		for _, tableName := range allDistinctMetricTables {
   594  			foundDbnamesMap := make(map[string]bool)
   595  			foundDbnamesArr := make([]string, 0)
   596  			metricName := strings.Replace(tableName, "public.", "", 1)
   597  
   598  			logger.Debugf("Refreshing all_distinct_dbname_metrics listing for metric: %s", metricName)
   599  			rows, _ := pgw.sinkDb.Query(pgw.ctx, fmt.Sprintf(sqlDistinct, tableName, tableName))
   600  			ret, err := pgx.CollectRows(rows, pgx.RowTo[string])
   601  			// ret, err := DBExecRead(mainContext, metricDb, fmt.Sprintf(sqlDistinct, tableName, tableName))
   602  			if err != nil {
   603  				logger.Errorf("Could not refresh Postgres all_distinct_dbname_metrics listing table for '%s': %s", metricName, err)
   604  				break
   605  			}
   606  			for _, drDbname := range ret {
   607  				foundDbnamesMap[drDbname] = true // "set" behaviour, don't want duplicates
   608  			}
   609  
   610  			// delete all that are not known and add all that are not there
   611  			for k := range foundDbnamesMap {
   612  				foundDbnamesArr = append(foundDbnamesArr, k)
   613  			}
   614  			if len(foundDbnamesArr) == 0 { // delete all entries for given metric
   615  				logger.Debugf("Deleting Postgres all_distinct_dbname_metrics listing table entries for metric '%s':", metricName)
   616  
   617  				_, err = pgw.sinkDb.Exec(pgw.ctx, sqlDeleteAll, metricName)
   618  				if err != nil {
   619  					logger.Errorf("Could not delete Postgres all_distinct_dbname_metrics listing table entries for metric '%s': %s", metricName, err)
   620  				}
   621  				continue
   622  			}
   623  			cmdTag, err := pgw.sinkDb.Exec(pgw.ctx, sqlDelete, foundDbnamesArr, metricName)
   624  			if err != nil {
   625  				logger.Errorf("Could not refresh Postgres all_distinct_dbname_metrics listing table for metric '%s': %s", metricName, err)
   626  			} else if cmdTag.RowsAffected() > 0 {
   627  				logger.Infof("Removed %d stale entries from all_distinct_dbname_metrics listing table for metric: %s", cmdTag.RowsAffected(), metricName)
   628  			}
   629  			cmdTag, err = pgw.sinkDb.Exec(pgw.ctx, sqlAdd, foundDbnamesArr, metricName)
   630  			if err != nil {
   631  				logger.Errorf("Could not refresh Postgres all_distinct_dbname_metrics listing table for metric '%s': %s", metricName, err)
   632  			} else if cmdTag.RowsAffected() > 0 {
   633  				logger.Infof("Added %d entry to the Postgres all_distinct_dbname_metrics listing table for metric: %s", cmdTag.RowsAffected(), metricName)
   634  			}
   635  			time.Sleep(time.Minute)
   636  		}
   637  
   638  	}
   639  }
