    42  	for _, f := range files {

    55  func getFileWithNextModTimestamp(logsGlobPath, currentFile string) (string, error) {
    56  	var nextFile string
    57  	var nextMod time.Time
    58  
    59  	files, err := filepath.Glob(logsGlobPath)
    60  	if err != nil {
    61  		return "", err
    62  	}
    63  
    64  	fiCurrent, err := os.Stat(currentFile)
    65  	if err != nil {
    66  		return "", err
    67  	}
    68  
    69  	for _, f := range files {
    70  		if f == currentFile {
    71  			continue
    72  		}
    73  		fi, err := os.Stat(f)
    74  		if err != nil {
    75  			continue
    76  		}
    77  		//log.Debugf("Stat().ModTime() for %s: %v", f, fi.ModTime())
    78  		if (nextMod.IsZero() || fi.ModTime().Before(nextMod)) && fi.ModTime().After(fiCurrent.ModTime()) {
    79  			nextMod = fi.ModTime()
    80  			nextFile = f
    81  		}
    82  	}
    83  	return nextFile, nil

   113  func ParseLogs(ctx context.Context, mdb *sources.MonitoredDatabase, realDbname, metricName string, configMap map[string]float64, storeCh chan<- []MeasurementEnvelope) {
   114  
   115  	var latest, previous, serverMessagesLang string
   116  	var latestHandle *os.File
   117  	var reader *bufio.Reader
   118  	var linesRead = 0 // to skip over already parsed lines on Postgres server restart for example
   119  	var logsMatchRegex, logsMatchRegexPrev, logsGlobPath string
   120  	var lastSendTime time.Time                    // to storage channel
   121  	var eventCounts = make(map[string]int64)      // for the specific DB. [WARNING: 34, ERROR: 10, ...], zeroed on storage send
   122  	var eventCountsTotal = make(map[string]int64) // for the whole instance
   123  	var hostConfig sources.HostConfigAttrs
   124  	var config = configMap
   125  	var interval float64
   126  	var err error
   127  	var firstRun = true
   128  	var csvlogRegex *regexp.Regexp
   129  	logger := log.GetLogger(ctx).WithField("source", mdb.Name)
   130  	for { // re-try loop. re-start in case of FS errors or just to refresh host config
   131  		select {
   132  		case <-ctx.Done():
   133  			return
   134  		default:
   135  			if interval == 0 {
   136  				interval = config[metricName]
   137  			}
   138  		}
   139  
   140  		if hostConfig.LogsMatchRegex != "" {
   141  			logsMatchRegex = hostConfig.LogsMatchRegex
   142  		}
   143  		if logsMatchRegex == "" {
   144  			logger.Debug("Log parsing enabled with default CSVLOG regex")
   145  			logsMatchRegex = CSVLogDefaultRegEx
   146  		}
   147  		if hostConfig.LogsGlobPath != "" {
   148  			logsGlobPath = hostConfig.LogsGlobPath
   149  		}
   150  		if logsGlobPath == "" {
   151  			logsGlobPath, err = tryDetermineLogFolder(ctx, mdb.Conn)
   152  			if err != nil {
   153  				logger.WithError(err).Print("Could not determine Postgres logs parsing folder. Configured logs_glob_path = ", logsGlobPath)
   154  				time.Sleep(60 * time.Second)
   155  				continue
   156  			}
   157  		}
   158  		serverMessagesLang, err = tryDetermineLogMessagesLanguage(ctx, mdb.Conn)
   159  		if err != nil {
   160  			logger.WithError(err).Warning("Could not determine language (lc_collate) used for server logs, cannot parse logs...")
   161  			time.Sleep(60 * time.Second)
   162  			continue
   163  		}
   164  
   165  		if logsMatchRegexPrev != logsMatchRegex { // avoid regex recompile if no changes
   166  			csvlogRegex, err = regexp.Compile(logsMatchRegex)
   167  			if err != nil {
   168  				logger.WithError(err).Print("Invalid regex: ", logsMatchRegex)
   169  				time.Sleep(60 * time.Second)
   170  				continue
   171  			}
   172  			logger.Infof("Changing logs parsing regex to: %s", logsMatchRegex)
   173  			logsMatchRegexPrev = logsMatchRegex
   174  		}
   175  
   176  		logger.Debugf("Considering log files determined by glob pattern: %s", logsGlobPath)
   177  
   178  		if latest == "" || firstRun {
   179  			globMatches, err := filepath.Glob(logsGlobPath)
   180  			if err != nil || len(globMatches) == 0 {
   181  				logger.Infof("No logfiles found to parse from glob '%s'. Sleeping 60s...", logsGlobPath)
   182  				time.Sleep(60 * time.Second)
   183  				continue
   184  			}
   185  
   186  			logger.Debugf("Found %v logfiles from glob pattern, picking the latest", len(globMatches))
   187  			if len(globMatches) > 1 {
   188  				// find latest timestamp
   189  				latest, _ = getFileWithLatestTimestamp(globMatches)
   190  				if latest == "" {
   191  					logger.Warningf("Could not determine the latest logfile. Sleeping 60s...")
   192  					time.Sleep(60 * time.Second)
   193  					continue
   194  				}
   195  
   196  			} else if len(globMatches) == 1 {
   197  				latest = globMatches[0]
   198  			}
   199  			logger.Infof("Starting to parse logfile: %s ", latest)
   200  		}
   201  
   202  		if latestHandle == nil {
   203  			latestHandle, err = os.Open(latest)
   204  			if err != nil {
   205  				logger.Warningf("Failed to open logfile %s: %s. Sleeping 60s...", latest, err)
   206  				time.Sleep(60 * time.Second)
   207  				continue
   208  			}
   209  			reader = bufio.NewReader(latestHandle)
   210  			if previous == latest && linesRead > 0 { // handle postmaster restarts
   211  				i := 1
   212  				for i <= linesRead {
   213  					_, err = reader.ReadString('\n')
   214  					if err == io.EOF && i < linesRead {
   215  						logger.Warningf("Failed to open logfile %s: %s. Sleeping 60s...", latest, err)
   216  						linesRead = 0
   217  						break
   218  					} else if err != nil {
   219  						logger.Warningf("Failed to skip %d logfile lines for %s, there might be duplicates reported. Error: %s", linesRead, latest, err)
   220  						time.Sleep(60 * time.Second)
   221  						linesRead = i
   222  						break
   223  					}
   224  					i++
   225  				}
   226  				logger.Debugf("Skipped %d already processed lines from %s", linesRead, latest)
   227  			} else if firstRun { // seek to end
   228  				_, _ = latestHandle.Seek(0, 2)
   229  				firstRun = false
   230  			}
   231  		}
   232  
   233  		var eofSleepMillis = 0
   234  		// readLoopStart := time.Now()
   235  
   236  		for {
   237  			// if readLoopStart.Add(time.Second * time.Duration(opts.Source.Refresh)).Before(time.Now()) {
   238  			// 	break // refresh config
   239  			// }
   240  			line, err := reader.ReadString('\n')
   241  			if err != nil && err != io.EOF {
   242  				logger.Warningf("Failed to read logfile %s: %s. Sleeping 60s...", latest, err)
   243  				err = latestHandle.Close()
   244  				if err != nil {
   245  					logger.Warningf("Failed to close logfile %s properly: %s", latest, err)
   246  				}
   247  				latestHandle = nil
   248  				time.Sleep(60 * time.Second)
   249  				break
   250  			}
   251  
   252  			if err == io.EOF {
   253  				//log.Debugf("EOF reached for logfile %s", latest)
   254  				if eofSleepMillis < 5000 && float64(eofSleepMillis) < interval*1000 {
   255  					eofSleepMillis += 100 // progressively sleep more if nothing going on but not more that 5s or metric interval
   256  				}
   257  				time.Sleep(time.Millisecond * time.Duration(eofSleepMillis))
   258  
   259  				// check for newly opened logfiles
   260  				file, _ := getFileWithNextModTimestamp(logsGlobPath, latest)
   261  				if file != "" {
   262  					previous = latest
   263  					latest = file
   264  					err = latestHandle.Close()
   265  					latestHandle = nil
   266  					if err != nil {
   267  						logger.Warningf("Failed to close logfile %s properly: %s", latest, err)
   268  					}
   269  					logger.Infof("Switching to new logfile: %s", file)
   270  					linesRead = 0
   271  					break
   272  				}
   273  			} else {
   274  				eofSleepMillis = 0
   275  				linesRead++
   276  			}
   277  
   278  			if err == nil && line != "" {
   279  
   280  				matches := csvlogRegex.FindStringSubmatch(line)
   281  				if len(matches) == 0 {
   282  					//log.Debugf("No logline regex match for line:") // normal case actually for queries spanning multiple loglines
   283  					//log.Debugf(line)
   284  					goto send_to_storage_if_needed
   285  				}
   286  
   287  				result := regexMatchesToMap(csvlogRegex, matches)
   288  				//log.Debugf("RegexMatchesToMap: %+v", result)
   289  				errorSeverity, ok := result["error_severity"]
   290  				if !ok {
   291  					logger.Error("error_severity group must be defined in parse regex:", csvlogRegex)
   292  					time.Sleep(time.Minute)
   293  					break
   294  				}
   295  				if serverMessagesLang != "en" {
   296  					errorSeverity = severityToEnglish(serverMessagesLang, errorSeverity)
   297  				}
   298  
   299  				databaseName, ok := result["database_name"]
   300  				if !ok {
   301  					logger.Error("database_name group must be defined in parse regex:", csvlogRegex)
   302  					time.Sleep(time.Minute)
   303  					break
   304  				}
   305  				if realDbname == databaseName {
   306  					eventCounts[errorSeverity]++
   307  				}
   308  				eventCountsTotal[errorSeverity]++
   309  			}
   310  
   311  		send_to_storage_if_needed:
   312  			if lastSendTime.IsZero() || lastSendTime.Before(time.Now().Add(-1*time.Second*time.Duration(interval))) {
   313  				logger.Debugf("Sending log event counts for last interval to storage channel. Local eventcounts: %+v, global eventcounts: %+v", eventCounts, eventCountsTotal)
   314  				metricStoreMessages := eventCountsToMetricStoreMessages(eventCounts, eventCountsTotal, mdb)
   315  				storeCh <- metricStoreMessages
   316  				zeroEventCounts(eventCounts)
   317  				zeroEventCounts(eventCountsTotal)
   318  				lastSendTime = time.Now()
   319  			}
   320  
   321  		} // file read loop
   322  	} // config loop
   323

   326  func severityToEnglish(serverLang, errorSeverity string) string {
   327  	//log.Debug("severityToEnglish", serverLang, errorSeverity)

   345  func tryDetermineLogFolder(ctx context.Context, conn db.PgxIface) (string, error) {
   346  	sql := `select current_setting('data_directory') as dd, current_setting('log_directory') as ld`
   347  	var dd, ld string
   348  	err := conn.QueryRow(ctx, sql).Scan(&dd, &ld)
   349  	if err != nil {
   350  		return "", err
   351  	}
   352  	if strings.HasPrefix(ld, "/") {
   353  		// we have a full path we can use

   355  	}

   372  func regexMatchesToMap(csvlogRegex *regexp.Regexp, matches []string) map[string]string {
   373  	result := make(map[string]string)
   374  	if len(matches) == 0 || csvlogRegex == nil {
   375  		return result
   376  	}
   377  	for i, name := range csvlogRegex.SubexpNames() {
   378  		if i != 0 && name != "" {
   379  			result[name] = matches[i]
   380  		}
   381  	}
   382  	return result
