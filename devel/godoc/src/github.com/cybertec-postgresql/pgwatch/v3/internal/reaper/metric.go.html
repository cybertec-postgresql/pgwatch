    99  				return logrus.WarnLevel
   100  			}
   101  			return logrus.InfoLevel
   102  		}(), "metrics and presets refreshed")
   103  	// update the monitored sources with real metric definitions from presets
   104  	for _, md := range r.monitoredSources {
   105  		if md.PresetMetrics > "" {
