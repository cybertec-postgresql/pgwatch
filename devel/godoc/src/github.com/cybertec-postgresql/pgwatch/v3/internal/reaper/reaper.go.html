    48  // Reap() starts the main monitoring loop. It is responsible for fetching metrics measurements
    49  // from the sources and storing them to the sinks. It also manages the lifecycle of
    50  // the metric gatherers. In case of a source or metric definition change, it will

    77  			go SyncMonitoredDBsToDatastore(ctx, monitoredSources, r.measurementCh)

   101  				srcL.Info("not added to monitoring due to 'master only' property")
   102  				continue
   103  			}
   104  			metricConfig = func() map[string]float64 {
   105  				if len(monitoredSource.Metrics) > 0 {
   106  					return monitoredSource.Metrics
   107  				}
   108  				if monitoredSource.PresetMetrics > "" {
   109  					return metricDefinitionMap.PresetDefs[monitoredSource.PresetMetrics].Metrics
   110  				}
   111  				return nil
   112  			}()
   113  			hostLastKnownStatusInRecovery[monitoredSource.Name] = dbSettings.IsInRecovery
   114  			if dbSettings.IsInRecovery {
   115  				metricConfig = func() map[string]float64 {
   116  					if len(monitoredSource.MetricsStandby) > 0 {
   117  						return monitoredSource.MetricsStandby
   118  					}
   119  					if monitoredSource.PresetMetricsStandby > "" {
   120  						return metricDefinitionMap.PresetDefs[monitoredSource.PresetMetricsStandby].Metrics
   121  					}
   122  					return nil
   123  				}()
   124  			}
   125  
   126  			if monitoredSource.IsPostgresSource() && !dbSettings.IsInRecovery && r.Metrics.CreateHelpers {
   127  				srcL.Info("trying to create helper objects if missing")
   128  				if err = TryCreateMetricsFetchingHelpers(ctx, monitoredSource); err != nil {
   129  					srcL.WithError(err).Warning("failed to create helper functions")
   130  				}
   131  			}
   132  
   133  			if monitoredSource.IsPostgresSource() {
   134  				var DBSizeMB int64
   135  
   136  				if r.Sources.MinDbSizeMB >= 8 { // an empty DB is a bit less than 8MB
   137  					DBSizeMB, _ = DBGetSizeMB(ctx, monitoredSource.Name) // ignore errors, i.e. only remove from monitoring when we're certain it's under the threshold
   138  					if DBSizeMB != 0 {
   139  						if DBSizeMB < r.Sources.MinDbSizeMB {
   140  							srcL.Infof("ignored due to the --min-db-size-mb filter, current size %d MB", DBSizeMB)
   141  							hostsToShutDownDueToRoleChange[monitoredSource.Name] = true // for the case when DB size was previosly above the threshold
   142  							SetUndersizedDBState(monitoredSource.Name, true)
   143  							continue
   144  						}
   145  						SetUndersizedDBState(monitoredSource.Name, false)
   146  					}
   147  				}
   148  				ver, err := GetMonitoredDatabaseSettings(ctx, monitoredSource, false)
   149  				if err == nil { // ok to ignore error, re-tried on next loop
   150  					lastKnownStatusInRecovery := hostLastKnownStatusInRecovery[monitoredSource.Name]
   151  					if ver.IsInRecovery && monitoredSource.OnlyIfMaster {
   152  						srcL.Info("to be removed from monitoring due to 'master only' property and status change")
   153  						hostsToShutDownDueToRoleChange[monitoredSource.Name] = true
   154  						SetRecoveryIgnoredDBState(monitoredSource.Name, true)
   155  						continue
   156  					} else if lastKnownStatusInRecovery != ver.IsInRecovery {
   157  						if ver.IsInRecovery && len(monitoredSource.MetricsStandby) > 0 {
   158  							srcL.Warning("Switching metrics collection to standby config...")
   159  							metricConfig = monitoredSource.MetricsStandby
   160  							hostLastKnownStatusInRecovery[monitoredSource.Name] = true
   161  						} else {
   162  							srcL.Warning("Switching metrics collection to primary config...")
   163  							metricConfig = monitoredSource.Metrics
   164  							hostLastKnownStatusInRecovery[monitoredSource.Name] = false
   165  							SetRecoveryIgnoredDBState(monitoredSource.Name, false)
   166  						}
   167  					}
   168  				}
   169  
   170  				if mainLoopCount == 0 && r.Sources.TryCreateListedExtsIfMissing != "" && !ver.IsInRecovery {
   171  					extsToCreate := strings.Split(r.Sources.TryCreateListedExtsIfMissing, ",")
   172  					extsCreated := TryCreateMissingExtensions(ctx, monitoredSource.Name, extsToCreate, ver.Extensions)
   173  					srcL.Infof("%d/%d extensions created based on --try-create-listed-exts-if-missing input %v", len(extsCreated), len(extsToCreate), extsCreated)
   174  				}
   175  			}
   176  
   177  			for metricName, interval := range metricConfig {
   178  				metric := metricName
   179  				metricDefOk := false
   180  
   181  				if strings.HasPrefix(metric, recoPrefix) {
   182  					metric = recoMetricName
   183  					metricDefOk = true
   184  				} else {
   185  					metricDefMapLock.RLock()
   186  					_, metricDefOk = metricDefinitionMap.MetricDefs[metric]
   187  					metricDefMapLock.RUnlock()
   188  				}
   189  
   190  				dbMetric := monitoredSource.Name + dbMetricJoinStr + metric
   191  				_, chOk := cancelFuncs[dbMetric]
   192  
   193  				if metricDefOk && !chOk { // initialize a new per db/per metric control channel
   194  					if interval > 0 {
   195  						hostMetricIntervalMap[dbMetric] = interval
   196  						srcL.WithField("metric", metric).WithField("interval", interval).Info("starting gatherer")
   197  						metricCtx, cancelFunc := context.WithCancel(ctx)
   198  						cancelFuncs[dbMetric] = cancelFunc
   199  
   200  						metricNameForStorage := metricName
   201  						if _, isSpecialMetric := specialMetrics[metricName]; !isSpecialMetric {
   202  							vme, err := GetMonitoredDatabaseSettings(ctx, monitoredSource, false)
   203  							if err != nil {
   204  								srcL.WithError(err).Warning("Failed to determine possible re-routing name, Grafana dashboards with re-routed metrics might not show all hosts")
   205  							} else {
   206  								mvp, err := GetMetricVersionProperties(metricName, vme, nil)
   207  								if err != nil && !strings.Contains(err.Error(), "too old") {
   208  									srcL.WithError(err).Warning("Failed to determine possible re-routing name, Grafana dashboards with re-routed metrics might not show all hosts")
   209  								} else if mvp.StorageName != "" {
   210  									metricNameForStorage = mvp.StorageName
   211  								}
   212  							}
   213  						}
   214  
   215  						if err := r.SinksWriter.SyncMetric(monitoredSource.Name, metricNameForStorage, "add"); err != nil {
   216  							srcL.Error(err)
   217  						}
   218  
   219  						go r.reapMetricMeasurements(metricCtx, monitoredSource, metric, metricConfig)
   220  					}
   221  				} else if (!metricDefOk && chOk) || interval <= 0 {
   222  					// metric definition files were recently removed or interval set to zero
   223  					if cancelFunc, isOk := cancelFuncs[dbMetric]; isOk {
   224  						cancelFunc()
   225  					}
   226  					srcL.WithField("metric", metric).Warning("shutting down gatherer...")
   227  					delete(cancelFuncs, dbMetric)
   228  				} else if !metricDefOk {
   229  					epoch, ok := lastSQLFetchError.Load(metric)
   230  					if !ok || ((time.Now().Unix() - epoch.(int64)) > 3600) { // complain only 1x per hour
   231  						srcL.WithField("metric", metric).Warning("metric definition not found")
   232  						lastSQLFetchError.Store(metric, time.Now().Unix())
   233  					}
   234  				} else {
   235  					// check if interval has changed
   236  					if hostMetricIntervalMap[dbMetric] != interval {
   237  						srcL.WithField("metric", metric).Info("updating interval...")
   238  						hostMetricIntervalMap[dbMetric] = interval
   239  					}
   240  				}
   241  			}
   242  		}
   243  
   244  		if mainLoopCount == 0 {
   245  			goto MainLoopSleep
   246  		}
   247  
   248  		// loop over existing channels and stop workers if DB or metric removed from config
   249  		// or state change makes it uninteresting
   250  		logger.Debug("checking if any workers need to be shut down...")
   251  		for dbMetric, cancelFunc := range cancelFuncs {
   252  			var currentMetricConfig map[string]float64
   253  			var dbInfo *sources.SourceConn
   254  			var ok, dbRemovedFromConfig bool
   255  			singleMetricDisabled := false
   256  			splits := strings.Split(dbMetric, dbMetricJoinStr)
   257  			db := splits[0]
   258  			metric := splits[1]
   259  			//log.Debugf("Checking if need to shut down worker for [%s:%s]...", db, metric)
   260  
   261  			_, wholeDbShutDownDueToRoleChange := hostsToShutDownDueToRoleChange[db]
   262  			if !wholeDbShutDownDueToRoleChange {
   263  				monitoredDbCacheLock.RLock()
   264  				dbInfo, ok = monitoredDbCache[db]
   265  				monitoredDbCacheLock.RUnlock()
   266  				if !ok { // normal removing of DB from config
   267  					dbRemovedFromConfig = true
   268  					logger.Debugf("DB %s removed from config, shutting down all metric worker processes...", db)
   269  				}
   270  			}
   271  
   272  			if !(wholeDbShutDownDueToRoleChange || dbRemovedFromConfig) { // maybe some single metric was disabled
   273  				MonitoredDatabasesSettingsLock.RLock()
   274  				verInfo, ok := MonitoredDatabasesSettings[db]
   275  				MonitoredDatabasesSettingsLock.RUnlock()
   276  				if !ok {
   277  					logger.Warningf("Could not find PG version info for DB %s, skipping shutdown check of metric worker process for %s", db, metric)
   278  					continue
   279  				}
   280  				if verInfo.IsInRecovery && dbInfo.PresetMetricsStandby > "" || !verInfo.IsInRecovery && dbInfo.PresetMetrics > "" {
   281  					continue // no need to check presets for single metric disabling
   282  				}
   283  				if verInfo.IsInRecovery && len(dbInfo.MetricsStandby) > 0 {
   284  					currentMetricConfig = dbInfo.MetricsStandby
   285  				} else {
   286  					currentMetricConfig = dbInfo.Metrics
   287  				}
   288  
   289  				interval, isMetricActive := currentMetricConfig[metric]
   290  				if !isMetricActive || interval <= 0 {
   291  					singleMetricDisabled = true
   292  				}
   293  			}
   294  
   295  			if ctx.Err() != nil || wholeDbShutDownDueToRoleChange || dbRemovedFromConfig || singleMetricDisabled {
   296  				logger.WithField("source", db).WithField("metric", metric).Info("stoppin gatherer...")
   297  				cancelFunc()
   298  				delete(cancelFuncs, dbMetric)
   299  				if err := r.SinksWriter.SyncMetric(db, metric, "remove"); err != nil {
   300  					logger.Error(err)
   301  				}
   302  			}
   303  		}
   304  
   305  		// Destroy conn pools and metric writers
   306  		CloseResourcesForRemovedMonitoredDBs(r.SinksWriter, monitoredSources, prevLoopMonitoredDBs, hostsToShutDownDueToRoleChange)
   307  
   308  	MainLoopSleep:
   309  		mainLoopCount++
   310  		prevLoopMonitoredDBs = slices.Clone(monitoredSources)
   311  
   312  		logger.Debugf("main sleeping %ds...", r.Sources.Refresh)
   313  		select {
   314  		case <-time.After(time.Second * time.Duration(r.Sources.Refresh)):
   315  			// pass
   316  		case <-ctx.Done():
   317  			return
   318  		}
   319  	}

   323  func (r *Reaper) reapMetricMeasurements(ctx context.Context,
   324  	mdb *sources.SourceConn,
   325  	metricName string,
   326  	configMap map[string]float64) {
   327  
   328  	hostState := make(map[string]map[string]string)
   329  	var lastUptimeS int64 = -1 // used for "server restarted" event detection
   330  	var lastErrorNotificationTime time.Time
   331  	var vme MonitoredDatabaseSettings
   332  	var mvp metrics.Metric
   333  	var err error
   334  	failedFetches := 0
   335  	lastDBVersionFetchTime := time.Unix(0, 0) // check DB ver. ev. 5 min
   336  
   337  	l := r.logger.WithField("source", mdb.Name).WithField("metric", metricName)
   338  	if metricName == specialMetricServerLogEventCounts {
   339  		MonitoredDatabasesSettingsLock.RLock()
   340  		realDbname := MonitoredDatabasesSettings[mdb.Name].RealDbname // to manage 2 sets of event counts - monitored DB + global
   341  		MonitoredDatabasesSettingsLock.RUnlock()
   342  		metrics.ParseLogs(ctx, mdb, realDbname, metricName, configMap, r.measurementCh) // no return
   343  		return
   344  	}
   345  
   346  	for {
   347  		interval := configMap[metricName]
   348  		if lastDBVersionFetchTime.Add(time.Minute * time.Duration(5)).Before(time.Now()) {
   349  			vme, err = GetMonitoredDatabaseSettings(ctx, mdb, false) // in case of errors just ignore metric "disabled" time ranges
   350  			if err != nil {
   351  				lastDBVersionFetchTime = time.Now()
   352  			}
   353  
   354  			mvp, err = GetMetricVersionProperties(metricName, vme, nil)
   355  			if err != nil {
   356  				l.Errorf("Could not get metric version properties: %v", err)
   357  				return
   358  			}
   359  		}
   360  
   361  		var metricStoreMessages []metrics.MeasurementEnvelope
   362  		var err error
   363  		mfm := MetricFetchConfig{
   364  			DBUniqueName:        mdb.Name,
   365  			DBUniqueNameOrig:    mdb.GetDatabaseName(),
   366  			MetricName:          metricName,
   367  			Source:              mdb.Kind,
   368  			Interval:            time.Second * time.Duration(interval),
   369  			StmtTimeoutOverride: 0,
   370  		}
   371  
   372  		// 1st try local overrides for some metrics if operating in push mode
   373  		if r.Metrics.DirectOSStats && IsDirectlyFetchableMetric(metricName) {
   374  			metricStoreMessages, err = FetchStatsDirectlyFromOS(ctx, mfm, vme, mvp)
   375  			if err != nil {
   376  				l.WithError(err).Errorf("Could not reader metric directly from OS")
   377  			}
   378  		}
   379  		t1 := time.Now()
   380  		if metricStoreMessages == nil {
   381  			metricStoreMessages, err = FetchMetrics(ctx, mfm, hostState, r.measurementCh, r.Options)
   382  		}
   383  		t2 := time.Now()
   384  
   385  		if t2.Sub(t1) > (time.Second * time.Duration(interval)) {
   386  			l.Warningf("Total fetching time of %vs bigger than %vs interval", t2.Sub(t1).Truncate(time.Millisecond*100).Seconds(), interval)
   387  		}
   388  
   389  		if err != nil {
   390  			failedFetches++
   391  			// complain only 1x per 10min per host/metric...
   392  			if lastErrorNotificationTime.IsZero() || lastErrorNotificationTime.Add(time.Second*time.Duration(600)).Before(time.Now()) {
   393  				l.WithError(err).Error("failed to fetch metric data")
   394  				if failedFetches > 1 {
   395  					l.Errorf("Total failed fetches: %d", failedFetches)
   396  				}
   397  				lastErrorNotificationTime = time.Now()
   398  			}
   399  		} else if metricStoreMessages != nil {
   400  			if len(metricStoreMessages[0].Data) > 0 {
   401  
   402  				// pick up "server restarted" events here to avoid doing extra selects from CheckForPGObjectChangesAndStore code
   403  				if metricName == "db_stats" {
   404  					postmasterUptimeS, ok := (metricStoreMessages[0].Data)[0]["postmaster_uptime_s"]
   405  					if ok {
   406  						if lastUptimeS != -1 {
   407  							if postmasterUptimeS.(int64) < lastUptimeS { // restart (or possibly also failover when host is routed) happened
   408  								message := "Detected server restart (or failover) of \"" + mdb.Name + "\""
   409  								l.Warning(message)
   410  								detectedChangesSummary := make(metrics.Measurements, 0)
   411  								entry := metrics.Measurement{"details": message, "epoch_ns": (metricStoreMessages[0].Data)[0]["epoch_ns"]}
   412  								detectedChangesSummary = append(detectedChangesSummary, entry)
   413  								metricStoreMessages = append(metricStoreMessages,
   414  									metrics.MeasurementEnvelope{
   415  										DBName:     mdb.Name,
   416  										SourceType: string(mdb.Kind),
   417  										MetricName: "object_changes",
   418  										Data:       detectedChangesSummary,
   419  										CustomTags: metricStoreMessages[0].CustomTags,
   420  									})
   421  							}
   422  						}
   423  						lastUptimeS = postmasterUptimeS.(int64)
   424  					}
   425  				}
   426  				r.measurementCh <- metricStoreMessages
   427  			}
   428  		}
   429  
   430  		select {
   431  		case <-ctx.Done():
   432  			return
   433  		case <-time.After(time.Second * time.Duration(interval)):
   434  			// continue
   435  		}
   436  	}

   510  func PutToInstanceCache(msg MetricFetchConfig, data metrics.Measurements) {
   511  	if len(data) == 0 {
   512  		return
   513  	}
   514  	dataCopy := deepCopyMetricData(data)
   515  	instanceMetricCacheLock.Lock()
   516  	instanceMetricCache[msg.DBUniqueNameOrig+msg.MetricName] = dataCopy
   517  	instanceMetricCacheLock.Unlock()
   518  
   519  	instanceMetricCacheTimestampLock.Lock()
   520  	instanceMetricCacheTimestamp[msg.DBUniqueNameOrig+msg.MetricName] = time.Now()
   521  	instanceMetricCacheTimestampLock.Unlock()

   525  	newData := make(metrics.Measurements, len(data))

   538  func FetchMetrics(ctx context.Context,
   539  	msg MetricFetchConfig,
   540  	hostState map[string]map[string]string,
   541  	storageCh chan<- []metrics.MeasurementEnvelope,
   542  	opts *cmdopts.Options) ([]metrics.MeasurementEnvelope, error) {
   543  
   544  	var dbSettings MonitoredDatabaseSettings
   545  	var dbVersion int
   546  	var err error
   547  	var sql string
   548  	var data, cachedData metrics.Measurements
   549  	var md *sources.SourceConn
   550  	var fromCache, isCacheable bool
   551  
   552  	md, err = GetMonitoredDatabaseByUniqueName(msg.DBUniqueName)
   553  	if err != nil {
   554  		log.GetLogger(ctx).Errorf("[%s:%s] could not get monitored DB details", msg.DBUniqueName, err)
   555  		return nil, err
   556  	}
   557  
   558  	dbSettings, err = GetMonitoredDatabaseSettings(ctx, md, false)
   559  	if err != nil {
   560  		log.GetLogger(ctx).Error("failed to fetch pg version for ", msg.DBUniqueName, msg.MetricName, err)
   561  		return nil, err
   562  	}
   563  	if msg.MetricName == specialMetricDbSize || msg.MetricName == specialMetricTableStats {
   564  		if dbSettings.ExecEnv == execEnvAzureSingle && dbSettings.ApproxDBSizeB > 1e12 { // 1TB
   565  			subsMetricName := msg.MetricName + "_approx"
   566  			mvpApprox, err := GetMetricVersionProperties(subsMetricName, dbSettings, nil)
   567  			if err == nil && mvpApprox.StorageName == msg.MetricName {
   568  				log.GetLogger(ctx).Infof("[%s:%s] Transparently swapping metric to %s due to hard-coded rules...", msg.DBUniqueName, msg.MetricName, subsMetricName)
   569  				msg.MetricName = subsMetricName
   570  			}
   571  		}
   572  	}
   573  	dbVersion = dbSettings.Version
   574  
   575  	if msg.Source == sources.SourcePgBouncer {
   576  		dbVersion = 0 // version is 0.0 for all pgbouncer sql per convention
   577  	}
   578  
   579  	mvp, err := GetMetricVersionProperties(msg.MetricName, dbSettings, nil)
   580  	if err != nil && msg.MetricName != recoMetricName {
   581  		epoch, ok := lastSQLFetchError.Load(msg.MetricName + dbMetricJoinStr + fmt.Sprintf("%v", dbVersion))
   582  		if !ok || ((time.Now().Unix() - epoch.(int64)) > 3600) { // complain only 1x per hour
   583  			log.GetLogger(ctx).Infof("Failed to get SQL for metric '%s', version '%s': %v", msg.MetricName, dbSettings.VersionStr, err)
   584  			lastSQLFetchError.Store(msg.MetricName+dbMetricJoinStr+fmt.Sprintf("%v", dbVersion), time.Now().Unix())
   585  		}
   586  		if strings.Contains(err.Error(), "too old") {
   587  			return nil, nil
   588  		}
   589  		return nil, err
   590  	}
   591  
   592  	isCacheable = IsCacheableMetric(msg, mvp)
   593  	if isCacheable && opts.Metrics.InstanceLevelCacheMaxSeconds > 0 && msg.Interval.Seconds() > float64(opts.Metrics.InstanceLevelCacheMaxSeconds) {
   594  		cachedData = GetFromInstanceCacheIfNotOlderThanSeconds(msg, opts.Metrics.InstanceLevelCacheMaxSeconds)
   595  		if len(cachedData) > 0 {
   596  			fromCache = true
   597  			goto send_to_storageChannel
   598  		}
   599  	}
   600  
   601  	sql = mvp.GetSQL(dbVersion)
   602  
   603  	if sql == "" && !(msg.MetricName == specialMetricChangeEvents || msg.MetricName == recoMetricName) {
   604  		// let's ignore dummy SQLs
   605  		log.GetLogger(ctx).Debugf("[%s:%s] Ignoring fetch message - got an empty/dummy SQL string", msg.DBUniqueName, msg.MetricName)
   606  		return nil, nil
   607  	}
   608  
   609  	if (mvp.PrimaryOnly() && dbSettings.IsInRecovery) || (mvp.StandbyOnly() && !dbSettings.IsInRecovery) {
   610  		log.GetLogger(ctx).Debugf("[%s:%s] Skipping fetching of  as server not in wanted state (IsInRecovery=%v)", msg.DBUniqueName, msg.MetricName, dbSettings.IsInRecovery)
   611  		return nil, nil
   612  	}
   613  
   614  	if msg.MetricName == specialMetricChangeEvents { // special handling, multiple queries + stateful
   615  		CheckForPGObjectChangesAndStore(ctx, msg.DBUniqueName, dbSettings, storageCh, hostState) // TODO no hostState for Prometheus currently
   616  	} else if msg.MetricName == recoMetricName {
   617  		if data, err = GetRecommendations(ctx, msg.DBUniqueName, dbSettings); err != nil {
   618  			return nil, err
   619  		}
   620  	} else if msg.Source == sources.SourcePgPool {
   621  		if data, err = FetchMetricsPgpool(ctx, msg, dbSettings, mvp); err != nil {
   622  			return nil, err
   623  		}
   624  	} else {
   625  		data, err = DBExecReadByDbUniqueName(ctx, msg.DBUniqueName, sql)
   626  
   627  		if err != nil {
   628  			// let's soften errors to "info" from functions that expect the server to be a primary to reduce noise
   629  			if strings.Contains(err.Error(), "recovery is in progress") {
   630  				MonitoredDatabasesSettingsLock.RLock()
   631  				ver := MonitoredDatabasesSettings[msg.DBUniqueName]
   632  				MonitoredDatabasesSettingsLock.RUnlock()
   633  				if ver.IsInRecovery {
   634  					log.GetLogger(ctx).Debugf("[%s:%s] failed to fetch metrics: %s", msg.DBUniqueName, msg.MetricName, err)
   635  					return nil, err
   636  				}
   637  			}
   638  
   639  			if msg.MetricName == specialMetricInstanceUp {
   640  				log.GetLogger(ctx).WithError(err).Debugf("[%s:%s] failed to fetch metrics. marking instance as not up", msg.DBUniqueName, msg.MetricName)
   641  				data = make(metrics.Measurements, 1)
   642  				data[0] = metrics.Measurement{"epoch_ns": time.Now().UnixNano(), "is_up": 0} // should be updated if the "instance_up" metric definition is changed
   643  				goto send_to_storageChannel
   644  			}
   645  
   646  			log.GetLogger(ctx).
   647  				WithFields(map[string]any{"source": msg.DBUniqueName, "metric": msg.MetricName}).
   648  				WithError(err).Error("failed to fetch metrics")
   649  
   650  			return nil, err
   651  		}
   652  
   653  		log.GetLogger(ctx).WithFields(map[string]any{"source": msg.DBUniqueName, "metric": msg.MetricName, "rows": len(data)}).Info("measurements fetched")
   654  	}
   655  
   656  	if isCacheable && opts.Metrics.InstanceLevelCacheMaxSeconds > 0 && msg.Interval.Seconds() > float64(opts.Metrics.InstanceLevelCacheMaxSeconds) {
   657  		PutToInstanceCache(msg, data)
   658  	}
   659  
   660  send_to_storageChannel:
   661  
   662  	if (opts.Sinks.RealDbnameField > "" || opts.Sinks.SystemIdentifierField > "") && msg.Source == sources.SourcePostgres {
   663  		MonitoredDatabasesSettingsLock.RLock()
   664  		ver := MonitoredDatabasesSettings[msg.DBUniqueName]
   665  		MonitoredDatabasesSettingsLock.RUnlock()
   666  		data = AddDbnameSysinfoIfNotExistsToQueryResultData(data, ver, opts)
   667  	}
   668  
   669  	if mvp.StorageName != "" {
   670  		log.GetLogger(ctx).Debugf("[%s] rerouting metric %s data to %s based on metric attributes", msg.DBUniqueName, msg.MetricName, mvp.StorageName)
   671  		msg.MetricName = mvp.StorageName
   672  	}
   673  	if fromCache {
   674  		log.GetLogger(ctx).Infof("[%s:%s] loaded %d rows from the instance cache", msg.DBUniqueName, msg.MetricName, len(cachedData))
   675  		return []metrics.MeasurementEnvelope{{DBName: msg.DBUniqueName, MetricName: msg.MetricName, Data: cachedData, CustomTags: md.CustomTags,
   676  			MetricDef: mvp, RealDbname: dbSettings.RealDbname, SystemIdentifier: dbSettings.SystemIdentifier}}, nil
   677  	}
   678  	return []metrics.MeasurementEnvelope{{DBName: msg.DBUniqueName, MetricName: msg.MetricName, Data: data, CustomTags: md.CustomTags,
   679  		MetricDef: mvp, RealDbname: dbSettings.RealDbname, SystemIdentifier: dbSettings.SystemIdentifier}}, nil
   680  
   681  }
