   265  					}

   270  					}

   285  				}
   286  			}

   314  				}
   315  			}
   316

   320  				MonitoredDatabasesSettingsLock.RUnlock()

   322  					logger.Warningf("Could not find PG version info for DB %s, skipping shutdown check of metric worker process for %s", db, metric)
   323  					continue
   324  				}
   325  				if verInfo.IsInRecovery && dbInfo.PresetMetricsStandby > "" || !verInfo.IsInRecovery && dbInfo.PresetMetrics > "" {
   326  					continue // no need to check presets for single metric disabling
   327  				}
   328  				if verInfo.IsInRecovery && len(dbInfo.MetricsStandby) > 0 {
   329  					currentMetricConfig = dbInfo.MetricsStandby
   330  				} else {
   331  					currentMetricConfig = dbInfo.Metrics
   332  				}
   333  
   334  				interval, isMetricActive := currentMetricConfig[metric]
   335  				if !isMetricActive || interval <= 0 {
   336  					singleMetricDisabled = true
   337  				}
   338  			}
   339  
   340  			if mainContext.Err() != nil || wholeDbShutDownDueToRoleChange || dbRemovedFromConfig || singleMetricDisabled {
   341  				logger.Infof("shutting down gatherer for [%s:%s] ...", db, metric)
   342  				cancelFunc()
   343  				delete(cancelFuncs, dbMetric)
   344  				logger.Debugf("cancel function for [%s:%s] deleted", db, metric)
   345  				gatherersShutDown++
   346  				ClearDBUnreachableStateIfAny(db)
   347  				if err := measurementsWriter.SyncMetrics(db, metric, "remove"); err != nil {
   348  					logger.Error(err)
   349  				}
   350  			}
   351  		}
   352  
   353  		if gatherersShutDown > 0 {
   354  			logger.Warningf("sent STOP message to %d gatherers (it might take some time for them to stop though)", gatherersShutDown)
   355  		}
   356  
   357  		// Destroy conn pools and metric writers
   358  		CloseResourcesForRemovedMonitoredDBs(measurementsWriter, monitoredDbs, prevLoopMonitoredDBs, hostsToShutDownDueToRoleChange)
   359  
   360  	MainLoopSleep:
   361  		mainLoopCount++
   362  		prevLoopMonitoredDBs = slices.Clone(monitoredDbs)
   363  
   364  		logger.Debugf("main sleeping %ds...", opts.Sources.Refresh)

   366  		case <-time.After(time.Second * time.Duration(opts.Sources.Refresh)):

   368  				logger.Error("could not fetch active hosts, using last valid config data:", err)
   369  			}
   370  		case <-mainContext.Done():
   371  			return
   372  		}
   373  	}

   377  func (r *Reaper) reapMetricMeasurementsFromSource(ctx context.Context,
   378  	dbUniqueName, dbUniqueNameOrig string,
   379  	srcType sources.Kind,
   380  	metricName string,
   381  	configMap map[string]float64) {
   382  
   383  	hostState := make(map[string]map[string]string)
   384  	var lastUptimeS int64 = -1 // used for "server restarted" event detection
   385  	var lastErrorNotificationTime time.Time
   386  	var vme MonitoredDatabaseSettings
   387  	var mvp metrics.Metric
   388  	var err error
   389  	failedFetches := 0
   390  	lastDBVersionFetchTime := time.Unix(0, 0) // check DB ver. ev. 5 min
   391  
   392  	l := log.GetLogger(ctx).WithField("source", dbUniqueName).WithField("metric", metricName)
   393  	if metricName == specialMetricServerLogEventCounts {
   394  		mdb, err := GetMonitoredDatabaseByUniqueName(dbUniqueName)
   395  		if err != nil {
   396  			return
   397  		}
   398  		MonitoredDatabasesSettingsLock.RLock()
   399  		realDbname := MonitoredDatabasesSettings[dbUniqueName].RealDbname // to manage 2 sets of event counts - monitored DB + global
   400  		MonitoredDatabasesSettingsLock.RUnlock()
   401  		conn := mdb.Conn
   402  		metrics.ParseLogs(ctx, conn, mdb, realDbname, metricName, configMap, r.measurementCh) // no return
   403  		return
   404  	}
   405  
   406  	for {
   407  		interval := configMap[metricName]
   408  		if lastDBVersionFetchTime.Add(time.Minute * time.Duration(5)).Before(time.Now()) {
   409  			vme, err = GetMonitoredDatabaseSettings(ctx, dbUniqueName, srcType, false) // in case of errors just ignore metric "disabled" time ranges
   410  			if err != nil {
   411  				lastDBVersionFetchTime = time.Now()
   412  			}
   413  
   414  			mvp, err = GetMetricVersionProperties(metricName, vme, nil)
   415  			if err != nil {
   416  				l.Errorf("Could not get metric version properties: %v", err)
   417  				return
   418  			}
   419  		}
   420  
   421  		var metricStoreMessages []metrics.MeasurementEnvelope
   422  		var err error
   423  		mfm := MetricFetchConfig{
   424  			DBUniqueName:        dbUniqueName,
   425  			DBUniqueNameOrig:    dbUniqueNameOrig,
   426  			MetricName:          metricName,
   427  			Source:              srcType,
   428  			Interval:            time.Second * time.Duration(interval),
   429  			StmtTimeoutOverride: 0,
   430  		}
   431  
   432  		// 1st try local overrides for some metrics if operating in push mode
   433  		if r.opts.Metrics.DirectOSStats && IsDirectlyFetchableMetric(metricName) {
   434  			metricStoreMessages, err = FetchStatsDirectlyFromOS(ctx, mfm, vme, mvp)
   435  			if err != nil {
   436  				l.WithError(err).Errorf("Could not reader metric directly from OS")
   437  			}
   438  		}
   439  		t1 := time.Now()
   440  		if metricStoreMessages == nil {
   441  			metricStoreMessages, err = FetchMetrics(ctx, mfm, hostState, r.measurementCh, "", r.opts)
   442  		}
   443  		t2 := time.Now()
   444  
   445  		if t2.Sub(t1) > (time.Second * time.Duration(interval)) {
   446  			l.Warningf("Total fetching time of %vs bigger than %vs interval", t2.Sub(t1).Truncate(time.Millisecond*100).Seconds(), interval)
   447  		}
   448  
   449  		if err != nil {
   450  			failedFetches++
   451  			// complain only 1x per 10min per host/metric...
   452  			if lastErrorNotificationTime.IsZero() || lastErrorNotificationTime.Add(time.Second*time.Duration(600)).Before(time.Now()) {
   453  				l.WithError(err).Error("failed to fetch metric data")
   454  				if failedFetches > 1 {
   455  					l.Errorf("Total failed fetches: %d", failedFetches)
   456  				}
   457  				lastErrorNotificationTime = time.Now()
   458  			}
   459  		} else if metricStoreMessages != nil {
   460  			if len(metricStoreMessages[0].Data) > 0 {
   461  
   462  				// pick up "server restarted" events here to avoid doing extra selects from CheckForPGObjectChangesAndStore code
   463  				if metricName == "db_stats" {
   464  					postmasterUptimeS, ok := (metricStoreMessages[0].Data)[0]["postmaster_uptime_s"]
   465  					if ok {
   466  						if lastUptimeS != -1 {
   467  							if postmasterUptimeS.(int64) < lastUptimeS { // restart (or possibly also failover when host is routed) happened
   468  								message := "Detected server restart (or failover) of \"" + dbUniqueName + "\""
   469  								l.Warning(message)
   470  								detectedChangesSummary := make(metrics.Measurements, 0)
   471  								entry := metrics.Measurement{"details": message, "epoch_ns": (metricStoreMessages[0].Data)[0]["epoch_ns"]}
   472  								detectedChangesSummary = append(detectedChangesSummary, entry)
   473  								metricStoreMessages = append(metricStoreMessages,
   474  									metrics.MeasurementEnvelope{
   475  										DBName:     dbUniqueName,
   476  										SourceType: string(srcType),
   477  										MetricName: "object_changes",
   478  										Data:       detectedChangesSummary,
   479  										CustomTags: metricStoreMessages[0].CustomTags,
   480  									})
   481  							}
   482  						}
   483  						lastUptimeS = postmasterUptimeS.(int64)
   484  					}
   485  				}
   486  
   487  				r.measurementCh <- metricStoreMessages
   488  			}
   489  		}
   490  
   491  		select {
   492  		case <-ctx.Done():

   599  func FetchMetrics(ctx context.Context,
   600  	msg MetricFetchConfig,
   601  	hostState map[string]map[string]string,
   602  	storageCh chan<- []metrics.MeasurementEnvelope,
   603  	context string,
   604  	opts *cmdopts.Options) ([]metrics.MeasurementEnvelope, error) {
   605  
   606  	var dbSettings MonitoredDatabaseSettings
   607  	var dbVersion int
   608  	var err error
   609  	var sql string
   610  	var data, cachedData metrics.Measurements
   611  	var md *sources.MonitoredDatabase
   612  	var fromCache, isCacheable bool
   613  
   614  	md, err = GetMonitoredDatabaseByUniqueName(msg.DBUniqueName)
   615  	if err != nil {
   616  		log.GetLogger(ctx).Errorf("[%s:%s] could not get monitored DB details", msg.DBUniqueName, err)
   617  		return nil, err
   618  	}
   619  
   620  	dbSettings, err = GetMonitoredDatabaseSettings(ctx, msg.DBUniqueName, msg.Source, false)
   621  	if err != nil {
   622  		log.GetLogger(ctx).Error("failed to fetch pg version for ", msg.DBUniqueName, msg.MetricName, err)
   623  		return nil, err
   624  	}
   625  	if msg.MetricName == specialMetricDbSize || msg.MetricName == specialMetricTableStats {
   626  		if dbSettings.ExecEnv == execEnvAzureSingle && dbSettings.ApproxDBSizeB > 1e12 { // 1TB
   627  			subsMetricName := msg.MetricName + "_approx"
   628  			mvpApprox, err := GetMetricVersionProperties(subsMetricName, dbSettings, nil)
   629  			if err == nil && mvpApprox.StorageName == msg.MetricName {
   630  				log.GetLogger(ctx).Infof("[%s:%s] Transparently swapping metric to %s due to hard-coded rules...", msg.DBUniqueName, msg.MetricName, subsMetricName)
   631  				msg.MetricName = subsMetricName
   632  			}
   633  		}
   634  	}
   635  	dbVersion = dbSettings.Version
   636  
   637  	if msg.Source == sources.SourcePgBouncer {
   638  		dbVersion = 0 // version is 0.0 for all pgbouncer sql per convention
   639  	}
   640  
   641  	mvp, err := GetMetricVersionProperties(msg.MetricName, dbSettings, nil)
   642  	if err != nil && msg.MetricName != recoMetricName {
   643  		epoch, ok := lastSQLFetchError.Load(msg.MetricName + dbMetricJoinStr + fmt.Sprintf("%v", dbVersion))
   644  		if !ok || ((time.Now().Unix() - epoch.(int64)) > 3600) { // complain only 1x per hour
   645  			log.GetLogger(ctx).Infof("Failed to get SQL for metric '%s', version '%s': %v", msg.MetricName, dbSettings.VersionStr, err)
   646  			lastSQLFetchError.Store(msg.MetricName+dbMetricJoinStr+fmt.Sprintf("%v", dbVersion), time.Now().Unix())
   647  		}
   648  		if strings.Contains(err.Error(), "too old") {
   649  			return nil, nil
   650  		}
   651  		return nil, err
   652  	}
   653  
   654  	isCacheable = IsCacheableMetric(msg, mvp)
   655  	if isCacheable && opts.Metrics.InstanceLevelCacheMaxSeconds > 0 && msg.Interval.Seconds() > float64(opts.Metrics.InstanceLevelCacheMaxSeconds) {
   656  		cachedData = GetFromInstanceCacheIfNotOlderThanSeconds(msg, opts.Metrics.InstanceLevelCacheMaxSeconds)
   657  		if len(cachedData) > 0 {
   658  			fromCache = true
   659  			goto send_to_storageChannel
   660  		}
   661  	}
   662  
   663  	sql = mvp.GetSQL(dbVersion)
   664  
   665  	if sql == "" && !(msg.MetricName == specialMetricChangeEvents || msg.MetricName == recoMetricName) {
   666  		// let's ignore dummy SQLs
   667  		log.GetLogger(ctx).Debugf("[%s:%s] Ignoring fetch message - got an empty/dummy SQL string", msg.DBUniqueName, msg.MetricName)
   668  		return nil, nil
   669  	}
   670  
   671  	if (mvp.PrimaryOnly() && dbSettings.IsInRecovery) || (mvp.StandbyOnly() && !dbSettings.IsInRecovery) {
   672  		log.GetLogger(ctx).Debugf("[%s:%s] Skipping fetching of  as server not in wanted state (IsInRecovery=%v)", msg.DBUniqueName, msg.MetricName, dbSettings.IsInRecovery)
   673  		return nil, nil
   674  	}
   675  
   676  	if msg.MetricName == specialMetricChangeEvents && context != contextPrometheusScrape { // special handling, multiple queries + stateful
   677  		CheckForPGObjectChangesAndStore(ctx, msg.DBUniqueName, dbSettings, storageCh, hostState) // TODO no hostState for Prometheus currently
   678  	} else if msg.MetricName == recoMetricName && context != contextPrometheusScrape {
   679  		if data, err = GetRecommendations(ctx, msg.DBUniqueName, dbSettings); err != nil {
   680  			return nil, err
   681  		}
   682  	} else if msg.Source == sources.SourcePgPool {
   683  		if data, err = FetchMetricsPgpool(ctx, msg, dbSettings, mvp); err != nil {
   684  			return nil, err
   685  		}
   686  	} else {
   687  		data, err = DBExecReadByDbUniqueName(ctx, msg.DBUniqueName, sql)
   688  
   689  		if err != nil {
   690  			// let's soften errors to "info" from functions that expect the server to be a primary to reduce noise
   691  			if strings.Contains(err.Error(), "recovery is in progress") {
   692  				MonitoredDatabasesSettingsLock.RLock()
   693  				ver := MonitoredDatabasesSettings[msg.DBUniqueName]
   694  				MonitoredDatabasesSettingsLock.RUnlock()
   695  				if ver.IsInRecovery {
   696  					log.GetLogger(ctx).Debugf("[%s:%s] failed to fetch metrics: %s", msg.DBUniqueName, msg.MetricName, err)
   697  					return nil, err
   698  				}
   699  			}
   700  
   701  			if msg.MetricName == specialMetricInstanceUp {
   702  				log.GetLogger(ctx).WithError(err).Debugf("[%s:%s] failed to fetch metrics. marking instance as not up", msg.DBUniqueName, msg.MetricName)
   703  				data = make(metrics.Measurements, 1)
   704  				data[0] = metrics.Measurement{"epoch_ns": time.Now().UnixNano(), "is_up": 0} // should be updated if the "instance_up" metric definition is changed
   705  				goto send_to_storageChannel
   706  			}
   707  
   708  			if strings.Contains(err.Error(), "connection refused") {
   709  				SetDBUnreachableState(msg.DBUniqueName)
   710  			}
   711  
   712  			log.GetLogger(ctx).Infof("[%s:%s] failed to fetch metrics: %s", msg.DBUniqueName, msg.MetricName, err)
   713  
   714  			return nil, err
   715  		}
   716  
   717  		log.GetLogger(ctx).WithFields(map[string]any{"source": msg.DBUniqueName, "metric": msg.MetricName, "rows": len(data)}).Info("measurements fetched")
   718  		if regexIsPgbouncerMetrics.MatchString(msg.MetricName) { // clean unwanted pgbouncer pool stats here as not possible in SQL
   719  			data = FilterPgbouncerData(ctx, data, md.GetDatabaseName(), dbSettings)
   720  		}
   721  
   722  		ClearDBUnreachableStateIfAny(msg.DBUniqueName)
   723  
   724  	}
   725  
   726  	if isCacheable && opts.Metrics.InstanceLevelCacheMaxSeconds > 0 && msg.Interval.Seconds() > float64(opts.Metrics.InstanceLevelCacheMaxSeconds) {
   727  		PutToInstanceCache(msg, data)
   728  	}
   729  
   730  send_to_storageChannel:
   731  
   732  	if (opts.Sinks.RealDbnameField > "" || opts.Sinks.SystemIdentifierField > "") && msg.Source == sources.SourcePostgres {
   733  		MonitoredDatabasesSettingsLock.RLock()
   734  		ver := MonitoredDatabasesSettings[msg.DBUniqueName]
   735  		MonitoredDatabasesSettingsLock.RUnlock()
   736  		data = AddDbnameSysinfoIfNotExistsToQueryResultData(data, ver, opts)
   737  	}
   738  
   739  	if mvp.StorageName != "" {
   740  		log.GetLogger(ctx).Debugf("[%s] rerouting metric %s data to %s based on metric attributes", msg.DBUniqueName, msg.MetricName, mvp.StorageName)
   741  		msg.MetricName = mvp.StorageName
   742  	}
   743  	if fromCache {
   744  		log.GetLogger(ctx).Infof("[%s:%s] loaded %d rows from the instance cache", msg.DBUniqueName, msg.MetricName, len(cachedData))
   745  		return []metrics.MeasurementEnvelope{{DBName: msg.DBUniqueName, MetricName: msg.MetricName, Data: cachedData, CustomTags: md.CustomTags,
   746  			MetricDef: mvp, RealDbname: dbSettings.RealDbname, SystemIdentifier: dbSettings.SystemIdentifier}}, nil
   747  	}
   748  	return []metrics.MeasurementEnvelope{{DBName: msg.DBUniqueName, MetricName: msg.MetricName, Data: data, CustomTags: md.CustomTags,
   749  		MetricDef: mvp, RealDbname: dbSettings.RealDbname, SystemIdentifier: dbSettings.SystemIdentifier}}, nil
   750  
   751  }

   753  var pgBouncerNumericCountersStartVersion = 01_12_00 // pgBouncer changed internal counters data type in v1.12

   755  func FilterPgbouncerData(ctx context.Context, data metrics.Measurements, databaseToKeep string, vme MonitoredDatabaseSettings) metrics.Measurements {
   756  	filteredData := make(metrics.Measurements, 0)
   757  
   758  	for _, dr := range data {
   759  		//log.Debugf("bouncer dr: %+v", dr)
   760  		if _, ok := dr["database"]; !ok {
   761  			log.GetLogger(ctx).Warning("Expected 'database' key not found from pgbouncer_stats, not storing data")
   762  			continue
   763  		}
   764  		if (len(databaseToKeep) > 0 && dr["database"] != databaseToKeep) || dr["database"] == "pgbouncer" { // always ignore the internal 'pgbouncer' DB
   765  			log.GetLogger(ctx).Debugf("Skipping bouncer stats for pool entry %v as not the specified DBName of %s", dr["database"], databaseToKeep)
   766  			continue // and all others also if a DB / pool name was specified in config
   767  		}
   768  
   769  		dr["tag_database"] = dr["database"] // support multiple databases / pools via tags if DbName left empty
   770  		delete(dr, "database")              // remove the original pool name
   771  
   772  		if vme.Version >= pgBouncerNumericCountersStartVersion { // v1.12 counters are of type numeric instead of int64
   773  			for k, v := range dr {
   774  				if k == "tag_database" {
   775  					continue
   776  				}
   777  				decimalCounter, err := decimal.NewFromString(string(v.([]uint8)))
   778  				if err != nil {
   779  					log.GetLogger(ctx).Errorf("Could not parse \"%+v\" to Decimal: %s", string(v.([]uint8)), err)
   780  					return filteredData
   781  				}
   782  				dr[k] = decimalCounter.IntPart() // technically could cause overflow...but highly unlikely for 2^63
   783  			}
   784  		}
   785  		filteredData = append(filteredData, dr)
   786  	}
   787  
   788  	return filteredData
   789  }
