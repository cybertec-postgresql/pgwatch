    66  // Reap() starts the main monitoring loop. It is responsible for fetching metrics measurements
    67  // from the sources and storing them to the sinks. It also manages the lifecycle of
    68  // the metric gatherers. In case of a source or metric definition change, it will
    69  // start or stop the gatherers accordingly.
    70  func (r *Reaper) Reap(ctx context.Context) {
    71  	var err error
    72  	logger := r.logger
    73  
    74  	go r.WriteMeasurements(ctx)
    75  	go r.WriteMonitoredSources(ctx)
    76  
    77  	r.ready.Store(true)
    78  
    79  	for { //main loop
    80  		if r.Logging.LogLevel == "debug" {
    81  			r.PrintMemStats()
    82  		}
    83  		if err = r.LoadSources(); err != nil {
    84  			logger.WithError(err).Error("could not refresh active sources, using last valid cache")
    85  		}
    86  		if err = r.LoadMetrics(); err != nil {
    87  			logger.WithError(err).Error("could not refresh metric definitions, using last valid cache")
    88  		}
    89  
    90  		// UpdateMonitoredDBCache(r.monitoredSources)
    91  		hostsToShutDownDueToRoleChange := make(map[string]bool) // hosts went from master to standby and have "only if master" set
    92  		for _, monitoredSource := range r.monitoredSources {
    93  			srcL := logger.WithField("source", monitoredSource.Name)
    94  
    95  			if monitoredSource.Connect(ctx, r.Sources) != nil {
    96  				srcL.WithError(err).Warning("could not init connection, retrying on next iteration")
    97  				continue
    98  			}
    99  
   100  			if err = monitoredSource.FetchRuntimeInfo(ctx, true); err != nil {
   101  				srcL.WithError(err).Error("could not start metric gathering")
   102  				continue
   103  			}
   104  			srcL.WithField("recovery", monitoredSource.IsInRecovery).Infof("Connect OK. Version: %s", monitoredSource.VersionStr)
   105  			if monitoredSource.IsInRecovery && monitoredSource.OnlyIfMaster {
   106  				srcL.Info("not added to monitoring due to 'master only' property")
   107  				continue
   108  			}
   109  
   110  			if monitoredSource.IsInRecovery && len(monitoredSource.MetricsStandby) > 0 {
   111  				metricsConfig = monitoredSource.MetricsStandby
   112  			} else {
   113  				metricsConfig = monitoredSource.Metrics
   114  			}
   115  			hostLastKnownStatusInRecovery[monitoredSource.Name] = monitoredSource.IsInRecovery
   116  
   117  			r.CreateSourceHelpers(ctx, srcL, monitoredSource)
   118  
   119  			if monitoredSource.IsPostgresSource() {
   120  				if r.Sources.MinDbSizeMB >= 8 { // an empty DB is a bit less than 8MB
   121  					DBSizeMB := monitoredSource.ApproxDbSize / 1048576 // only remove from monitoring when we're certain it's under the threshold
   122  					if DBSizeMB != 0 && DBSizeMB < r.Sources.MinDbSizeMB {
   123  						srcL.Infof("ignored due to the --min-db-size-mb filter, current size %d MB", DBSizeMB)
   124  						hostsToShutDownDueToRoleChange[monitoredSource.Name] = true // for the case when DB size was previosly above the threshold
   125  						continue
   126  					}
   127  				}
   128  
   129  				lastKnownStatusInRecovery := hostLastKnownStatusInRecovery[monitoredSource.Name]
   130  				if monitoredSource.IsInRecovery && monitoredSource.OnlyIfMaster {
   131  					srcL.Info("to be removed from monitoring due to 'master only' property and status change")
   132  					hostsToShutDownDueToRoleChange[monitoredSource.Name] = true
   133  					continue
   134  				} else if lastKnownStatusInRecovery != monitoredSource.IsInRecovery {
   135  					if monitoredSource.IsInRecovery && len(monitoredSource.MetricsStandby) > 0 {
   136  						srcL.Warning("Switching metrics collection to standby config...")
   137  						metricsConfig = monitoredSource.MetricsStandby
   138  						hostLastKnownStatusInRecovery[monitoredSource.Name] = true
   139  					} else {
   140  						srcL.Warning("Switching metrics collection to primary config...")
   141  						metricsConfig = monitoredSource.Metrics
   142  						hostLastKnownStatusInRecovery[monitoredSource.Name] = false
   143  					}
   144  				}
   145  
   146  			}
   147  
   148  			for metricName, interval := range metricsConfig {
   149  				metric := metricName
   150  				metricDefExists := false
   151  				var mvp metrics.Metric
   152  
   153  				if strings.HasPrefix(metric, recoPrefix) {
   154  					metric = recoMetricName
   155  					metricDefExists = true
   156  				} else {
   157  					mvp, metricDefExists = metricDefs.GetMetricDef(metric)
   158  				}
   159  
   160  				dbMetric := monitoredSource.Name + dbMetricJoinStr + metric
   161  				_, cancelFuncExists := r.cancelFuncs[dbMetric]
   162  
   163  				if metricDefExists && !cancelFuncExists { // initialize a new per db/per metric control channel
   164  					if interval > 0 {
   165  						srcL.WithField("metric", metric).WithField("interval", interval).Info("starting gatherer")
   166  						metricCtx, cancelFunc := context.WithCancel(ctx)
   167  						r.cancelFuncs[dbMetric] = cancelFunc
   168  
   169  						metricNameForStorage := metricName
   170  						if _, isSpecialMetric := specialMetrics[metricName]; !isSpecialMetric && mvp.StorageName > "" {
   171  							metricNameForStorage = mvp.StorageName
   172  						}
   173  
   174  						if err := r.SinksWriter.SyncMetric(monitoredSource.Name, metricNameForStorage, sinks.AddOp); err != nil {
   175  							srcL.Error(err)
   176  						}
   177  
   178  						go r.reapMetricMeasurements(metricCtx, monitoredSource, metric)
   179  					}
   180  				} else if (!metricDefExists && cancelFuncExists) || interval <= 0 {
   181  					// metric definition files were recently removed or interval set to zero
   182  					if cancelFunc, isOk := r.cancelFuncs[dbMetric]; isOk {
   183  						cancelFunc()
   184  					}
   185  					srcL.WithField("metric", metric).Warning("shutting down gatherer...")
   186  					delete(r.cancelFuncs, dbMetric)
   187  				} else if !metricDefExists {
   188  					epoch, ok := lastSQLFetchError.Load(metric)
   189  					if !ok || ((time.Now().Unix() - epoch.(int64)) > 3600) { // complain only 1x per hour
   190  						srcL.WithField("metric", metric).Warning("metric definition not found")
   191  						lastSQLFetchError.Store(metric, time.Now().Unix())
   192  					}
   193  				}
   194  			}
   195  		}
   196  
   197  		r.ShutdownOldWorkers(ctx, hostsToShutDownDueToRoleChange)
   198  
   199  		r.prevLoopMonitoredDBs = slices.Clone(r.monitoredSources)
   200  		select {
   201  		case <-time.After(time.Second * time.Duration(r.Sources.Refresh)):
   202  			logger.Debugf("wake up after %d seconds", r.Sources.Refresh)
   203  		case <-ctx.Done():
   204  			return
   205  		}
   206  	}
   207  }

   209  // CreateSourceHelpers creates the extensions and metric helpers for the monitored source
   210  func (r *Reaper) CreateSourceHelpers(ctx context.Context, srcL log.Logger, monitoredSource *sources.SourceConn) {
   211  	if r.prevLoopMonitoredDBs.GetMonitoredDatabase(monitoredSource.Name) != nil {
   212  		return // already created
   213  	}
   214  	if !monitoredSource.IsPostgresSource() || monitoredSource.IsInRecovery {
   215  		return // no need to create anything for non-postgres sources
   216  	}
   217  
   218  	if r.Sources.TryCreateListedExtsIfMissing > "" {
   219  		srcL.Info("trying to create extensions if missing")
   220  		extsToCreate := strings.Split(r.Sources.TryCreateListedExtsIfMissing, ",")
   221  		extsCreated := TryCreateMissingExtensions(ctx, monitoredSource, extsToCreate, monitoredSource.Extensions)
   222  		srcL.Infof("%d/%d extensions created based on --try-create-listed-exts-if-missing input %v", len(extsCreated), len(extsToCreate), extsCreated)
   223  	}
   224  
   225  	if r.Sources.CreateHelpers {
   226  		srcL.Info("trying to create helper objects if missing")
   227  		if err := TryCreateMetricsFetchingHelpers(ctx, monitoredSource); err != nil {
   228  			srcL.WithError(err).Warning("failed to create helper functions")
   229  		}
   230  	}

   234  func (r *Reaper) ShutdownOldWorkers(ctx context.Context, hostsToShutDownDueToRoleChange map[string]bool) {
   235  	logger := r.logger
   236  	// loop over existing channels and stop workers if DB or metric removed from config
   237  	// or state change makes it uninteresting
   238  	logger.Debug("checking if any workers need to be shut down...")
   239  	for dbMetric, cancelFunc := range r.cancelFuncs {
   240  		var currentMetricConfig map[string]float64
   241  		var md *sources.SourceConn
   242  		var dbRemovedFromConfig bool
   243  		singleMetricDisabled := false
   244  		splits := strings.Split(dbMetric, dbMetricJoinStr)
   245  		db := splits[0]
   246  		metric := splits[1]
   247  
   248  		_, wholeDbShutDownDueToRoleChange := hostsToShutDownDueToRoleChange[db]
   249  		if !wholeDbShutDownDueToRoleChange {
   250  			md = r.monitoredSources.GetMonitoredDatabase(db)
   251  			if md == nil { // normal removing of DB from config
   252  				dbRemovedFromConfig = true
   253  				logger.Debugf("DB %s removed from config, shutting down all metric worker processes...", db)
   254  			}
   255  		}
   256  
   257  		if !(wholeDbShutDownDueToRoleChange || dbRemovedFromConfig) { // maybe some single metric was disabled
   258  			if md.IsInRecovery && len(md.MetricsStandby) > 0 {
   259  				currentMetricConfig = md.MetricsStandby
   260  			} else {
   261  				currentMetricConfig = md.Metrics
   262  			}
   263  			interval, isMetricActive := currentMetricConfig[metric]
   264  			singleMetricDisabled = !isMetricActive || interval <= 0
   265  		}
   266  
   267  		if ctx.Err() != nil || wholeDbShutDownDueToRoleChange || dbRemovedFromConfig || singleMetricDisabled {
   268  			logger.WithField("source", db).WithField("metric", metric).Info("stopping gatherer...")
   269  			cancelFunc()
   270  			delete(r.cancelFuncs, dbMetric)
   271  			if err := r.SinksWriter.SyncMetric(db, metric, sinks.DeleteOp); err != nil {
   272  				logger.Error(err)
   273  			}
   274  		}
   275  	}
   276  
   277  	// Destroy conn pools and metric writers
   278  	r.CloseResourcesForRemovedMonitoredDBs(hostsToShutDownDueToRoleChange)
   279  }

   281  func (r *Reaper) reapMetricMeasurements(ctx context.Context, md *sources.SourceConn, metricName string) {
   282  	hostState := make(map[string]map[string]string)
   283  	var lastUptimeS int64 = -1 // used for "server restarted" event detection
   284  	var lastErrorNotificationTime time.Time
   285  	var err error
   286  	var ok bool
   287  
   288  	failedFetches := 0
   289  	lastDBVersionFetchTime := time.Unix(0, 0) // check DB ver. ev. 5 min
   290  
   291  	l := r.logger.WithField("source", md.Name).WithField("metric", metricName)
   292  	if metricName == specialMetricServerLogEventCounts {
   293  		metrics.ParseLogs(ctx, md, md.RealDbname, md.GetMetricInterval(metricName), r.measurementCh)
   294  		return
   295  	}
   296  
   297  	for {
   298  		interval := md.GetMetricInterval(metricName)
   299  		if lastDBVersionFetchTime.Add(time.Minute * time.Duration(5)).Before(time.Now()) {
   300  			// in case of errors just ignore metric "disabled" time ranges
   301  			if err = md.FetchRuntimeInfo(ctx, false); err != nil {
   302  				lastDBVersionFetchTime = time.Now()
   303  			}
   304  
   305  			if _, ok = metricDefs.GetMetricDef(metricName); !ok {
   306  				l.Error("Could not get metric version properties")
   307  				return
   308  			}
   309  		}
   310  
   311  		var metricStoreMessages *metrics.MeasurementEnvelope
   312  
   313  		// 1st try local overrides for some metrics if operating in push mode
   314  		if r.Metrics.DirectOSStats && IsDirectlyFetchableMetric(metricName) {
   315  			metricStoreMessages, err = r.FetchStatsDirectlyFromOS(ctx, md, metricName)
   316  			if err != nil {
   317  				l.WithError(err).Errorf("Could not reader metric directly from OS")
   318  			}
   319  		}
   320  		t1 := time.Now()
   321  		if metricStoreMessages == nil {
   322  			metricStoreMessages, err = r.FetchMetric(ctx, md, metricName, hostState)
   323  		}
   324  
   325  		if time.Since(t1) > (time.Second * time.Duration(interval)) {
   326  			l.Warningf("Total fetching time of %v bigger than %vs interval", time.Since(t1), interval)
   327  		}
   328  
   329  		if err != nil {
   330  			failedFetches++
   331  			// complain only 1x per 10min per host/metric...
   332  			if time.Since(lastErrorNotificationTime) > time.Minute*10 {
   333  				l.WithError(err).WithField("count", failedFetches).Error("failed to fetch metric data")
   334  				lastErrorNotificationTime = time.Now()
   335  			}
   336  		} else if metricStoreMessages != nil && len(metricStoreMessages.Data) > 0 {
   337  			r.measurementCh <- *metricStoreMessages
   338  			// pick up "server restarted" events here to avoid doing extra selects from CheckForPGObjectChangesAndStore code
   339  			if metricName == "db_stats" {
   340  				postmasterUptimeS, ok := (metricStoreMessages.Data)[0]["postmaster_uptime_s"]
   341  				if ok {
   342  					if lastUptimeS != -1 {
   343  						if postmasterUptimeS.(int64) < lastUptimeS { // restart (or possibly also failover when host is routed) happened
   344  							message := "Detected server restart (or failover)"
   345  							l.Warning(message)
   346  							detectedChangesSummary := make(metrics.Measurements, 0)
   347  							entry := metrics.NewMeasurement(metricStoreMessages.Data.GetEpoch())
   348  							entry["details"] = message
   349  							detectedChangesSummary = append(detectedChangesSummary, entry)
   350  							r.measurementCh <- metrics.MeasurementEnvelope{
   351  								DBName:     md.Name,
   352  								SourceType: string(md.Kind),
   353  								MetricName: "object_changes",
   354  								Data:       detectedChangesSummary,
   355  								CustomTags: metricStoreMessages.CustomTags,
   356  							}
   357  						}
   358  					}
   359  					lastUptimeS = postmasterUptimeS.(int64)
   360  				}
   361  			}
   362  		}
   363  
   364  		select {
   365  		case <-ctx.Done():
   366  			return
   367  		case <-time.After(time.Second * time.Duration(interval)):
   368  			// continue
   369  		}
   370  	}
   371  }

   373  // LoadSources loads sources from the reader
   374  func (r *Reaper) LoadSources() (err error) {
   375  	if DoesEmergencyTriggerfileExist(r.Metrics.EmergencyPauseTriggerfile) {
   376  		r.logger.Warningf("Emergency pause triggerfile detected at %s, ignoring currently configured DBs", r.Metrics.EmergencyPauseTriggerfile)
   377  		r.monitoredSources = make([]*sources.SourceConn, 0)
   378  		return nil
   379  	}
   380  	var newSrcs sources.SourceConns
   381  	if newSrcs, err = r.monitoredSources.SyncFromReader(r.SourcesReaderWriter); err != nil {
   382  		return err
   383  	}
   384  	for i, newMD := range newSrcs {
   385  		md := r.monitoredSources.GetMonitoredDatabase(newMD.Name)
   386  		if md == nil {
   387  			continue
   388  		}
   389  		if md.Equal(newMD.Source) {
   390  			// replace with the existing connection if the source is the same
   391  			newSrcs[i] = md
   392  			continue
   393  		}
   394  	}
   395  	r.monitoredSources = newSrcs
   396  	r.logger.WithField("sources", len(r.monitoredSources)).Info("sources refreshed")
   397  	return nil
   398  }

   407  				db := metrics.NewMeasurement(now)

   409  				db["master_only"] = mdb.OnlyIfMaster

   412  				}

   414  					DBName:     mdb.Name,
   415  					MetricName: monitoredDbsDatastoreSyncMetricName,
   416  					Data:       metrics.Measurements{db},
   417  				}
   418  			}
   419  		}
   420  		select {
   421  		case <-time.After(time.Second * monitoredDbsDatastoreSyncIntervalSeconds):
   422  			// continue
   423  		case <-ctx.Done():
   424  			return
   425  		}
   426  	}

   455  func (r *Reaper) FetchMetric(ctx context.Context, md *sources.SourceConn, metricName string, hostState map[string]map[string]string) (_ *metrics.MeasurementEnvelope, err error) {
   456  	var sql string
   457  	var data metrics.Measurements
   458  	var metric metrics.Metric
   459  	var fromCache bool
   460  	var cacheKey string
   461  	var ok bool
   462  
   463  	if metric, ok = metricDefs.GetMetricDef(metricName); !ok {
   464  		return nil, metrics.ErrMetricNotFound
   465  	}
   466  	l := r.logger.WithField("source", md.Name).WithField("metric", metricName)
   467  
   468  	if metric.IsInstanceLevel && r.Metrics.InstanceLevelCacheMaxSeconds > 0 && time.Second*time.Duration(md.GetMetricInterval(metricName)) < r.Metrics.CacheAge() {
   469  		cacheKey = fmt.Sprintf("%s:%s", md.GetClusterIdentifier(), metricName)
   470  	}
   471  	if data = r.measurementCache.Get(cacheKey, r.Metrics.CacheAge()); len(data) > 0 {
   472  		fromCache = true
   473  		goto send_to_storageChannel
   474  	}
   475  
   476  	sql = metric.GetSQL(md.Version)
   477  	if sql == "" && !(metricName == specialMetricChangeEvents || metricName == recoMetricName) {
   478  		l.Warning("Ignoring fetching because of empty SQL")
   479  		return nil, nil
   480  	}
   481  
   482  	if (metric.PrimaryOnly() && md.IsInRecovery) || (metric.StandbyOnly() && !md.IsInRecovery) {
   483  		l.Debug("Skipping fetching of as server in wrong IsInRecovery: ", md.IsInRecovery)
   484  		return nil, nil
   485  	}
   486  
   487  	switch metricName {
   488  	case specialMetricChangeEvents:
   489  		r.CheckForPGObjectChangesAndStore(ctx, md.Name, md, hostState) // TODO no hostState for Prometheus currently
   490  		return nil, nil
   491  	case recoMetricName:
   492  		if data, err = GetRecommendations(ctx, md); err != nil {
   493  			return nil, err
   494  		}
   495  	default:
   496  		if data, err = QueryMeasurements(ctx, md, sql); err != nil {
   497  			// let's soften errors to "info" from functions that expect the server to be a primary to reduce noise
   498  			if strings.Contains(err.Error(), "recovery is in progress") && md.IsInRecovery {
   499  				l.Debugf("[%s:%s] failed to fetch metrics: %s", md.Name, metricName, err)
   500  				return nil, err
   501  			}
   502  			if metricName == specialMetricInstanceUp {
