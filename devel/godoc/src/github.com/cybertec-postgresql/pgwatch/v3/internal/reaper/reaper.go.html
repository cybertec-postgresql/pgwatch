    53  // Reap() starts the main monitoring loop. It is responsible for fetching metrics measurements
    54  // from the sources and storing them to the sinks. It also manages the lifecycle of
    55  // the metric gatherers. In case of a source or metric definition change, it will
    56  // start or stop the gatherers accordingly.
    57  func (r *Reaper) Reap(ctx context.Context) (err error) {
    58  
    59  	logger := r.logger
    60  
    61  	go r.WriteMeasurements(ctx)
    62  	go r.WriteMonitoredSources(ctx)
    63  
    64  	r.ready.Store(true)
    65  
    66  	for { //main loop
    67  		if err = r.LoadSources(); err != nil {
    68  			logger.WithError(err).Error("could not refresh active sources, using last valid cache")
    69  		}
    70  		if err = r.LoadMetrics(); err != nil {
    71  			logger.WithError(err).Error("could not refresh metric definitions, using last valid cache")
    72  		}
    73  
    74  		UpdateMonitoredDBCache(r.monitoredSources)
    75  		hostsToShutDownDueToRoleChange := make(map[string]bool) // hosts went from master to standby and have "only if master" set
    76  		for _, monitoredSource := range r.monitoredSources {
    77  			srcL := logger.WithField("source", monitoredSource.Name)
    78  
    79  			if monitoredSource.Connect(ctx, r.Sources) != nil {
    80  				srcL.WithError(err).Warning("could not init connection, retrying on next iteration")
    81  				continue
    82  			}
    83  
    84  			if err = monitoredSource.FetchRuntimeInfo(ctx, true); err != nil {
    85  				srcL.WithError(err).Error("could not start metric gathering")
    86  				continue
    87  			}
    88  			srcL.WithField("recovery", monitoredSource.IsInRecovery).Infof("Connect OK. Version: %s", monitoredSource.VersionStr)
    89  			if monitoredSource.IsInRecovery && monitoredSource.OnlyIfMaster {
    90  				srcL.Info("not added to monitoring due to 'master only' property")
    91  				continue
    92  			}
    93  
    94  			if monitoredSource.IsInRecovery && len(monitoredSource.MetricsStandby) > 0 {
    95  				metricsConfig = monitoredSource.MetricsStandby
    96  			} else {
    97  				metricsConfig = monitoredSource.Metrics
    98  			}
    99  			hostLastKnownStatusInRecovery[monitoredSource.Name] = monitoredSource.IsInRecovery
   100  
   101  			r.CreateSourceHelpers(ctx, srcL, monitoredSource)
   102  
   103  			if monitoredSource.IsPostgresSource() {
   104  				var DBSizeMB int64
   105  
   106  				if r.Sources.MinDbSizeMB >= 8 { // an empty DB is a bit less than 8MB
   107  					DBSizeMB, _ = DBGetSizeMB(ctx, monitoredSource.Name) // ignore errors, i.e. only remove from monitoring when we're certain it's under the threshold
   108  					if DBSizeMB != 0 {
   109  						if DBSizeMB < r.Sources.MinDbSizeMB {
   110  							srcL.Infof("ignored due to the --min-db-size-mb filter, current size %d MB", DBSizeMB)
   111  							hostsToShutDownDueToRoleChange[monitoredSource.Name] = true // for the case when DB size was previosly above the threshold
   112  							continue
   113  						}
   114  					}
   115  				}
   116  
   117  				lastKnownStatusInRecovery := hostLastKnownStatusInRecovery[monitoredSource.Name]
   118  				if monitoredSource.IsInRecovery && monitoredSource.OnlyIfMaster {
   119  					srcL.Info("to be removed from monitoring due to 'master only' property and status change")
   120  					hostsToShutDownDueToRoleChange[monitoredSource.Name] = true
   121  					continue
   122  				} else if lastKnownStatusInRecovery != monitoredSource.IsInRecovery {
   123  					if monitoredSource.IsInRecovery && len(monitoredSource.MetricsStandby) > 0 {
   124  						srcL.Warning("Switching metrics collection to standby config...")
   125  						metricsConfig = monitoredSource.MetricsStandby
   126  						hostLastKnownStatusInRecovery[monitoredSource.Name] = true
   127  					} else {
   128  						srcL.Warning("Switching metrics collection to primary config...")
   129  						metricsConfig = monitoredSource.Metrics
   130  						hostLastKnownStatusInRecovery[monitoredSource.Name] = false
   131  					}
   132  				}
   133  
   134  			}
   135  
   136  			for metricName, interval := range metricsConfig {
   137  				metric := metricName
   138  				metricDefExists := false
   139  				var mvp metrics.Metric
   140  
   141  				if strings.HasPrefix(metric, recoPrefix) {
   142  					metric = recoMetricName
   143  					metricDefExists = true
   144  				} else {
   145  					mvp, metricDefExists = metricDefs.GetMetricDef(metric)
   146  				}
   147  
   148  				dbMetric := monitoredSource.Name + dbMetricJoinStr + metric
   149  				_, cancelFuncExists := r.cancelFuncs[dbMetric]
   150  
   151  				if metricDefExists && !cancelFuncExists { // initialize a new per db/per metric control channel
   152  					if interval > 0 {
   153  						srcL.WithField("metric", metric).WithField("interval", interval).Info("starting gatherer")
   154  						metricCtx, cancelFunc := context.WithCancel(ctx)
   155  						r.cancelFuncs[dbMetric] = cancelFunc
   156  
   157  						metricNameForStorage := metricName
   158  						if _, isSpecialMetric := specialMetrics[metricName]; !isSpecialMetric && mvp.StorageName > "" {
   159  							metricNameForStorage = mvp.StorageName
   160  						}
   161  
   162  						if err := r.SinksWriter.SyncMetric(monitoredSource.Name, metricNameForStorage, "add"); err != nil {
   163  							srcL.Error(err)
   164  						}
   165  
   166  						go r.reapMetricMeasurements(metricCtx, monitoredSource, metric)
   167  					}
   168  				} else if (!metricDefExists && cancelFuncExists) || interval <= 0 {
   169  					// metric definition files were recently removed or interval set to zero
   170  					if cancelFunc, isOk := r.cancelFuncs[dbMetric]; isOk {
   171  						cancelFunc()
   172  					}
   173  					srcL.WithField("metric", metric).Warning("shutting down gatherer...")
   174  					delete(r.cancelFuncs, dbMetric)
   175  				} else if !metricDefExists {
   176  					epoch, ok := lastSQLFetchError.Load(metric)
   177  					if !ok || ((time.Now().Unix() - epoch.(int64)) > 3600) { // complain only 1x per hour
   178  						srcL.WithField("metric", metric).Warning("metric definition not found")
   179  						lastSQLFetchError.Store(metric, time.Now().Unix())
   180  					}
   181  				}
   182  			}
   183  		}
   184  
   185  		r.ShutdownOldWorkers(ctx, hostsToShutDownDueToRoleChange)
   186  
   187  		r.prevLoopMonitoredDBs = slices.Clone(r.monitoredSources)
   188  		select {
   189  		case <-time.After(time.Second * time.Duration(r.Sources.Refresh)):
   190  			logger.Debugf("wake up after %d seconds", r.Sources.Refresh)
   191  		case <-ctx.Done():
   192  			return
   193  		}
   194  	}
   195  }

   198  func (r *Reaper) CreateSourceHelpers(ctx context.Context, srcL log.LoggerIface, monitoredSource *sources.SourceConn) {
   199  	if r.prevLoopMonitoredDBs.GetMonitoredDatabase(monitoredSource.Name) != nil {

   211  	}
   212  
   213  	if r.Sources.CreateHelpers {
   214  		srcL.Info("trying to create helper objects if missing")
   215  		if err := TryCreateMetricsFetchingHelpers(ctx, monitoredSource); err != nil {
   216  			srcL.WithError(err).Warning("failed to create helper functions")
   217  		}
   218  	}

   222  func (r *Reaper) ShutdownOldWorkers(ctx context.Context, hostsToShutDownDueToRoleChange map[string]bool) {
   223  	logger := r.logger
   224  	// loop over existing channels and stop workers if DB or metric removed from config
   225  	// or state change makes it uninteresting
   226  	logger.Debug("checking if any workers need to be shut down...")
   227  	for dbMetric, cancelFunc := range r.cancelFuncs {
   228  		var currentMetricConfig map[string]float64
   229  		var md *sources.SourceConn
   230  		var ok, dbRemovedFromConfig bool
   231  		singleMetricDisabled := false
   232  		splits := strings.Split(dbMetric, dbMetricJoinStr)
   233  		db := splits[0]
   234  		metric := splits[1]
   235  
   236  		_, wholeDbShutDownDueToRoleChange := hostsToShutDownDueToRoleChange[db]
   237  		if !wholeDbShutDownDueToRoleChange {
   238  			monitoredDbCacheLock.RLock()
   239  			md, ok = monitoredDbCache[db]
   240  			monitoredDbCacheLock.RUnlock()
   241  			if !ok { // normal removing of DB from config
   242  				dbRemovedFromConfig = true
   243  				logger.Debugf("DB %s removed from config, shutting down all metric worker processes...", db)
   244  			}
   245  		}
   246  
   247  		if !(wholeDbShutDownDueToRoleChange || dbRemovedFromConfig) { // maybe some single metric was disabled
   248  			if md.IsInRecovery && len(md.MetricsStandby) > 0 {
   249  				currentMetricConfig = md.MetricsStandby
   250  			} else {
   251  				currentMetricConfig = md.Metrics
   252  			}
   253  			interval, isMetricActive := currentMetricConfig[metric]
   254  			singleMetricDisabled = !isMetricActive || interval <= 0
   255  		}
   256  
   257  		if ctx.Err() != nil || wholeDbShutDownDueToRoleChange || dbRemovedFromConfig || singleMetricDisabled {
   258  			logger.WithField("source", db).WithField("metric", metric).Info("stopping gatherer...")
   259  			cancelFunc()
   260  			delete(r.cancelFuncs, dbMetric)
   261  			if err := r.SinksWriter.SyncMetric(db, metric, "remove"); err != nil {
   262  				logger.Error(err)
   263  			}
   264  		}
   265  	}
   266  
   267  	// Destroy conn pools and metric writers
   268  	r.CloseResourcesForRemovedMonitoredDBs(hostsToShutDownDueToRoleChange)
   269  }

   272  func (r *Reaper) reapMetricMeasurements(ctx context.Context, md *sources.SourceConn, metricName string) {
   273  	hostState := make(map[string]map[string]string)
   274  	var lastUptimeS int64 = -1 // used for "server restarted" event detection
   275  	var lastErrorNotificationTime time.Time
   276  	// var vme MonitoredDatabaseSettings
   277  	var mvp metrics.Metric
   278  	var err error
   279  	var ok bool
   280  	var envelopes []metrics.MeasurementEnvelope
   281  
   282  	failedFetches := 0
   283  	lastDBVersionFetchTime := time.Unix(0, 0) // check DB ver. ev. 5 min
   284  
   285  	l := r.logger.WithField("source", md.Name).WithField("metric", metricName)
   286  	if metricName == specialMetricServerLogEventCounts {
   287  		metrics.ParseLogs(ctx, md, md.RealDbname, md.GetMetricInterval(metricName), r.measurementCh) // no return
   288  		return
   289  	}
   290  
   291  	for {
   292  		interval := md.GetMetricInterval(metricName)
   293  		if lastDBVersionFetchTime.Add(time.Minute * time.Duration(5)).Before(time.Now()) {
   294  			// in case of errors just ignore metric "disabled" time ranges
   295  			if err = md.FetchRuntimeInfo(ctx, false); err != nil {
   296  				lastDBVersionFetchTime = time.Now()
   297  			}
   298  
   299  			if mvp, ok = metricDefs.GetMetricDef(metricName); !ok {
   300  				l.Errorf("Could not get metric version properties: %s", metricName)
   301  				return
   302  			}
   303  		}
   304  
   305  		var metricStoreMessages *metrics.MeasurementEnvelope
   306  		mfm := MetricFetchConfig{
   307  			DBUniqueName:        md.Name,
   308  			DBUniqueNameOrig:    md.GetDatabaseName(),
   309  			MetricName:          metricName,
   310  			Source:              md.Kind,
   311  			Interval:            time.Second * time.Duration(interval),
   312  			StmtTimeoutOverride: 0,
   313  		}
   314  
   315  		// 1st try local overrides for some metrics if operating in push mode
   316  		if r.Metrics.DirectOSStats && IsDirectlyFetchableMetric(metricName) {
   317  			metricStoreMessages, err = FetchStatsDirectlyFromOS(ctx, mfm, md, mvp)
   318  			if err != nil {
   319  				l.WithError(err).Errorf("Could not reader metric directly from OS")
   320  			}
   321  		}
   322  		t1 := time.Now()
   323  		if metricStoreMessages == nil {
   324  			metricStoreMessages, err = r.FetchMetric(ctx, mfm, hostState)
   325  		}
   326  		t2 := time.Now()
   327  
   328  		if t2.Sub(t1) > (time.Second * time.Duration(interval)) {
   329  			l.Warningf("Total fetching time of %vs bigger than %vs interval", t2.Sub(t1).Truncate(time.Millisecond*100).Seconds(), interval)
   330  		}
   331  
   332  		if err != nil {
   333  			failedFetches++
   334  			// complain only 1x per 10min per host/metric...
   335  			if lastErrorNotificationTime.IsZero() || lastErrorNotificationTime.Add(time.Second*time.Duration(600)).Before(time.Now()) {
   336  				l.WithError(err).Error("failed to fetch metric data")
   337  				if failedFetches > 1 {
   338  					l.Errorf("Total failed fetches: %d", failedFetches)
   339  				}
   340  				lastErrorNotificationTime = time.Now()
   341  			}
   342  		} else if metricStoreMessages != nil && len(metricStoreMessages.Data) > 0 {
   343  			envelopes = append(envelopes, *metricStoreMessages)
   344  			// pick up "server restarted" events here to avoid doing extra selects from CheckForPGObjectChangesAndStore code
   345  			if metricName == "db_stats" {
   346  				postmasterUptimeS, ok := (metricStoreMessages.Data)[0]["postmaster_uptime_s"]
   347  				if ok {
   348  					if lastUptimeS != -1 {
   349  						if postmasterUptimeS.(int64) < lastUptimeS { // restart (or possibly also failover when host is routed) happened
   350  							message := "Detected server restart (or failover) of \"" + md.Name + "\""
   351  							l.Warning(message)
   352  							detectedChangesSummary := make(metrics.Measurements, 0)
   353  							entry := metrics.NewMeasurement(metricStoreMessages.Data.GetEpoch())
   354  							entry["details"] = message
   355  							detectedChangesSummary = append(detectedChangesSummary, entry)
   356  							envelopes = append(envelopes,
   357  								metrics.MeasurementEnvelope{
   358  									DBName:     md.Name,
   359  									SourceType: string(md.Kind),
   360  									MetricName: "object_changes",
   361  									Data:       detectedChangesSummary,
   362  									CustomTags: metricStoreMessages.CustomTags,
   363  								})
   364  						}
   365  					}
   366  					lastUptimeS = postmasterUptimeS.(int64)
   367  				}
   368  			}
   369  			r.measurementCh <- envelopes
   370  		}
   371  
   372  		select {
   373  		case <-ctx.Done():
   374  			return
   375  		case <-time.After(time.Second * time.Duration(interval)):
   376  			// continue
   377  		}
   378  	}
   379  }

   395  // WriteMonitoredSources writes actively monitored DBs listing to sinks
   396  // every monitoredDbsDatastoreSyncIntervalSeconds (default 10min)
   397  func (r *Reaper) WriteMonitoredSources(ctx context.Context) {
   398  	for {
   399  		if len(monitoredDbCache) > 0 {
   400  			msms := make([]metrics.MeasurementEnvelope, len(monitoredDbCache))
   401  			now := time.Now().UnixNano()
   402  
   403  			monitoredDbCacheLock.RLock()
   404  			for _, mdb := range monitoredDbCache {
   405  				db := metrics.NewMeasurement(now)
   406  				db["tag_group"] = mdb.Group
   407  				db["master_only"] = mdb.OnlyIfMaster
   408  				for k, v := range mdb.CustomTags {
   409  					db[metrics.TagPrefix+k] = v
   410  				}
   411  				msms = append(msms, metrics.MeasurementEnvelope{
   412  					DBName:     mdb.Name,
   413  					MetricName: monitoredDbsDatastoreSyncMetricName,
   414  					Data:       metrics.Measurements{db},
   415  				})
   416  			}
   417  			monitoredDbCacheLock.RUnlock()
   418  			r.measurementCh <- msms
   419  		}
   420  		select {
   421  		case <-time.After(time.Second * monitoredDbsDatastoreSyncIntervalSeconds):
   422  			// continue
   423  		case <-ctx.Done():
   424  			return
   425  		}
   426  	}

   465  func (r *Reaper) FetchMetric(ctx context.Context, msg MetricFetchConfig, hostState map[string]map[string]string) (*metrics.MeasurementEnvelope, error) {
   466  
   467  	// var dbSettings MonitoredDatabaseSettings
   468  	var dbVersion int
   469  	var err error
   470  	var sql string
   471  	var data, cachedData metrics.Measurements
   472  	var md *sources.SourceConn
   473  	var metric metrics.Metric
   474  	var fromCache bool
   475  	var cacheKey string
   476  	var ok bool
   477  
   478  	if md, err = GetMonitoredDatabaseByUniqueName(ctx, msg.DBUniqueName); err != nil {
   479  		return nil, err
   480  	}
   481  
   482  	if metric, ok = metricDefs.GetMetricDef(msg.MetricName); !ok {
   483  		return nil, metrics.ErrMetricNotFound
   484  	}
   485  
   486  	if err = md.FetchRuntimeInfo(ctx, false); err != nil {
   487  		return nil, err
   488  	}
   489  	if msg.MetricName == specialMetricDbSize || msg.MetricName == specialMetricTableStats {
   490  		if md.ExecEnv == sources.EnvAzureSingle && md.ApproxDBSizeB > 1e12 { // 1TB
   491  			subsMetricName := msg.MetricName + "_approx"
   492  			mvpApprox, ok := metricDefs.GetMetricDef(subsMetricName)
   493  			if ok && mvpApprox.StorageName == msg.MetricName {
   494  				log.GetLogger(ctx).Infof("[%s:%s] Transparently swapping metric to %s due to hard-coded rules...", msg.DBUniqueName, msg.MetricName, subsMetricName)
   495  				msg.MetricName = subsMetricName
   496  			}
   497  		}
   498  	}
   499  	dbVersion = md.Version
   500  
   501  	if msg.Source == sources.SourcePgBouncer {
   502  		dbVersion = 0 // version is 0.0 for all pgbouncer sql per convention
   503  	}
   504  
   505  	if metric.IsInstanceLevel && r.Metrics.InstanceLevelCacheMaxSeconds > 0 && msg.Interval < r.Metrics.CacheAge() {
   506  		cacheKey = fmt.Sprintf("%s:%s:%d:%s",
   507  			md.SystemIdentifier,
   508  			md.ConnConfig.ConnConfig.Host,
   509  			md.ConnConfig.ConnConfig.Port,
   510  			msg.MetricName)
   511  	}

   514  		goto send_to_storageChannel
   515  	}

   517  	sql = metric.GetSQL(dbVersion)
   518  	if sql == "" && !(msg.MetricName == specialMetricChangeEvents || msg.MetricName == recoMetricName) {

   521  		return nil, nil
   522  	}
   523  
   524  	if (metric.PrimaryOnly() && md.IsInRecovery) || (metric.StandbyOnly() && !md.IsInRecovery) {
   525  		log.GetLogger(ctx).Debugf("[%s:%s] Skipping fetching of  as server not in wanted state (IsInRecovery=%v)", msg.DBUniqueName, msg.MetricName, md.IsInRecovery)
   526  		return nil, nil
   527  	}
   528

   530  		r.CheckForPGObjectChangesAndStore(ctx, msg.DBUniqueName, md, hostState) // TODO no hostState for Prometheus currently
   531  	} else if msg.MetricName == recoMetricName {
   532  		if data, err = GetRecommendations(ctx, msg.DBUniqueName, md); err != nil {
   533  			return nil, err
   534  		}
   535  	} else if msg.Source == sources.SourcePgPool {
   536  		if data, err = FetchMetricsPgpool(ctx, msg, md, metric); err != nil {
   537  			return nil, err
   538  		}
   539  	} else {
   540  		data, err = QueryMeasurements(ctx, msg.DBUniqueName, sql)
   541  
   542  		if err != nil {
   543  			// let's soften errors to "info" from functions that expect the server to be a primary to reduce noise
   544  			if strings.Contains(err.Error(), "recovery is in progress") && md.IsInRecovery {
   545  				log.GetLogger(ctx).Debugf("[%s:%s] failed to fetch metrics: %s", msg.DBUniqueName, msg.MetricName, err)
   546  				return nil, err
   547  			}
   548  
   549  			if msg.MetricName == specialMetricInstanceUp {
   550  				log.GetLogger(ctx).WithError(err).Debugf("[%s:%s] failed to fetch metrics. marking instance as not up", msg.DBUniqueName, msg.MetricName)
   551  				data = make(metrics.Measurements, 1)
   552  				data[0] = metrics.NewMeasurement(time.Now().UnixNano())
   553  				data[0]["is_up"] = 0 // should be updated if the "instance_up" metric definition is changed
   554  				goto send_to_storageChannel
   555  			}
   556  
   557  			log.GetLogger(ctx).
   558  				WithFields(map[string]any{"source": msg.DBUniqueName, "metric": msg.MetricName}).
   559  				WithError(err).Error("failed to fetch metrics")
   560  
   561  			return nil, err
   562  		}
   563  
   564  		log.GetLogger(ctx).WithFields(map[string]any{"source": msg.DBUniqueName, "metric": msg.MetricName, "rows": len(data)}).Info("measurements fetched")
   565  	}
   566  
   567  	r.measurementCache.Put(cacheKey, data)
   568  
   569  send_to_storageChannel:
   570  
   571  	if (r.Sinks.RealDbnameField > "" || r.Sinks.SystemIdentifierField > "") && msg.Source == sources.SourcePostgres {
   572  		data = r.AddSysinfoToMeasurements(data, md)
   573  	}
   574  
   575  	if metric.StorageName != "" {
   576  		log.GetLogger(ctx).Debugf("[%s] rerouting metric %s data to %s based on metric attributes", msg.DBUniqueName, msg.MetricName, metric.StorageName)
   577  		msg.MetricName = metric.StorageName
   578  	}
   579  	if fromCache {
   580  		log.GetLogger(ctx).Infof("[%s:%s] loaded %d rows from the instance cache", msg.DBUniqueName, msg.MetricName, len(cachedData))
   581  		return &metrics.MeasurementEnvelope{DBName: msg.DBUniqueName, MetricName: msg.MetricName, Data: cachedData, CustomTags: md.CustomTags,
   582  			MetricDef: metric, RealDbname: md.RealDbname, SystemIdentifier: md.SystemIdentifier}, nil
   583  	}
   584  	return &metrics.MeasurementEnvelope{DBName: msg.DBUniqueName, MetricName: msg.MetricName, Data: data, CustomTags: md.CustomTags,
   585  		MetricDef: metric, RealDbname: md.RealDbname, SystemIdentifier: md.SystemIdentifier}, nil
