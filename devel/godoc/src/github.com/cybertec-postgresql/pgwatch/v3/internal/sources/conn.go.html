   126  // VersionToInt parses a given version and returns an integer  or
   127  // an error if unable to parse the version. Only parses valid semantic versions.
   128  // Performs checking that can find errors within the version.
   129  // Examples: v1.2 -> 01_02_00, v9.6.3 -> 09_06_03, v11 -> 11_00_00
   130  var regVer = regexp.MustCompile(`(\d+).?(\d*).?(\d*)`)
   131  
   132  func VersionToInt(version string) (v int) {
   133  	if matches := regVer.FindStringSubmatch(version); len(matches) > 1 {
   134  		for i, match := range matches[1:] {
   135  			v += func() (m int) { m, _ = strconv.Atoi(match); return }() * int(math.Pow10(4-i*2))
   136  		}
   137  	}
   138  	return
   139  }

   141  func (md *SourceConn) FetchRuntimeInfo(ctx context.Context, forceRefetch bool) (err error) {
   142  	if ctx.Err() != nil {
   143  		return ctx.Err()
   144  	}
   145  
   146  	if !forceRefetch && md.LastCheckedOn.After(time.Now().Add(time.Minute*-2)) { // use cached version for 2 min
   147  		return nil
   148  	}
   149  
   150  	dbNewSettings := RuntimeInfo{
   151  		Extensions: make(map[string]int),
   152  	}
   153  
   154  	switch md.Kind {
   155  	case SourcePgBouncer, SourcePgPool:
   156  		if dbNewSettings.VersionStr, dbNewSettings.Version, err = md.FetchVersion(ctx, func() string {
   157  			if md.Kind == SourcePgBouncer {
   158  				return "SHOW VERSION"
   159  			}
   160  			return "SHOW POOL_VERSION"
   161  		}()); err != nil {
   162  			return
   163  		}
   164  	default:
   165  		sql := `select /* pgwatch_generated */ 
   166  	div(current_setting('server_version_num')::int, 10000) as ver, 
   167  	version(), 
   168  	pg_is_in_recovery(), 
   169  	current_database()::TEXT,
   170  	system_identifier,
   171  	current_setting('is_superuser')::bool
   172  FROM
   173  	pg_control_system()`
   174  
   175  		err = md.Conn.QueryRow(ctx, sql).
   176  			Scan(&dbNewSettings.Version, &dbNewSettings.VersionStr,
   177  				&dbNewSettings.IsInRecovery, &dbNewSettings.RealDbname,
   178  				&dbNewSettings.SystemIdentifier, &dbNewSettings.IsSuperuser)
   179  		if err != nil {
   180  			return err
   181  		}
   182  
   183  		dbNewSettings.ExecEnv = md.DiscoverPlatform(ctx)
   184  		dbNewSettings.ApproxDbSize = md.FetchApproxSize(ctx)
   185  
   186  		sqlExtensions := `select /* pgwatch_generated */ extname::text, (regexp_matches(extversion, $$\d+\.?\d+?$$))[1]::text as extversion from pg_extension order by 1;`
   187  		var res pgx.Rows
   188  		res, err = md.Conn.Query(ctx, sqlExtensions)
   189  		if err == nil {
   190  			var ext string
   191  			var ver string
   192  			_, err = pgx.ForEachRow(res, []any{&ext, &ver}, func() error {
   193  				extver := VersionToInt(ver)
   194  				if extver == 0 {
   195  					return fmt.Errorf("unexpected extension %s version input: %s", ext, ver)
   196  				}
   197  				dbNewSettings.Extensions[ext] = extver
   198  				return nil
   199  			})
   200  		}
   201  
   202  	}
   203  	dbNewSettings.LastCheckedOn = time.Now()
   204  	md.RuntimeInfo = dbNewSettings // store the new settings in the struct
   205  	return err
   206  }

   216  // TryDiscoverPlatform tries to discover the platform based on the database version string and some special settings
   217  // that are only available on certain platforms. Returns the platform name or "UNKNOWN" if not sure.
   218  func (md *SourceConn) DiscoverPlatform(ctx context.Context) (platform string) {
   219  	if md.ExecEnv != "" {
   220  		return md.ExecEnv // carry over as not likely to change ever
   221  	}
   222  	sql := `select /* pgwatch_generated */
   223  	case
   224  	  when exists (select * from pg_settings where name = 'pg_qs.host_database' and setting = 'azure_sys') and version() ~* 'compiled by Visual C' then 'AZURE_SINGLE'
   225  	  when exists (select * from pg_settings where name = 'pg_qs.host_database' and setting = 'azure_sys') and version() ~* 'compiled by gcc' then 'AZURE_FLEXIBLE'
   226  	  when exists (select * from pg_settings where name = 'cloudsql.supported_extensions') then 'GOOGLE'
   227  	else
   228  	  'UNKNOWN'
   229  	end as exec_env`
   230  	_ = md.Conn.QueryRow(ctx, sql).Scan(&platform)
   231  	return
   232  }

   261  // SyncFromReader will update the monitored databases with the latest configuration from the reader.
   262  // Any resolution errors will be returned, e.g. etcd unavailability.
   263  // It's up to the caller to proceed with the databases available or stop the execution due to errors.
   264  func (mds SourceConns) SyncFromReader(r Reader) (newmds SourceConns, err error) {
   265  	srcs, err := r.GetSources()
   266  	if err != nil {
   267  		return nil, err
   268  	}
   269  	newmds, err = srcs.ResolveDatabases()
   270  	for _, newMD := range newmds {
   271  		md := mds.GetMonitoredDatabase(newMD.Name)
   272  		if md == nil {
   273  			continue
   274  		}
   275  		if reflect.DeepEqual(md.Source, newMD.Source) {
   276  			// keep the existing connection if the source is the same
   277  			newMD.Conn = md.Conn
   278  			newMD.ConnConfig = md.ConnConfig
   279  			continue
   280  		}
   281  		if md.Conn != nil {
   282  			md.Conn.Close()
   283  		}
   284  	}
   285  	return newmds, err
   286  }
