    80  	if md.Conn == nil {
    81  		if err = md.ParseConfig(); err != nil {

    87  		if opts.MaxParallelConnectionsPerDb > 0 {

    94  	}

   145  // VersionToInt parses a given version and returns an integer  or
   146  // an error if unable to parse the version. Only parses valid semantic versions.
   147  // Performs checking that can find errors within the version.
   148  // Examples: v1.2 -> 01_02_00, v9.6.3 -> 09_06_03, v11 -> 11_00_00
   149  var regVer = regexp.MustCompile(`(\d+).?(\d*).?(\d*)`)
   150  
   151  func VersionToInt(version string) (v int) {
   152  	if matches := regVer.FindStringSubmatch(version); len(matches) > 1 {
   153  		for i, match := range matches[1:] {
   154  			v += func() (m int) { m, _ = strconv.Atoi(match); return }() * int(math.Pow10(4-i*2))
   155  		}
   156  	}
   157  	return
   158  }

   160  func (md *SourceConn) FetchRuntimeInfo(ctx context.Context, forceRefetch bool) (err error) {
   161  	if ctx.Err() != nil {
   162  		return ctx.Err()
   163  	}
   164  
   165  	if !forceRefetch && md.LastCheckedOn.After(time.Now().Add(time.Minute*-2)) { // use cached version for 2 min
   166  		return nil
   167  	}
   168  	switch md.Kind {
   169  	case SourcePgBouncer, SourcePgPool:
   170  		if md.VersionStr, md.Version, err = md.FetchVersion(ctx, func() string {
   171  			if md.Kind == SourcePgBouncer {
   172  				return "SHOW VERSION"
   173  			}
   174  			return "SHOW POOL_VERSION"
   175  		}()); err != nil {
   176  			return
   177  		}
   178  	default:
   179  		sql := `select /* pgwatch_generated */ 
   180  	div(current_setting('server_version_num')::int, 10000) as ver, 
   181  	version(), 
   182  	pg_is_in_recovery(), 
   183  	current_database()::TEXT,
   184  	system_identifier,
   185  	current_setting('is_superuser')::bool
   186  FROM
   187  	pg_control_system()`
   188  
   189  		err = md.Conn.QueryRow(ctx, sql).
   190  			Scan(&md.Version, &md.VersionStr,
   191  				&md.IsInRecovery, &md.RealDbname,
   192  				&md.SystemIdentifier, &md.IsSuperuser)
   193  		if err != nil {
   194  			return err
   195  		}
   196  
   197  		md.ExecEnv = md.DiscoverPlatform(ctx)
   198  		md.ApproxDbSize = md.FetchApproxSize(ctx)
   199  
   200  		sqlExtensions := `select /* pgwatch_generated */ extname::text, (regexp_matches(extversion, $$\d+\.?\d+?$$))[1]::text as extversion from pg_extension order by 1;`
   201  		var res pgx.Rows
   202  		res, err = md.Conn.Query(ctx, sqlExtensions)
   203  		if err == nil {
   204  			var ext string
   205  			var ver string
   206  			_, err = pgx.ForEachRow(res, []any{&ext, &ver}, func() error {
   207  				extver := VersionToInt(ver)
   208  				if extver == 0 {
   209  					return fmt.Errorf("unexpected extension %s version input: %s", ext, ver)
   210  				}
   211  				md.Extensions[ext] = extver
   212  				return nil
   213  			})
   214  		}
   215  
   216  	}
   217  	md.LastCheckedOn = time.Now()
   218  	return err
   219  }

   229  // TryDiscoverPlatform tries to discover the platform based on the database version string and some special settings
   230  // that are only available on certain platforms. Returns the platform name or "UNKNOWN" if not sure.
   231  func (md *SourceConn) DiscoverPlatform(ctx context.Context) (platform string) {
   232  	if md.ExecEnv != "" {
   233  		return md.ExecEnv // carry over as not likely to change ever
   234  	}
   235  	sql := `select /* pgwatch_generated */
   236  	case
   237  	  when exists (select * from pg_settings where name = 'pg_qs.host_database' and setting = 'azure_sys') and version() ~* 'compiled by Visual C' then 'AZURE_SINGLE'
   238  	  when exists (select * from pg_settings where name = 'pg_qs.host_database' and setting = 'azure_sys') and version() ~* 'compiled by gcc' then 'AZURE_FLEXIBLE'
   239  	  when exists (select * from pg_settings where name = 'cloudsql.supported_extensions') then 'GOOGLE'
   240  	else
   241  	  'UNKNOWN'
   242  	end as exec_env`
   243  	_ = md.Conn.QueryRow(ctx, sql).Scan(&platform)
   244  	return
   245  }
