    17  func QueryMeasurements(ctx context.Context, dbUnique string, sql string, args ...any) (metrics.Measurements, error) {
    18  	var conn db.PgxIface
    19  	var md *sources.SourceConn
    20  	var err error
    21  	var tx pgx.Tx
    22  	if strings.TrimSpace(sql) == "" {
    23  		return nil, errors.New("empty SQL")
    24  	}
    25  	if md, err = GetMonitoredDatabaseByUniqueName(ctx, dbUnique); err != nil {
    26  		return nil, err
    27  	}
    28  	conn = md.Conn
    29  	if md.IsPostgresSource() {
    30  		// we don't want transaction for non-postgres sources, e.g. pgbouncer
    31  		if tx, err = conn.Begin(ctx); err != nil {
    32  			return nil, err
    33  		}
    34  		defer func() { _ = tx.Commit(ctx) }()
    35  		_, err = tx.Exec(ctx, "SET LOCAL lock_timeout TO '100ms'")
    36  		if err != nil {
    37  			return nil, err
    38  		}
    39  		conn = tx
    40  	} else {
    41  		// we want simple protocol for non-postgres connections, e.g. pgpool
    42  		args = append([]any{pgx.QueryExecModeSimpleProtocol}, args...)
    43  	}
    44  	rows, err := conn.Query(ctx, sql, args...)
    45  	if err == nil {
    46  		return pgx.CollectRows(rows, metrics.RowToMeasurement)
    47  	}
    48  	return nil, err
    49  }

    84  			}
    85  		} else { // check for new / delete
    86  			if !firstRun {
    87  				log.GetLogger(ctx).Info("detected new sproc:", dr["tag_sproc"], ", oid:", dr["tag_oid"])
    88  				dr["event"] = "create"
    89  				detectedChanges = append(detectedChanges, dr)
    90  				changeCounts.Created++
    91  			}
    92  			hostState["sproc_hashes"][objIdent] = dr["md5"].(string)
    93  		}
    94  	}
    95  	// detect deletes
    96  	if !firstRun && len(hostState["sproc_hashes"]) != len(data) {
    97  		deletedSProcs := make([]string, 0)

    99  		currentOidMap := make(map[string]bool)
   100  		for _, dr := range data {
   101  			currentOidMap[dr["tag_sproc"].(string)+dbMetricJoinStr+dr["tag_oid"].(string)] = true
   102  		}

   104  			_, ok := currentOidMap[sprocIdent]

   116  		}

   118  			delete(hostState["sproc_hashes"], deletedSProc)

   123  		storageCh <- []metrics.MeasurementEnvelope{{DBName: md.Name, MetricName: "sproc_changes", Data: detectedChanges, CustomTags: md.CustomTags}}

   283  func DetectPrivilegeChanges(ctx context.Context, md *sources.SourceConn, storageCh chan<- []metrics.MeasurementEnvelope, hostState map[string]map[string]string) ChangeDetectionResults {
   284  	detectedChanges := make(metrics.Measurements, 0)
   285  	var firstRun bool
   286  	var changeCounts ChangeDetectionResults
   287  
   288  	log.GetLogger(ctx).Debugf("[%s][%s] checking object privilege changes...", md.Name, specialMetricChangeEvents)
   289  	if _, ok := hostState["object_privileges"]; !ok {
   290  		firstRun = true
   291  		hostState["object_privileges"] = make(map[string]string)
   292  	}
   293  
   294  	mvp, ok := metricDefs.GetMetricDef("privilege_changes")
   295  	if !ok || mvp.GetSQL(int(md.Version)) == "" {
   296  		log.GetLogger(ctx).Warningf("[%s][%s] could not get SQL for 'privilege_changes'. cannot detect privilege changes", md.Name, specialMetricChangeEvents)
   297  		return changeCounts
   298  	}
   299  
   300  	// returns rows of: object_type, tag_role, tag_object, privilege_type
   301  	data, err := QueryMeasurements(ctx, md.Name, mvp.GetSQL(int(md.Version)))
   302  	if err != nil {
   303  		log.GetLogger(ctx).Errorf("[%s][%s] failed to fetch object privileges info: %v", md.Name, specialMetricChangeEvents, err)
   304  		return changeCounts
   305  	}
   306  
   307  	currentState := make(map[string]bool)
   308  	for _, dr := range data {
   309  		objIdent := fmt.Sprintf("%s#:#%s#:#%s#:#%s", dr["object_type"], dr["tag_role"], dr["tag_object"], dr["privilege_type"])
   310  		if firstRun {
   311  			hostState["object_privileges"][objIdent] = ""
   312  		} else {
   313  			_, ok := hostState["object_privileges"][objIdent]
   314  			if !ok {
   315  				log.GetLogger(ctx).Infof("[%s][%s] detected new object privileges: role=%s, object_type=%s, object=%s, privilege_type=%s",
   316  					md.Name, specialMetricChangeEvents, dr["tag_role"], dr["object_type"], dr["tag_object"], dr["privilege_type"])
   317  				dr["event"] = "GRANT"
   318  				detectedChanges = append(detectedChanges, dr)
   319  				changeCounts.Created++
   320  				hostState["object_privileges"][objIdent] = ""
   321  			}
   322  			currentState[objIdent] = true
   323  		}
   324  	}
   325  	// check revokes - exists in old state only
   326  	if !firstRun && len(currentState) > 0 {
   327  		for objPrevRun := range hostState["object_privileges"] {
   328  			if _, ok := currentState[objPrevRun]; !ok {
   329  				splits := strings.Split(objPrevRun, "#:#")
   330  				log.GetLogger(ctx).Infof("[%s][%s] detected removed object privileges: role=%s, object_type=%s, object=%s, privilege_type=%s",
   331  					md.Name, specialMetricChangeEvents, splits[1], splits[0], splits[2], splits[3])
   332  				revokeEntry := metrics.NewMeasurement(data.GetEpoch())
   333  				revokeEntry["object_type"] = splits[0]
   334  				revokeEntry["tag_role"] = splits[1]
   335  				revokeEntry["tag_object"] = splits[2]
   336  				revokeEntry["privilege_type"] = splits[3]
   337  				revokeEntry["event"] = "REVOKE"
   338  				detectedChanges = append(detectedChanges, revokeEntry)
   339  				changeCounts.Dropped++
   340  				delete(hostState["object_privileges"], objPrevRun)
   341  			}
   342  		}
   343  	}
   344  
   345  	log.GetLogger(ctx).Debugf("[%s][%s] detected %d object privilege changes...", md.Name, specialMetricChangeEvents, len(detectedChanges))
   346  	if len(detectedChanges) > 0 {
   347  		storageCh <- []metrics.MeasurementEnvelope{
   348  			{
   349  				DBName:     md.Name,
   350  				MetricName: "privilege_changes",
   351  				Data:       detectedChanges,
   352  				CustomTags: md.CustomTags,
   353  			}}
   354  	}
   355  
   356  	return changeCounts
   357  }

   422  func (r *Reaper) CheckForPGObjectChangesAndStore(ctx context.Context, dbUnique string, md *sources.SourceConn, hostState map[string]map[string]string) {
   423  	storageCh := r.measurementCh
   424  	sprocCounts := DetectSprocChanges(ctx, md, storageCh, hostState) // TODO some of Detect*() code could be unified...
   425  	tableCounts := DetectTableChanges(ctx, md, storageCh, hostState)
   426  	indexCounts := DetectIndexChanges(ctx, md, storageCh, hostState)
   427  	confCounts := DetectConfigurationChanges(ctx, md, storageCh, hostState)
   428  	privChangeCounts := DetectPrivilegeChanges(ctx, md, storageCh, hostState)
   429  
   430  	// need to send info on all object changes as one message as Grafana applies "last wins" for annotations with similar timestamp
   431  	message := ""
   432  	if sprocCounts.Altered > 0 || sprocCounts.Created > 0 || sprocCounts.Dropped > 0 {
   433  		message += fmt.Sprintf(" sprocs %d/%d/%d", sprocCounts.Created, sprocCounts.Altered, sprocCounts.Dropped)
   434  	}
   435  	if tableCounts.Altered > 0 || tableCounts.Created > 0 || tableCounts.Dropped > 0 {
   436  		message += fmt.Sprintf(" tables/views %d/%d/%d", tableCounts.Created, tableCounts.Altered, tableCounts.Dropped)
   437  	}
   438  	if indexCounts.Altered > 0 || indexCounts.Created > 0 || indexCounts.Dropped > 0 {
   439  		message += fmt.Sprintf(" indexes %d/%d/%d", indexCounts.Created, indexCounts.Altered, indexCounts.Dropped)
   440  	}
   441  	if confCounts.Altered > 0 || confCounts.Created > 0 {
   442  		message += fmt.Sprintf(" configuration %d/%d/%d", confCounts.Created, confCounts.Altered, confCounts.Dropped)
   443  	}
   444  	if privChangeCounts.Dropped > 0 || privChangeCounts.Created > 0 {
   445  		message += fmt.Sprintf(" privileges %d/%d/%d", privChangeCounts.Created, privChangeCounts.Altered, privChangeCounts.Dropped)
   446  	}
   447  
   448  	if message > "" {
   449  		message = "Detected changes for \"" + dbUnique + "\" [Created/Altered/Dropped]:" + message
   450  		log.GetLogger(ctx).Info(message)
   451  		detectedChangesSummary := make(metrics.Measurements, 0)
   452  		influxEntry := metrics.NewMeasurement(time.Now().UnixNano())
   453  		influxEntry["details"] = message

   466  // Called once on daemon startup if some commonly wanted extension (most notably pg_stat_statements) is missing.
   467  // With newer Postgres version can even succeed if the user is not a real superuser due to some cloud-specific
   468  // whitelisting or "trusted extensions" (a feature from v13). Ignores errors.
   469  func TryCreateMissingExtensions(ctx context.Context, dbUnique string, extensionNames []string, existingExtensions map[string]int) []string {
   470  	// TODO: move to sources package and use direct pgx connection
   471  	sqlAvailable := `select name::text from pg_available_extensions`
   472  	extsCreated := make([]string, 0)
   473  
   474  	// For security reasons don't allow to execute random strings but check that it's an existing extension
   475  	data, err := QueryMeasurements(ctx, dbUnique, sqlAvailable)
   476  	if err != nil {
   477  		log.GetLogger(ctx).Infof("[%s] Failed to get a list of available extensions: %v", dbUnique, err)
   478  		return extsCreated
   479  	}
   480  
   481  	availableExts := make(map[string]bool)
   482  	for _, row := range data {
   483  		availableExts[row["name"].(string)] = true
   484  	}
   485  
   486  	for _, extToCreate := range extensionNames {
   487  		if _, ok := existingExtensions[extToCreate]; ok {
   488  			continue
   489  		}
   490  		_, ok := availableExts[extToCreate]
   491  		if !ok {
   492  			log.GetLogger(ctx).Errorf("[%s] Requested extension %s not available on instance, cannot try to create...", dbUnique, extToCreate)
   493  		} else {
   494  			sqlCreateExt := `create extension ` + extToCreate
   495  			_, err := QueryMeasurements(ctx, dbUnique, sqlCreateExt)
   496  			if err != nil {
   497  				log.GetLogger(ctx).Errorf("[%s] Failed to create extension %s (based on --try-create-listed-exts-if-missing input): %v", dbUnique, extToCreate, err)
   498  			}
   499  			extsCreated = append(extsCreated, extToCreate)
   500  		}
   501  	}
   502  
   503  	return extsCreated
   504  }
