    23  func QueryMeasurements(ctx context.Context, dbUnique string, sql string, args ...any) (metrics.Measurements, error) {
    24  	var conn db.PgxIface
    25  	var md *sources.SourceConn
    26  	var err error
    27  	var tx pgx.Tx
    28  	if strings.TrimSpace(sql) == "" {
    29  		return nil, errors.New("empty SQL")
    30  	}
    31  	if md, err = GetMonitoredDatabaseByUniqueName(dbUnique); err != nil {
    32  		return nil, err
    33  	}
    34  	conn = md.Conn
    35  	if md.IsPostgresSource() {
    36  		// we don't want transaction for non-postgres sources, e.g. pgbouncer
    37  		if tx, err = conn.Begin(ctx); err != nil {
    38  			return nil, err
    39  		}
    40  		defer func() { _ = tx.Commit(ctx) }()
    41  		_, err = tx.Exec(ctx, "SET LOCAL lock_timeout TO '100ms'")
    42  		if err != nil {
    43  			return nil, err
    44  		}
    45  		conn = tx
    46  	}
    47  	rows, err := conn.Query(ctx, sql, args...)
    48  	if err == nil {
    49  		return pgx.CollectRows(rows, pgx.RowToMap)
    50  	}
    51  	return nil, err
    52  }

   104  // VersionToInt parses a given version and returns an integer  or
   105  // an error if unable to parse the version. Only parses valid semantic versions.
   106  // Performs checking that can find errors within the version.
   107  // Examples: v1.2 -> 01_02_00, v9.6.3 -> 09_06_03, v11 -> 11_00_00
   108  var regVer = regexp.MustCompile(`(\d+).?(\d*).?(\d*)`)

   110  func VersionToInt(version string) (v int) {
   111  	if matches := regVer.FindStringSubmatch(version); len(matches) > 1 {
   112  		for i, match := range matches[1:] {
   113  			v += func() (m int) { m, _ = strconv.Atoi(match); return }() * int(math.Pow10(4-i*2))
   114  		}
   115  	}
   116  	return
   117  }

   121  func GetMonitoredDatabaseSettings(ctx context.Context, md *sources.SourceConn, noCache bool) (MonitoredDatabaseSettings, error) {
   122  	var dbSettings MonitoredDatabaseSettings
   123  	var dbNewSettings MonitoredDatabaseSettings
   124  	var ok bool
   125  
   126  	l := log.GetLogger(ctx).WithField("source", md.Name).WithField("kind", md.Kind)
   127  
   128  	sqlExtensions := `select /* pgwatch_generated */ extname::text, (regexp_matches(extversion, $$\d+\.?\d+?$$))[1]::text as extversion from pg_extension order by 1;`
   129  
   130  	MonitoredDatabasesSettingsLock.Lock()
   131  	getVerLock, ok := MonitoredDatabasesSettingsGetLock[md.Name]
   132  	if !ok {
   133  		MonitoredDatabasesSettingsGetLock[md.Name] = &sync.RWMutex{}
   134  		getVerLock = MonitoredDatabasesSettingsGetLock[md.Name]
   135  	}
   136  	dbSettings, ok = MonitoredDatabasesSettings[md.Name]
   137  	MonitoredDatabasesSettingsLock.Unlock()
   138  
   139  	if !noCache && ok && dbSettings.LastCheckedOn.After(time.Now().Add(time.Minute*-2)) { // use cached version for 2 min
   140  		return dbSettings, nil
   141  	}
   142  	getVerLock.Lock() // limit to 1 concurrent version info fetch per DB
   143  	defer getVerLock.Unlock()
   144  	l.Debug("determining DB version and recovery status...")
   145  
   146  	if dbNewSettings.Extensions == nil {
   147  		dbNewSettings.Extensions = make(map[string]int)
   148  	}
   149  
   150  	switch md.Kind {
   151  	case sources.SourcePgBouncer:
   152  		if err := md.Conn.QueryRow(ctx, "SHOW VERSION").Scan(&dbNewSettings.VersionStr); err != nil {
   153  			return dbNewSettings, err
   154  		}
   155  		matches := rBouncerAndPgpoolVerMatch.FindStringSubmatch(dbNewSettings.VersionStr)
   156  		if len(matches) != 1 {
   157  			return dbSettings, fmt.Errorf("unexpected PgBouncer version input: %s", dbNewSettings.VersionStr)
   158  		}
   159  		dbNewSettings.Version = VersionToInt(matches[0])
   160  	case sources.SourcePgPool:
   161  		if err := md.Conn.QueryRow(ctx, "SHOW POOL_VERSION").Scan(&dbNewSettings.VersionStr); err != nil {
   162  			return dbNewSettings, err
   163  		}
   164  
   165  		matches := rBouncerAndPgpoolVerMatch.FindStringSubmatch(dbNewSettings.VersionStr)
   166  		if len(matches) != 1 {
   167  			return dbSettings, fmt.Errorf("unexpected PgPool version input: %s", dbNewSettings.VersionStr)
   168  		}
   169  		dbNewSettings.Version = VersionToInt(matches[0])
   170  	default:
   171  		sql := `select /* pgwatch_generated */ 
   172  	div(current_setting('server_version_num')::int, 10000) as ver, 
   173  	version(), 
   174  	pg_is_in_recovery(), 
   175  	current_database()::TEXT,
   176  	system_identifier
   177  FROM
   178  	pg_control_system()`
   179  
   180  		err := md.Conn.QueryRow(ctx, sql).
   181  			Scan(&dbNewSettings.Version, &dbNewSettings.VersionStr,
   182  				&dbNewSettings.IsInRecovery, &dbNewSettings.RealDbname,
   183  				&dbNewSettings.SystemIdentifier)
   184  		if err != nil {
   185  			if noCache {
   186  				return dbSettings, err
   187  			}
   188  			l.Error("DBGetPGVersion failed, using old cached value: ", err)
   189  			return dbSettings, nil
   190  		}
   191  
   192  		if dbSettings.ExecEnv != "" {
   193  			dbNewSettings.ExecEnv = dbSettings.ExecEnv // carry over as not likely to change ever
   194  		} else {
   195  			l.Debugf("determining the execution env...")
   196  			dbNewSettings.ExecEnv = md.DiscoverPlatform(ctx)
   197  		}
   198  
   199  		// to work around poor Azure Single Server FS functions performance for some metrics + the --min-db-size-mb filter
   200  		if dbNewSettings.ExecEnv == execEnvAzureSingle {
   201  			if approxSize, err := md.GetApproxSize(ctx); err == nil {
   202  				dbNewSettings.ApproxDBSizeB = approxSize
   203  			} else {
   204  				dbNewSettings.ApproxDBSizeB = dbSettings.ApproxDBSizeB
   205  			}
   206  		}
   207  
   208  		l.Debugf("[%s] determining if monitoring user is a superuser...", md.Name)
   209  		sqlSu := `select /* pgwatch_generated */ rolsuper from pg_roles r where rolname = session_user`
   210  
   211  		if err = md.Conn.QueryRow(ctx, sqlSu).Scan(&dbNewSettings.IsSuperuser); err != nil {
   212  			l.Errorf("[%s] failed to determine if monitoring user is a superuser: %v", md.Name, err)
   213  		}
   214  
   215  		l.Debugf("[%s] determining installed extensions info...", md.Name)
   216  		data, err := QueryMeasurements(ctx, md.Name, sqlExtensions)
   217  		if err != nil {
   218  			l.Errorf("[%s] failed to determine installed extensions info: %v", md.Name, err)
   219  		} else {
   220  			for _, dr := range data {
   221  				extver := VersionToInt(dr["extversion"].(string))
   222  				if extver == 0 {
   223  					l.Error("failed to determine extension version info for extension: ", dr["extname"])
   224  					continue
   225  				}
   226  				dbNewSettings.Extensions[dr["extname"].(string)] = extver
   227  			}
   228  			l.Debugf("[%s] installed extensions: %+v", md.Name, dbNewSettings.Extensions)
   229  		}
   230  
   231  	}
   232  
   233  	dbNewSettings.LastCheckedOn = time.Now()
   234  	MonitoredDatabasesSettingsLock.Lock()
   235  	MonitoredDatabasesSettings[md.Name] = dbNewSettings
   236  	MonitoredDatabasesSettingsLock.Unlock()
   237  
   238  	return dbNewSettings, nil
   239  }

   301  				influxEntry["tag_oid"] = splits[1]

   304  				} else {
   305  					influxEntry["epoch_ns"] = time.Now().UnixNano()
   306  				}

   310  			}
   311  		}
   312  		for _, deletedSProc := range deletedSProcs {
   313  			delete(hostState["sproc_hashes"], deletedSProc)
   314  		}
   315  	}

   319  		storageCh <- []metrics.MeasurementEnvelope{{DBName: dbUnique, MetricName: "sproc_changes", Data: detectedChanges, CustomTags: md.CustomTags}}
   320  	}
   321  
   322  	return changeCounts

   491  func DetectPrivilegeChanges(ctx context.Context, dbUnique string, vme MonitoredDatabaseSettings, storageCh chan<- []metrics.MeasurementEnvelope, hostState map[string]map[string]string) ChangeDetectionResults {
   492  	detectedChanges := make(metrics.Measurements, 0)
   493  	var firstRun bool
   494  	var changeCounts ChangeDetectionResults
   495  
   496  	log.GetLogger(ctx).Debugf("[%s][%s] checking object privilege changes...", dbUnique, specialMetricChangeEvents)
   497  	if _, ok := hostState["object_privileges"]; !ok {
   498  		firstRun = true
   499  		hostState["object_privileges"] = make(map[string]string)
   500  	}
   501  
   502  	mvp, err := GetMetricVersionProperties("privilege_changes", vme, nil)
   503  	if err != nil || mvp.GetSQL(int(vme.Version)) == "" {
   504  		log.GetLogger(ctx).Warningf("[%s][%s] could not get SQL for 'privilege_changes'. cannot detect privilege changes", dbUnique, specialMetricChangeEvents)
   505  		return changeCounts
   506  	}
   507  
   508  	// returns rows of: object_type, tag_role, tag_object, privilege_type
   509  	data, err := QueryMeasurements(ctx, dbUnique, mvp.GetSQL(int(vme.Version)))
   510  	if err != nil {
   511  		log.GetLogger(ctx).Errorf("[%s][%s] failed to fetch object privileges info: %v", dbUnique, specialMetricChangeEvents, err)
   512  		return changeCounts
   513  	}
   514  
   515  	currentState := make(map[string]bool)
   516  	for _, dr := range data {

   526  				detectedChanges = append(detectedChanges, dr)
   527  				changeCounts.Created++
   528  				hostState["object_privileges"][objIdent] = ""
   529  			}
   530  			currentState[objIdent] = true
   531  		}
   532  	}
   533  	// check revokes - exists in old state only
   534  	if !firstRun && len(currentState) > 0 {
   535  		for objPrevRun := range hostState["object_privileges"] {
   536  			if _, ok := currentState[objPrevRun]; !ok {
   537  				splits := strings.Split(objPrevRun, "#:#")
   538  				log.GetLogger(ctx).Infof("[%s][%s] detected removed object privileges: role=%s, object_type=%s, object=%s, privilege_type=%s",
   539  					dbUnique, specialMetricChangeEvents, splits[1], splits[0], splits[2], splits[3])
   540  				revokeEntry := make(metrics.Measurement)

   543  				} else {

   545  				}
   546  				revokeEntry["object_type"] = splits[0]
   547  				revokeEntry["tag_role"] = splits[1]

   549  				revokeEntry["privilege_type"] = splits[3]
   550  				revokeEntry["event"] = "REVOKE"
   551  				detectedChanges = append(detectedChanges, revokeEntry)

   562  			{
   563  				DBName:     dbUnique,
   564  				MetricName: "privilege_changes",
   565  				Data:       detectedChanges,
   566  				CustomTags: md.CustomTags,
   567  			}}
   568  	}
   569

   620  		}
   621  	}
   622  
   623  	log.GetLogger(ctx).Debugf("[%s][%s] detected %d configuration changes", dbUnique, specialMetricChangeEvents, len(detectedChanges))
   624  	if len(detectedChanges) > 0 {
   625  		md, _ := GetMonitoredDatabaseByUniqueName(dbUnique)
   626  		storageCh <- []metrics.MeasurementEnvelope{{
   627  			DBName:     dbUnique,
   628  			MetricName: "configuration_changes",
   629  			Data:       detectedChanges,
   630  			CustomTags: md.CustomTags,
   631  		}}
   632  	}

   683  func FetchMetricsPgpool(ctx context.Context, msg MetricFetchConfig, vme MonitoredDatabaseSettings, mvp metrics.Metric) (metrics.Measurements, error) {
   684  	var retData = make(metrics.Measurements, 0)
   685  	epochNs := time.Now().UnixNano()
   686  
   687  	sqlLines := strings.Split(strings.ToUpper(mvp.GetSQL(int(vme.Version))), "\n")
   688  
   689  	for _, sql := range sqlLines {
   690  		if strings.HasPrefix(sql, "SHOW POOL_NODES") {
   691  			data, err := QueryMeasurements(ctx, msg.DBUniqueName, sql)
   692  			if err != nil {
   693  				log.GetLogger(ctx).Errorf("[%s][%s] Could not fetch PgPool statistics: %v", msg.DBUniqueName, msg.MetricName, err)
   694  				return data, err
   695  			}
   696  
   697  			for _, row := range data {
   698  				retRow := make(metrics.Measurement)
   699  				retRow[epochColumnName] = epochNs
   700  				for k, v := range row {
   701  					vs := string(v.([]byte))
   702  					// need 1 tag so that Influx would not merge rows
   703  					if k == "node_id" {
   704  						retRow["tag_node_id"] = vs
   705  						continue
   706  					}
   707  
   708  					retRow[k] = vs
   709  					if k == "status" { // was changed from numeric to string at some pgpool version so leave the string
   710  						// but also add "status_num" field
   711  						switch vs {
   712  						case "up":
   713  							retRow["status_num"] = 1
   714  						case "down":
   715  							retRow["status_num"] = 0
   716  						default:
   717  							i, err := strconv.ParseInt(vs, 10, 64)
   718  							if err == nil {
   719  								retRow["status_num"] = i
   720  							}
   721  						}
   722  						continue
   723  					}
   724  					// everything is returned as text, so try to convert all numerics into ints / floats
   725  					if k != "lb_weight" {
   726  						i, err := strconv.ParseInt(vs, 10, 64)
   727  						if err == nil {
   728  							retRow[k] = i
   729  							continue
   730  						}
   731  					}
   732  					f, err := strconv.ParseFloat(vs, 64)
   733  					if err == nil {
   734  						retRow[k] = f
   735  						continue
   736  					}
   737  				}
   738  				retData = append(retData, retRow)
   739  			}
   740  		} else if strings.HasPrefix(sql, "SHOW POOL_PROCESSES") {
   741  			if len(retData) == 0 {
   742  				log.GetLogger(ctx).Warningf("[%s][%s] SHOW POOL_NODES needs to be placed before SHOW POOL_PROCESSES. ignoring SHOW POOL_PROCESSES", msg.DBUniqueName, msg.MetricName)
   743  				continue
   744  			}
   745  
   746  			data, err := QueryMeasurements(ctx, msg.DBUniqueName, sql)
   747  			if err != nil {
   748  				log.GetLogger(ctx).Errorf("[%s][%s] Could not fetch PgPool statistics: %v", msg.DBUniqueName, msg.MetricName, err)
   749  				continue
   750  			}
   751  
   752  			// summarize processesTotal / processes_active over all rows
   753  			processesTotal := 0
   754  			processesActive := 0
   755  			for _, row := range data {
   756  				processesTotal++
   757  				v, ok := row["database"]
   758  				if !ok {
   759  					log.GetLogger(ctx).Infof("[%s][%s] column 'database' not found from data returned by SHOW POOL_PROCESSES, check pool version / SQL definition", msg.DBUniqueName, msg.MetricName)
   760  					continue
   761  				}
   762  				if len(v.([]byte)) > 0 {
   763  					processesActive++
   764  				}
   765  			}
   766  
   767  			for _, retRow := range retData {
   768  				retRow["processes_total"] = processesTotal
   769  				retRow["processes_active"] = processesActive
   770  			}
   771  		}
   772  	}
   773  	return retData, nil
   774  }

   776  // Called once on daemon startup if some commonly wanted extension (most notably pg_stat_statements) is missing.
   777  // With newer Postgres version can even succeed if the user is not a real superuser due to some cloud-specific
   778  // whitelisting or "trusted extensions" (a feature from v13). Ignores errors.
   779  func TryCreateMissingExtensions(ctx context.Context, dbUnique string, extensionNames []string, existingExtensions map[string]int) []string {
   780  	sqlAvailable := `select name::text from pg_available_extensions`
   781  	extsCreated := make([]string, 0)
   782  
   783  	// For security reasons don't allow to execute random strings but check that it's an existing extension
   784  	data, err := QueryMeasurements(ctx, dbUnique, sqlAvailable)
   785  	if err != nil {
   786  		log.GetLogger(ctx).Infof("[%s] Failed to get a list of available extensions: %v", dbUnique, err)
   787  		return extsCreated
   788  	}
   789  
   790  	availableExts := make(map[string]bool)
   791  	for _, row := range data {
   792  		availableExts[row["name"].(string)] = true
   793  	}
   794  
   795  	for _, extToCreate := range extensionNames {
   796  		if _, ok := existingExtensions[extToCreate]; ok {
   797  			continue
   798  		}
   799  		_, ok := availableExts[extToCreate]
   800  		if !ok {
   801  			log.GetLogger(ctx).Errorf("[%s] Requested extension %s not available on instance, cannot try to create...", dbUnique, extToCreate)
   802  		} else {
   803  			sqlCreateExt := `create extension ` + extToCreate
   804  			_, err := QueryMeasurements(ctx, dbUnique, sqlCreateExt)
   805  			if err != nil {
   806  				log.GetLogger(ctx).Errorf("[%s] Failed to create extension %s (based on --try-create-listed-exts-if-missing input): %v", dbUnique, extToCreate, err)
   807  			}
   808  			extsCreated = append(extsCreated, extToCreate)
   809  		}
   810  	}
   811  
   812  	return extsCreated
