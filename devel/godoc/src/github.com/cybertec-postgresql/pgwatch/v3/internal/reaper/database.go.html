    18  func QueryMeasurements(ctx context.Context, md *sources.SourceConn, sql string, args ...any) (metrics.Measurements, error) {
    19  	var conn db.PgxIface
    20  	var err error
    21  	var tx pgx.Tx
    22  	if strings.TrimSpace(sql) == "" {
    23  		return nil, errors.New("empty SQL")
    24  	}
    25  
    26  	conn = md.Conn
    27  	if md.IsPostgresSource() {
    28  		// we don't want transaction for non-postgres sources, e.g. pgbouncer
    29  		if tx, err = conn.Begin(ctx); err != nil {
    30  			return nil, err
    31  		}
    32  		defer func() { _ = tx.Commit(ctx) }()
    33  		_, err = tx.Exec(ctx, "SET LOCAL lock_timeout TO '100ms'")
    34  		if err != nil {
    35  			return nil, err
    36  		}
    37  		conn = tx
    38  	} else {
    39  		// we want simple protocol for non-postgres connections, e.g. pgpool
    40  		args = append([]any{pgx.QueryExecModeSimpleProtocol}, args...)
    41  	}
    42  	rows, err := conn.Query(ctx, sql, args...)
    43  	if err == nil {
    44  		return pgx.CollectRows(rows, metrics.RowToMeasurement)
    45  	}
    46  	return nil, err
    47  }

    56  		firstRun = true
    57  		hostState["sproc_hashes"] = make(map[string]string)
    58  	}
    59  
    60  	mvp, ok := metricDefs.GetMetricDef("sproc_hashes")
    61  	if !ok {
    62  		log.GetLogger(ctx).Error("could not get sproc_hashes sql")
    63  		return changeCounts
    64  	}
    65  
    66  	data, err := QueryMeasurements(ctx, md, mvp.GetSQL(int(md.Version)))
    67  	if err != nil {
    68  		log.GetLogger(ctx).Error("could not read sproc_hashes from monitored host: ", md.Name, ", err:", err)
    69  		return changeCounts
    70  	}
    71  
    72  	for _, dr := range data {
    73  		objIdent := dr["tag_sproc"].(string) + dbMetricJoinStr + dr["tag_oid"].(string)
    74  		prevHash, ok := hostState["sproc_hashes"][objIdent]
    75  		if ok { // we have existing state
    76  			if prevHash != dr["md5"].(string) {
    77  				log.GetLogger(ctx).Info("detected change in sproc:", dr["tag_sproc"], ", oid:", dr["tag_oid"])
    78  				dr["event"] = "alter"
    79  				detectedChanges = append(detectedChanges, dr)
    80  				hostState["sproc_hashes"][objIdent] = dr["md5"].(string)
    81  				changeCounts.Altered++
    82  			}
    83  		} else { // check for new / delete
    84  			if !firstRun {
    85  				log.GetLogger(ctx).Info("detected new sproc:", dr["tag_sproc"], ", oid:", dr["tag_oid"])
    86  				dr["event"] = "create"
    87  				detectedChanges = append(detectedChanges, dr)
    88  				changeCounts.Created++
    89  			}
    90  			hostState["sproc_hashes"][objIdent] = dr["md5"].(string)
    91  		}
    92  	}
    93  	// detect deletes
    94  	if !firstRun && len(hostState["sproc_hashes"]) != len(data) {
    95  		deletedSProcs := make([]string, 0)
    96  		// turn resultset to map => [oid]=true for faster checks
    97  		currentOidMap := make(map[string]bool)
    98  		for _, dr := range data {
    99  			currentOidMap[dr["tag_sproc"].(string)+dbMetricJoinStr+dr["tag_oid"].(string)] = true
   100  		}
   101  		for sprocIdent := range hostState["sproc_hashes"] {
   102  			_, ok := currentOidMap[sprocIdent]
   103  			if !ok {
   104  				splits := strings.Split(sprocIdent, dbMetricJoinStr)
   105  				log.GetLogger(ctx).Info("detected delete of sproc:", splits[0], ", oid:", splits[1])
   106  				influxEntry := metrics.NewMeasurement(data.GetEpoch())
   107  				influxEntry["event"] = "drop"
   108  				influxEntry["tag_sproc"] = splits[0]
   109  				influxEntry["tag_oid"] = splits[1]
   110  				detectedChanges = append(detectedChanges, influxEntry)
   111  				deletedSProcs = append(deletedSProcs, sprocIdent)
   112  				changeCounts.Dropped++
   113  			}
   114  		}
   115  		for _, deletedSProc := range deletedSProcs {
   116  			delete(hostState["sproc_hashes"], deletedSProc)
   117  		}
   118  	}
   119  	log.GetLogger(ctx).Debugf("[%s][%s] detected %d sproc changes", md.Name, specialMetricChangeEvents, len(detectedChanges))
   120  	if len(detectedChanges) > 0 {
   121  		storageCh <- metrics.MeasurementEnvelope{
   122  			DBName:     md.Name,
   123  			MetricName: "sproc_changes",
   124  			Data:       detectedChanges,
   125  			CustomTags: md.CustomTags,
   126  		}
   127  	}
   128  
   129  	return changeCounts
   130  }

   296  func DetectPrivilegeChanges(ctx context.Context, md *sources.SourceConn, storageCh chan<- metrics.MeasurementEnvelope, hostState map[string]map[string]string) ChangeDetectionResults {
   297  	detectedChanges := make(metrics.Measurements, 0)
   298  	var firstRun bool
   299  	var changeCounts ChangeDetectionResults
   300  
   301  	log.GetLogger(ctx).Debugf("[%s][%s] checking object privilege changes...", md.Name, specialMetricChangeEvents)
   302  	if _, ok := hostState["object_privileges"]; !ok {
   303  		firstRun = true
   304  		hostState["object_privileges"] = make(map[string]string)
   305  	}
   306  
   307  	mvp, ok := metricDefs.GetMetricDef("privilege_changes")
   308  	if !ok || mvp.GetSQL(int(md.Version)) == "" {
   309  		log.GetLogger(ctx).Warningf("[%s][%s] could not get SQL for 'privilege_changes'. cannot detect privilege changes", md.Name, specialMetricChangeEvents)
   310  		return changeCounts
   311  	}
   312  
   313  	// returns rows of: object_type, tag_role, tag_object, privilege_type
   314  	data, err := QueryMeasurements(ctx, md, mvp.GetSQL(int(md.Version)))
   315  	if err != nil {
   316  		log.GetLogger(ctx).Errorf("[%s][%s] failed to fetch object privileges info: %v", md.Name, specialMetricChangeEvents, err)
   317  		return changeCounts
   318  	}
   319  
   320  	currentState := make(map[string]bool)
   321  	for _, dr := range data {
   322  		objIdent := fmt.Sprintf("%s#:#%s#:#%s#:#%s", dr["object_type"], dr["tag_role"], dr["tag_object"], dr["privilege_type"])
   323  		if firstRun {
   324  			hostState["object_privileges"][objIdent] = ""
   325  		} else {
   326  			_, ok := hostState["object_privileges"][objIdent]
   327  			if !ok {
   328  				log.GetLogger(ctx).Infof("[%s][%s] detected new object privileges: role=%s, object_type=%s, object=%s, privilege_type=%s",
   329  					md.Name, specialMetricChangeEvents, dr["tag_role"], dr["object_type"], dr["tag_object"], dr["privilege_type"])
   330  				dr["event"] = "GRANT"
   331  				detectedChanges = append(detectedChanges, dr)
   332  				changeCounts.Created++
   333  				hostState["object_privileges"][objIdent] = ""
   334  			}
   335  			currentState[objIdent] = true
   336  		}
   337  	}
   338  	// check revokes - exists in old state only
   339  	if !firstRun && len(currentState) > 0 {
   340  		for objPrevRun := range hostState["object_privileges"] {
   341  			if _, ok := currentState[objPrevRun]; !ok {
   342  				splits := strings.Split(objPrevRun, "#:#")
   343  				log.GetLogger(ctx).Infof("[%s][%s] detected removed object privileges: role=%s, object_type=%s, object=%s, privilege_type=%s",
   344  					md.Name, specialMetricChangeEvents, splits[1], splits[0], splits[2], splits[3])
   345  				revokeEntry := metrics.NewMeasurement(data.GetEpoch())
   346  				revokeEntry["object_type"] = splits[0]
   347  				revokeEntry["tag_role"] = splits[1]
   348  				revokeEntry["tag_object"] = splits[2]
   349  				revokeEntry["privilege_type"] = splits[3]
   350  				revokeEntry["event"] = "REVOKE"
   351  				detectedChanges = append(detectedChanges, revokeEntry)
   352  				changeCounts.Dropped++
   353  				delete(hostState["object_privileges"], objPrevRun)
   354  			}
   355  		}
   356  	}
   357  
   358  	log.GetLogger(ctx).Debugf("[%s][%s] detected %d object privilege changes...", md.Name, specialMetricChangeEvents, len(detectedChanges))
   359  	if len(detectedChanges) > 0 {
   360  		storageCh <- metrics.MeasurementEnvelope{
   361  			DBName:     md.Name,
   362  			MetricName: "privilege_changes",
   363  			Data:       detectedChanges,
   364  			CustomTags: md.CustomTags,
   365  		}
   366  	}
   367  
   368  	return changeCounts
   369  }

   434  func (r *Reaper) CheckForPGObjectChangesAndStore(ctx context.Context, dbUnique string, md *sources.SourceConn, hostState map[string]map[string]string) {
   435  	storageCh := r.measurementCh
   436  	sprocCounts := DetectSprocChanges(ctx, md, storageCh, hostState) // TODO some of Detect*() code could be unified...
   437  	tableCounts := DetectTableChanges(ctx, md, storageCh, hostState)
   438  	indexCounts := DetectIndexChanges(ctx, md, storageCh, hostState)
   439  	confCounts := DetectConfigurationChanges(ctx, md, storageCh, hostState)
   440  	privChangeCounts := DetectPrivilegeChanges(ctx, md, storageCh, hostState)
   441  
   442  	// need to send info on all object changes as one message as Grafana applies "last wins" for annotations with similar timestamp
   443  	message := ""
   444  	if sprocCounts.Altered > 0 || sprocCounts.Created > 0 || sprocCounts.Dropped > 0 {
   445  		message += fmt.Sprintf(" sprocs %d/%d/%d", sprocCounts.Created, sprocCounts.Altered, sprocCounts.Dropped)
   446  	}
   447  	if tableCounts.Altered > 0 || tableCounts.Created > 0 || tableCounts.Dropped > 0 {
   448  		message += fmt.Sprintf(" tables/views %d/%d/%d", tableCounts.Created, tableCounts.Altered, tableCounts.Dropped)
   449  	}
   450  	if indexCounts.Altered > 0 || indexCounts.Created > 0 || indexCounts.Dropped > 0 {
   451  		message += fmt.Sprintf(" indexes %d/%d/%d", indexCounts.Created, indexCounts.Altered, indexCounts.Dropped)
   452  	}
   453  	if confCounts.Altered > 0 || confCounts.Created > 0 {
   454  		message += fmt.Sprintf(" configuration %d/%d/%d", confCounts.Created, confCounts.Altered, confCounts.Dropped)
   455  	}
   456  	if privChangeCounts.Dropped > 0 || privChangeCounts.Created > 0 {
   457  		message += fmt.Sprintf(" privileges %d/%d/%d", privChangeCounts.Created, privChangeCounts.Altered, privChangeCounts.Dropped)
   458  	}
   459  
   460  	if message > "" {
   461  		message = "Detected changes for \"" + dbUnique + "\" [Created/Altered/Dropped]:" + message
   462  		log.GetLogger(ctx).Info(message)
   463  		detectedChangesSummary := make(metrics.Measurements, 0)
   464  		influxEntry := metrics.NewMeasurement(time.Now().UnixNano())

   478  // Called once on daemon startup if some commonly wanted extension (most notably pg_stat_statements) is missing.
   479  // With newer Postgres version can even succeed if the user is not a real superuser due to some cloud-specific
   480  // whitelisting or "trusted extensions" (a feature from v13). Ignores errors.
   481  func TryCreateMissingExtensions(ctx context.Context, md *sources.SourceConn, extensionNames []string, existingExtensions map[string]int) []string {
   482  	// TODO: move to sources package and use direct pgx connection
   483  	sqlAvailable := `select name::text from pg_available_extensions`
   484  	extsCreated := make([]string, 0)
   485  
   486  	// For security reasons don't allow to execute random strings but check that it's an existing extension
   487  	data, err := QueryMeasurements(ctx, md, sqlAvailable)
   488  	if err != nil {
   489  		log.GetLogger(ctx).Infof("[%s] Failed to get a list of available extensions: %v", md, err)
   490  		return extsCreated
   491  	}
   492  
   493  	availableExts := make(map[string]bool)
   494  	for _, row := range data {
   495  		availableExts[row["name"].(string)] = true
   496  	}
   497  
   498  	for _, extToCreate := range extensionNames {
   499  		if _, ok := existingExtensions[extToCreate]; ok {
   500  			continue
   501  		}
   502  		_, ok := availableExts[extToCreate]
   503  		if !ok {
   504  			log.GetLogger(ctx).Errorf("[%s] Requested extension %s not available on instance, cannot try to create...", md, extToCreate)
   505  		} else {
   506  			sqlCreateExt := `create extension ` + extToCreate
   507  			_, err := QueryMeasurements(ctx, md, sqlCreateExt)
   508  			if err != nil {
   509  				log.GetLogger(ctx).Errorf("[%s] Failed to create extension %s (based on --try-create-listed-exts-if-missing input): %v", md, extToCreate, err)
   510  			}
   511  			extsCreated = append(extsCreated, extToCreate)
   512  		}
   513  	}
   514  
   515  	return extsCreated
   516  }
