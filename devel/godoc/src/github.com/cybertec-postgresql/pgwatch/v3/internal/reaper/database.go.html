    18  func QueryMeasurements(ctx context.Context, dbUnique string, sql string, args ...any) (metrics.Measurements, error) {
    19  	var conn db.PgxIface
    20  	var md *sources.SourceConn
    21  	var err error
    22  	var tx pgx.Tx
    23  	if strings.TrimSpace(sql) == "" {
    24  		return nil, errors.New("empty SQL")
    25  	}
    26  	if md, err = GetMonitoredDatabaseByUniqueName(ctx, dbUnique); err != nil {
    27  		return nil, err
    28  	}
    29  	conn = md.Conn
    30  	if md.IsPostgresSource() {
    31  		// we don't want transaction for non-postgres sources, e.g. pgbouncer
    32  		if tx, err = conn.Begin(ctx); err != nil {
    33  			return nil, err
    34  		}
    35  		defer func() { _ = tx.Commit(ctx) }()
    36  		_, err = tx.Exec(ctx, "SET LOCAL lock_timeout TO '100ms'")
    37  		if err != nil {
    38  			return nil, err
    39  		}
    40  		conn = tx
    41  	}
    42  	rows, err := conn.Query(ctx, sql, args...)
    43  	if err == nil {
    44  		return pgx.CollectRows(rows, pgx.RowToMap)
    45  	}
    46  	return nil, err
    47  }

    52  	var changeCounts ChangeDetectionResults
    53  
    54  	log.GetLogger(ctx).Debugf("[%s][%s] checking for sproc changes...", md.Name, specialMetricChangeEvents)

    56  		firstRun = true
    57  		hostState["sproc_hashes"] = make(map[string]string)
    58  	}
    59  
    60  	mvp, ok := metricDefs.GetMetricDef("sproc_hashes")
    61  	if !ok {
    62  		log.GetLogger(ctx).Error("could not get sproc_hashes sql")
    63  		return changeCounts
    64  	}
    65  
    66  	data, err := QueryMeasurements(ctx, md.Name, mvp.GetSQL(int(md.Version)))
    67  	if err != nil {
    68  		log.GetLogger(ctx).Error("could not read sproc_hashes from monitored host: ", md.Name, ", err:", err)
    69  		return changeCounts
    70  	}
    71  
    72  	for _, dr := range data {
    73  		objIdent := dr["tag_sproc"].(string) + dbMetricJoinStr + dr["tag_oid"].(string)
    74  		prevHash, ok := hostState["sproc_hashes"][objIdent]
    75  		if ok { // we have existing state
    76  			if prevHash != dr["md5"].(string) {
    77  				log.GetLogger(ctx).Info("detected change in sproc:", dr["tag_sproc"], ", oid:", dr["tag_oid"])
    78  				dr["event"] = "alter"
    79  				detectedChanges = append(detectedChanges, dr)
    80  				hostState["sproc_hashes"][objIdent] = dr["md5"].(string)
    81  				changeCounts.Altered++
    82  			}
    83  		} else { // check for new / delete
    84  			if !firstRun {
    85  				log.GetLogger(ctx).Info("detected new sproc:", dr["tag_sproc"], ", oid:", dr["tag_oid"])
    86  				dr["event"] = "create"
    87  				detectedChanges = append(detectedChanges, dr)
    88  				changeCounts.Created++
    89  			}
    90  			hostState["sproc_hashes"][objIdent] = dr["md5"].(string)
    91  		}
    92  	}
    93  	// detect deletes
    94  	if !firstRun && len(hostState["sproc_hashes"]) != len(data) {
    95  		deletedSProcs := make([]string, 0)
    96  		// turn resultset to map => [oid]=true for faster checks
    97  		currentOidMap := make(map[string]bool)
    98  		for _, dr := range data {
    99  			currentOidMap[dr["tag_sproc"].(string)+dbMetricJoinStr+dr["tag_oid"].(string)] = true
   100  		}
   101  		for sprocIdent := range hostState["sproc_hashes"] {
   102  			_, ok := currentOidMap[sprocIdent]
   103  			if !ok {
   104  				splits := strings.Split(sprocIdent, dbMetricJoinStr)
   105  				log.GetLogger(ctx).Info("detected delete of sproc:", splits[0], ", oid:", splits[1])
   106  				influxEntry := metrics.NewMeasurement(data.GetEpoch())
   107  				influxEntry["event"] = "drop"
   108  				influxEntry["tag_sproc"] = splits[0]
   109  				influxEntry["tag_oid"] = splits[1]
   110  				detectedChanges = append(detectedChanges, influxEntry)
   111  				deletedSProcs = append(deletedSProcs, sprocIdent)
   112  				changeCounts.Dropped++
   113  			}
   114  		}
   115  		for _, deletedSProc := range deletedSProcs {
   116  			delete(hostState["sproc_hashes"], deletedSProc)
   117  		}
   118  	}
   119  	log.GetLogger(ctx).Debugf("[%s][%s] detected %d sproc changes", md.Name, specialMetricChangeEvents, len(detectedChanges))
   120  	if len(detectedChanges) > 0 {
   121  		storageCh <- []metrics.MeasurementEnvelope{{DBName: md.Name, MetricName: "sproc_changes", Data: detectedChanges, CustomTags: md.CustomTags}}
   122  	}
   123  
   124  	return changeCounts
   125  }

   281  func DetectPrivilegeChanges(ctx context.Context, md *sources.SourceConn, storageCh chan<- []metrics.MeasurementEnvelope, hostState map[string]map[string]string) ChangeDetectionResults {
   282  	detectedChanges := make(metrics.Measurements, 0)
   283  	var firstRun bool
   284  	var changeCounts ChangeDetectionResults
   285  
   286  	log.GetLogger(ctx).Debugf("[%s][%s] checking object privilege changes...", md.Name, specialMetricChangeEvents)
   287  	if _, ok := hostState["object_privileges"]; !ok {
   288  		firstRun = true
   289  		hostState["object_privileges"] = make(map[string]string)
   290  	}
   291  
   292  	mvp, ok := metricDefs.GetMetricDef("privilege_changes")
   293  	if !ok || mvp.GetSQL(int(md.Version)) == "" {
   294  		log.GetLogger(ctx).Warningf("[%s][%s] could not get SQL for 'privilege_changes'. cannot detect privilege changes", md.Name, specialMetricChangeEvents)
   295  		return changeCounts
   296  	}
   297  
   298  	// returns rows of: object_type, tag_role, tag_object, privilege_type
   299  	data, err := QueryMeasurements(ctx, md.Name, mvp.GetSQL(int(md.Version)))
   300  	if err != nil {
   301  		log.GetLogger(ctx).Errorf("[%s][%s] failed to fetch object privileges info: %v", md.Name, specialMetricChangeEvents, err)
   302  		return changeCounts
   303  	}
   304  
   305  	currentState := make(map[string]bool)
   306  	for _, dr := range data {
   307  		objIdent := fmt.Sprintf("%s#:#%s#:#%s#:#%s", dr["object_type"], dr["tag_role"], dr["tag_object"], dr["privilege_type"])
   308  		if firstRun {
   309  			hostState["object_privileges"][objIdent] = ""
   310  		} else {
   311  			_, ok := hostState["object_privileges"][objIdent]
   312  			if !ok {
   313  				log.GetLogger(ctx).Infof("[%s][%s] detected new object privileges: role=%s, object_type=%s, object=%s, privilege_type=%s",
   314  					md.Name, specialMetricChangeEvents, dr["tag_role"], dr["object_type"], dr["tag_object"], dr["privilege_type"])
   315  				dr["event"] = "GRANT"
   316  				detectedChanges = append(detectedChanges, dr)
   317  				changeCounts.Created++
   318  				hostState["object_privileges"][objIdent] = ""
   319  			}
   320  			currentState[objIdent] = true
   321  		}
   322  	}
   323  	// check revokes - exists in old state only
   324  	if !firstRun && len(currentState) > 0 {
   325  		for objPrevRun := range hostState["object_privileges"] {
   326  			if _, ok := currentState[objPrevRun]; !ok {
   327  				splits := strings.Split(objPrevRun, "#:#")
   328  				log.GetLogger(ctx).Infof("[%s][%s] detected removed object privileges: role=%s, object_type=%s, object=%s, privilege_type=%s",
   329  					md.Name, specialMetricChangeEvents, splits[1], splits[0], splits[2], splits[3])
   330  				revokeEntry := metrics.NewMeasurement(data.GetEpoch())
   331  				revokeEntry["object_type"] = splits[0]
   332  				revokeEntry["tag_role"] = splits[1]
   333  				revokeEntry["tag_object"] = splits[2]
   334  				revokeEntry["privilege_type"] = splits[3]
   335  				revokeEntry["event"] = "REVOKE"
   336  				detectedChanges = append(detectedChanges, revokeEntry)
   337  				changeCounts.Dropped++
   338  				delete(hostState["object_privileges"], objPrevRun)
   339  			}
   340  		}
   341  	}
   342  
   343  	log.GetLogger(ctx).Debugf("[%s][%s] detected %d object privilege changes...", md.Name, specialMetricChangeEvents, len(detectedChanges))
   344  	if len(detectedChanges) > 0 {
   345  		storageCh <- []metrics.MeasurementEnvelope{
   346  			{
   347  				DBName:     md.Name,
   348  				MetricName: "privilege_changes",
   349  				Data:       detectedChanges,
   350  				CustomTags: md.CustomTags,
   351  			}}
   352  	}
   353  
   354  	return changeCounts
   355  }

   420  func (r *Reaper) CheckForPGObjectChangesAndStore(ctx context.Context, dbUnique string, md *sources.SourceConn, hostState map[string]map[string]string) {
   421  	storageCh := r.measurementCh
   422  	sprocCounts := DetectSprocChanges(ctx, md, storageCh, hostState) // TODO some of Detect*() code could be unified...
   423  	tableCounts := DetectTableChanges(ctx, md, storageCh, hostState)
   424  	indexCounts := DetectIndexChanges(ctx, md, storageCh, hostState)
   425  	confCounts := DetectConfigurationChanges(ctx, md, storageCh, hostState)
   426  	privChangeCounts := DetectPrivilegeChanges(ctx, md, storageCh, hostState)
   427  
   428  	// need to send info on all object changes as one message as Grafana applies "last wins" for annotations with similar timestamp
   429  	message := ""
   430  	if sprocCounts.Altered > 0 || sprocCounts.Created > 0 || sprocCounts.Dropped > 0 {
   431  		message += fmt.Sprintf(" sprocs %d/%d/%d", sprocCounts.Created, sprocCounts.Altered, sprocCounts.Dropped)
   432  	}
   433  	if tableCounts.Altered > 0 || tableCounts.Created > 0 || tableCounts.Dropped > 0 {
   434  		message += fmt.Sprintf(" tables/views %d/%d/%d", tableCounts.Created, tableCounts.Altered, tableCounts.Dropped)
   435  	}
   436  	if indexCounts.Altered > 0 || indexCounts.Created > 0 || indexCounts.Dropped > 0 {
   437  		message += fmt.Sprintf(" indexes %d/%d/%d", indexCounts.Created, indexCounts.Altered, indexCounts.Dropped)
   438  	}
   439  	if confCounts.Altered > 0 || confCounts.Created > 0 {
   440  		message += fmt.Sprintf(" configuration %d/%d/%d", confCounts.Created, confCounts.Altered, confCounts.Dropped)
   441  	}
   442  	if privChangeCounts.Dropped > 0 || privChangeCounts.Created > 0 {
   443  		message += fmt.Sprintf(" privileges %d/%d/%d", privChangeCounts.Created, privChangeCounts.Altered, privChangeCounts.Dropped)
   444  	}
   445  
   446  	if message > "" {
   447  		message = "Detected changes for \"" + dbUnique + "\" [Created/Altered/Dropped]:" + message
   448  		log.GetLogger(ctx).Info(message)
   449  		detectedChangesSummary := make(metrics.Measurements, 0)
   450  		influxEntry := metrics.NewMeasurement(time.Now().UnixNano())
   451  		influxEntry["details"] = message
   452  		detectedChangesSummary = append(detectedChangesSummary, influxEntry)
   453  		md, _ := GetMonitoredDatabaseByUniqueName(ctx, dbUnique)
   454  		storageCh <- []metrics.MeasurementEnvelope{{DBName: dbUnique,
   455  			SourceType: string(md.Kind),

   464  // some extra work needed as pgpool SHOW commands don't specify the return data types for some reason
   465  func FetchMetricsPgpool(ctx context.Context, msg MetricFetchConfig, md *sources.SourceConn, mvp metrics.Metric) (metrics.Measurements, error) {
   466  	var retData = make(metrics.Measurements, 0)
   467  	epochNs := time.Now().UnixNano()
   468  
   469  	sqlLines := strings.Split(strings.ToUpper(mvp.GetSQL(int(md.Version))), "\n")
   470  
   471  	for _, sql := range sqlLines {
   472  		if strings.HasPrefix(sql, "SHOW POOL_NODES") {
   473  			data, err := QueryMeasurements(ctx, msg.DBUniqueName, sql)
   474  			if err != nil {
   475  				log.GetLogger(ctx).Errorf("[%s][%s] Could not fetch PgPool statistics: %v", msg.DBUniqueName, msg.MetricName, err)
   476  				return data, err
   477  			}
   478  
   479  			for _, row := range data {
   480  				retRow := metrics.NewMeasurement(epochNs)
   481  				for k, v := range row {
   482  					vs := string(v.([]byte))
   483  					// need 1 tag so that Influx would not merge rows
   484  					if k == "node_id" {
   485  						retRow["tag_node_id"] = vs
   486  						continue
   487  					}
   488  
   489  					retRow[k] = vs
   490  					if k == "status" { // was changed from numeric to string at some pgpool version so leave the string
   491  						// but also add "status_num" field
   492  						switch vs {
   493  						case "up":
   494  							retRow["status_num"] = 1
   495  						case "down":
   496  							retRow["status_num"] = 0
   497  						default:
   498  							i, err := strconv.ParseInt(vs, 10, 64)
   499  							if err == nil {
   500  								retRow["status_num"] = i
   501  							}
   502  						}
   503  						continue
   504  					}
   505  					// everything is returned as text, so try to convert all numerics into ints / floats
   506  					if k != "lb_weight" {
   507  						i, err := strconv.ParseInt(vs, 10, 64)
   508  						if err == nil {
   509  							retRow[k] = i
   510  							continue
   511  						}
   512  					}
   513  					f, err := strconv.ParseFloat(vs, 64)
   514  					if err == nil {
   515  						retRow[k] = f
   516  						continue
   517  					}
   518  				}
   519  				retData = append(retData, retRow)
   520  			}
   521  		} else if strings.HasPrefix(sql, "SHOW POOL_PROCESSES") {
   522  			if len(retData) == 0 {
   523  				log.GetLogger(ctx).Warningf("[%s][%s] SHOW POOL_NODES needs to be placed before SHOW POOL_PROCESSES. ignoring SHOW POOL_PROCESSES", msg.DBUniqueName, msg.MetricName)
   524  				continue
   525  			}
   526  
   527  			data, err := QueryMeasurements(ctx, msg.DBUniqueName, sql)
   528  			if err != nil {
   529  				log.GetLogger(ctx).Errorf("[%s][%s] Could not fetch PgPool statistics: %v", msg.DBUniqueName, msg.MetricName, err)
   530  				continue
   531  			}
   532  
   533  			// summarize processesTotal / processes_active over all rows
   534  			processesTotal := 0
   535  			processesActive := 0
   536  			for _, row := range data {
   537  				processesTotal++
   538  				v, ok := row["database"]
   539  				if !ok {
   540  					log.GetLogger(ctx).Infof("[%s][%s] column 'database' not found from data returned by SHOW POOL_PROCESSES, check pool version / SQL definition", msg.DBUniqueName, msg.MetricName)
   541  					continue
   542  				}
   543  				if len(v.([]byte)) > 0 {
   544  					processesActive++
   545  				}
   546  			}
   547  
   548  			for _, retRow := range retData {
   549  				retRow["processes_total"] = processesTotal
   550  				retRow["processes_active"] = processesActive
   551  			}
   552  		}
   553  	}
   554  	return retData, nil
   555  }

   557  // Called once on daemon startup if some commonly wanted extension (most notably pg_stat_statements) is missing.
   558  // With newer Postgres version can even succeed if the user is not a real superuser due to some cloud-specific
   559  // whitelisting or "trusted extensions" (a feature from v13). Ignores errors.
   560  func TryCreateMissingExtensions(ctx context.Context, dbUnique string, extensionNames []string, existingExtensions map[string]int) []string {
   561  	sqlAvailable := `select name::text from pg_available_extensions`
   562  	extsCreated := make([]string, 0)
   563  
   564  	// For security reasons don't allow to execute random strings but check that it's an existing extension
   565  	data, err := QueryMeasurements(ctx, dbUnique, sqlAvailable)
   566  	if err != nil {
   567  		log.GetLogger(ctx).Infof("[%s] Failed to get a list of available extensions: %v", dbUnique, err)
   568  		return extsCreated
   569  	}
   570  
   571  	availableExts := make(map[string]bool)
   572  	for _, row := range data {
   573  		availableExts[row["name"].(string)] = true
   574  	}
   575  
   576  	for _, extToCreate := range extensionNames {
   577  		if _, ok := existingExtensions[extToCreate]; ok {
   578  			continue
   579  		}
   580  		_, ok := availableExts[extToCreate]
   581  		if !ok {
   582  			log.GetLogger(ctx).Errorf("[%s] Requested extension %s not available on instance, cannot try to create...", dbUnique, extToCreate)
   583  		} else {
   584  			sqlCreateExt := `create extension ` + extToCreate
   585  			_, err := QueryMeasurements(ctx, dbUnique, sqlCreateExt)
   586  			if err != nil {
   587  				log.GetLogger(ctx).Errorf("[%s] Failed to create extension %s (based on --try-create-listed-exts-if-missing input): %v", dbUnique, extToCreate, err)
   588  			}
   589  			extsCreated = append(extsCreated, extToCreate)
   590  		}
   591  	}
   592  
   593  	return extsCreated
   594  }

   596  // Called once on daemon startup to try to create "metric fething helper" functions automatically
   597  func TryCreateMetricsFetchingHelpers(ctx context.Context, md *sources.SourceConn) (err error) {
   598  	metricConfig := func() map[string]float64 {
   599  		if len(md.Metrics) > 0 {
   600  			return md.Metrics
   601  		}
   602  		if md.PresetMetrics > "" {
   603  			return metricDefs.GetPresetMetrics(md.PresetMetrics)
   604  		}
   605  		return nil
   606  	}()
   607  	conf, err := pgx.ParseConfig(md.ConnStr)
   608  	if err != nil {
   609  		return err
   610  	}
   611  	conf.DefaultQueryExecMode = pgx.QueryExecModeExec
   612  	c, err := pgx.ConnectConfig(ctx, conf)
   613  	if err != nil {
   614  		return nil
   615  	}
   616  	defer c.Close(ctx)
   617  
   618  	for metricName := range metricConfig {
   619  		Metric, ok := metricDefs.GetMetricDef(metricName)
   620  		if !ok {
   621  			continue
   622  		}
   623  
   624  		_, err = c.Exec(ctx, Metric.InitSQL)
   625  		if err != nil {
   626  			log.GetLogger(ctx).Warningf("Failed to create a metric fetching helper for %s in %s: %v", md.Name, metricName, err)
   627  		} else {
   628  			log.GetLogger(ctx).Info("Successfully created metric fetching helper for", md.Name, metricName)
   629  		}
   630  	}
   631  	return nil
   632  }
