   218  	db, _ := pgxmock.NewPool()
   219  	src := sources.Source{
   220  		Name:      "test",

   223  		ConnStr:   "postgres://user:password@localhost:5432/mydatabase",
   224  	}
   225  	reader := testSourceReader{Sources: sources.Sources{src}}
   226  	// first read the sources

   228  	assert.NotNil(t, mds, "GetSources() = nil, want not nil")
   229  	// then resolve the databases

   232  	// pretend that we have a connection
   233  	mdbs[0].Conn = db
   234  	db.ExpectClose()
   235  	// sync the databases and make sure they are the same
   236  	newmdbs, _ := mdbs.SyncFromReader(reader)

   238  	assert.Equal(t, mdbs[0].ConnStr, newmdbs[0].ConnStr)
   239  	assert.Equal(t, db, newmdbs[0].Conn)
   240  	// change the connection string and check if databases are updated

   242  	newmdbs, _ = mdbs.SyncFromReader(reader)
   243  	assert.NotNil(t, newmdbs)
   244  	assert.NotEqual(t, mdbs[0].ConnStr, newmdbs[0].ConnStr)
   245  	assert.Nil(t, newmdbs[0].Conn)
   246  	assert.NoError(t, db.ExpectationsWereMet())
   247  	// change the unique name of the source and check if it's updated
   248  	reader.Sources[0].Name = "another"
   249  	newmdbs, _ = mdbs.SyncFromReader(reader)
   250  	assert.NotNil(t, newmdbs)
