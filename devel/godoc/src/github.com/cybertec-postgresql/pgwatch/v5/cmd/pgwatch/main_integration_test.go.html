    15  // startMain starts main() in a goroutine and returns a channel that is closed
    16  // when main() fully returns (including its deferred Exit() call). This must be
    17  // used instead of `go main()` + `<-mainCtx.Done()` because mainCtx.Done() closes
    18  // as soon as cancel() is called, before main()'s deferred Exit() has executed,
    19  // causing a race with the test's `defer func() { Exit = os.Exit }()`.
    20  func startMain() <-chan struct{} {
    21  	done := make(chan struct{})
    22  	go func() {
    23  		defer close(done)
    24  		main()
    25  	}()
    26  	return done
    27  }
