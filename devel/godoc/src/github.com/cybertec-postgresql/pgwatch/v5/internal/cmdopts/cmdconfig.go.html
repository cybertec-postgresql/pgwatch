    86  func (cmd *ConfigUpgradeCommand) Execute([]string) (err error) {
    87  	opts := cmd.owner
    88  	if err = opts.ValidateConfig(); err != nil {
    89  		return
    90  	}
    91  	ctx := context.Background()
    92  	// Upgrade metrics/sources configuration if it's postgres
    93  	if opts.IsPgConnStr(opts.Metrics.Metrics) && opts.IsPgConnStr(opts.Sources.Sources) {
    94  		err = opts.InitMetricReader(ctx)
    95  		if err != nil {
    96  			opts.CompleteCommand(ExitCodeConfigError)
    97  			return
    98  		}
    99  		if m, ok := opts.MetricsReaderWriter.(metrics.Migrator); ok {
   100  			err = m.Migrate()
   101  			if err != nil {
   102  				opts.CompleteCommand(ExitCodeConfigError)
   103  				return
   104  			}
   105  		}
   106  	} else {
   107  		opts.CompleteCommand(ExitCodeConfigError)
   108  		return errors.New("configuration storage does not support upgrade")
   109  	}
   110  	// Upgrade sinks configuration if it's postgres
   111  	if len(opts.Sinks.Sinks) > 0 {
   112  		err = opts.InitSinkWriter(ctx)
   113  		if err != nil {
   114  			opts.CompleteCommand(ExitCodeConfigError)
   115  			return
   116  		}
   117  		if m, ok := opts.SinksWriter.(metrics.Migrator); ok {
   118  			err = m.Migrate()
   119  			if err != nil {
   120  				opts.CompleteCommand(ExitCodeConfigError)
   121  				return
   122  			}
   123  		} else {
   124  			opts.CompleteCommand(ExitCodeConfigError)
   125  			return errors.New("sink storage does not support upgrade")
   126  		}
   127  	}
   128  	opts.CompleteCommand(ExitCodeOK)
   129  	return
   130  }
