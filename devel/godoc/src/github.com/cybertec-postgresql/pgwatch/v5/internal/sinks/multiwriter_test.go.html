   262  		{

   267  			expectNeedsMigrate: false,
   268  			expectError:        false,
   269  		},
   270  		{
   271  			name: "multiple writers one needs migration",
   272  			writers: []sinks.Writer{
   273  				&mockWriter{needsMigration: false},
   274  				&mockWriter{needsMigration: true},
   275  			},
   276  			expectNeedsMigrate: true,
   277  			expectError:        false,
   278  		},
   279  		{
   280  			name: "error checking migration",
   281  			writers: []sinks.Writer{
   282  				&mockWriter{needsMigrationErr: assert.AnError},
   283  			},
   284  			expectNeedsMigrate: false,
   285  			expectError:        true,
   286  		},
   287  		{
   288  			name: "mixed writers one needs migration",
   289  			writers: []sinks.Writer{
   290  				&mockWriter{},
   291  				&mockWriter{needsMigration: true},
   292  			},
   293  			expectNeedsMigrate: true,
   294  			expectError:        false,
   295  		},
   296  	}
   297  
   298  	for _, tt := range tests {
   299  		t.Run(tt.name, func(t *testing.T) {
   300  			mw := &sinks.MultiWriter{}
