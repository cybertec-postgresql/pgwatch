   332  func TestConcurrentSourceUpdates(t *testing.T) {
   333  	a := assert.New(t)
   334  	tempDir := t.TempDir()
   335  	tempFile := filepath.Join(tempDir, "sources.yaml")
   336  
   337  	yamlrw, err := sources.NewYAMLSourcesReaderWriter(ctx, tempFile)
   338  	a.NoError(err)
   339  
   340  	err = yamlrw.WriteSources(sources.Sources{})
   341  	a.NoError(err)
   342  
   343  	numGoroutines := 10
   344  	var wg sync.WaitGroup
   345  
   346  	// Each goroutine will add a unique source
   347  	for id := range numGoroutines {
   348  		wg.Go(func() {
   349  			testSource := sources.Source{
   350  				Name:          fmt.Sprintf("source_%d", id),
   351  				ConnStr:       fmt.Sprintf("postgresql://localhost/test_%d", id),
   352  				Kind:          sources.SourcePostgres,
   353  				PresetMetrics: "basic",
   354  			}
   355  			time.Sleep(time.Millisecond * time.Duration(id%3))
   356  			err := yamlrw.UpdateSource(testSource)
   357  			a.NoError(err, "Error during concurrent update")
   358  		})
   359  	}
   360  
   361  	wg.Wait()
   362  
   363  	finalSources, err := yamlrw.GetSources()
   364  	a.NoError(err)
   365  	a.Equal(numGoroutines, len(finalSources), "Some updates were lost due to race condition!")
   366  }
