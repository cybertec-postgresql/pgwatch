   400  	a := assert.New(t)
   401  	tempDir := t.TempDir()
   402  	tempFile := filepath.Join(tempDir, "metrics.yaml")
   403  
   404  	yamlrw, err := metrics.NewYAMLMetricReaderWriter(ctx, tempFile)
   405  	a.NoError(err)
   406  
   407  	// Create initial empty metrics file
   408  	initialMetrics := &metrics.Metrics{
   409  		MetricDefs: make(map[string]metrics.Metric),
   410  		PresetDefs: make(map[string]metrics.Preset),
   411  	}
   412  	err = yamlrw.WriteMetrics(initialMetrics)
   413  	a.NoError(err)
   414  
   415  	numGoroutines := 10
   416  	var wg sync.WaitGroup
   417  
   418  	// Each goroutine will add a unique metric
   419  	for id := range numGoroutines {
   420  		wg.Go(func() {
   421  			metricName := fmt.Sprintf("metric_%d", id)
   422  			testMetric := metrics.Metric{
   423  				Description: fmt.Sprintf("Test metric %d", id),
   424  				SQLs: map[int]string{
   425  					1: fmt.Sprintf("SELECT %d", id),
   426  				},
   427  			}
   428  			time.Sleep(time.Millisecond * time.Duration(id%3))
   429  			err := yamlrw.UpdateMetric(metricName, testMetric)
   430  			a.NoError(err, "Error during concurrent update")
   431  		})
   432  	}
   433  	wg.Wait()
   434  
   435  	finalMetrics, err := yamlrw.GetMetrics()
   436  	a.NoError(err)
   437  	a.Equal(numGoroutines, len(finalMetrics.MetricDefs), "Some updates were lost due to race condition!")
   438  }
