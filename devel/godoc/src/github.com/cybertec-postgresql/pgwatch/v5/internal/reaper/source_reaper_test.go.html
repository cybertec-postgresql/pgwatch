   540  // semPool wraps a db.PgxPoolIface and enforces single-connection semantics via a
   541  // semaphore channel of capacity 1. SendBatch acquires the semaphore; the returned
   542  // semBatchResults releases it on Close. Query non-blockingly tries to acquire the
   543  // semaphore and returns an error if it is still held by an unclosed batch result.
   544  // This simulates a pgxpool.Pool with MaxConns=1 to detect the deadlock described
   545  // in issue #1412 without a real database.
   546  type semPool struct {
   547  	db.PgxPoolIface
   548  	sem chan struct{}
   549  }
   550  
   551  func newSemPool(inner db.PgxPoolIface) *semPool {
   552  	return &semPool{PgxPoolIface: inner, sem: make(chan struct{}, 1)}
   553  }
   554  
   555  func (p *semPool) SendBatch(ctx context.Context, b *pgx.Batch) pgx.BatchResults {
   556  	p.sem <- struct{}{}
   557  	return &semBatchResults{inner: p.PgxPoolIface.SendBatch(ctx, b), pool: p}
   558  }
   559  
   560  func (p *semPool) Query(ctx context.Context, sql string, args ...any) (pgx.Rows, error) {
   561  	select {
   562  	case p.sem <- struct{}{}:
   563  		defer func() { <-p.sem }()
   564  	default:
   565  		return nil, fmt.Errorf("deadlock: batch connection not released before retry")
   566  	}
   567  	return p.PgxPoolIface.Query(ctx, sql, args...)
   568  }

   584  // TestSourceReaper_ExecuteBatch_NoDeadlockOnRetry is a regression test for issue #1412.
   585  // It verifies that the batch connection is released before individual retries run, preventing
   586  // a deadlock when max-parallel-connections-per-db=1.
   587  //
   588  // pgxmock does not simulate pool locking, so a semPool wrapper enforces single-connection
   589  // semantics: SendBatch acquires the semaphore; Close releases it; Query returns an error
   590  // if called while the semaphore is still held (i.e. the batch was not closed first).
   591  func TestSourceReaper_ExecuteBatch_NoDeadlockOnRetry(t *testing.T) {
   592  	ctx := log.WithLogger(context.Background(), log.NewNoopLogger())
   593  
   594  	metricDefs.MetricDefs["dl_bad"] = metrics.Metric{
   595  		SQLs: metrics.SQLs{0: "; -- invalid sql"},
   596  	}
   597  	metricDefs.MetricDefs["dl_good"] = metrics.Metric{
   598  		SQLs: metrics.SQLs{0: "SELECT 1 as value, 100::bigint as epoch_ns"},
   599  	}
   600  
   601  	mock, err := pgxmock.NewPool()
   602  	require.NoError(t, err)
   603  	defer mock.Close()
   604  
   605  	md := &sources.SourceConn{
   606  		Source: sources.Source{
   607  			Name:    "deadlock_test",
   608  			Kind:    sources.SourcePostgres,
   609  			Metrics: metrics.MetricIntervals{"dl_bad": 30, "dl_good": 30},
   610  		},
   611  		Conn: newSemPool(mock),
   612  		RuntimeInfo: sources.RuntimeInfo{
   613  			Version:     120000,
   614  			ChangeState: make(map[string]map[string]string),
   615  		},
   616  	}
   617  	r := &Reaper{
   618  		Options: &cmdopts.Options{
   619  			Metrics: metrics.CmdOpts{},
   620  			Sinks:   sinks.CmdOpts{},
   621  		},
   622  		measurementCh:    make(chan metrics.MeasurementEnvelope, 10),
   623  		measurementCache: NewInstanceMetricCache(),
   624  	}
   625  	sr := NewSourceReaper(r, md)
   626  
   627  	entries := []batchEntry{
   628  		{name: "dl_bad", metric: metricDefs.MetricDefs["dl_bad"], sql: "; -- invalid sql"},
   629  		{name: "dl_good", metric: metricDefs.MetricDefs["dl_good"], sql: "SELECT 1 as value, 100::bigint as epoch_ns"},
   630  	}
   631  
   632  	// Batch: dl_bad fails, dl_good cascades → both are retried individually.
   633  	// semPool.Query succeeds only if br.Close() was called first (semaphore released).
   634  	// Without the fix, Query returns a "deadlock" error and dl_good is wrongly degraded.
   635  	eb := mock.ExpectBatch()
   636  	eb.ExpectQuery("; -- invalid sql").WillReturnError(assert.AnError)
   637  	eb.ExpectQuery("SELECT 1").WillReturnError(assert.AnError) // cascade

   640  	// dl_good succeeds on retry → not degraded
   641  	mock.ExpectQuery("SELECT 1").WithArgs(pgx.QueryExecModeSimpleProtocol).
   642  		WillReturnRows(pgxmock.NewRows([]string{"epoch_ns", "value"}).AddRow(time.Now().UnixNano(), int64(1)))
   643  
   644  	err = sr.executeBatch(ctx, entries)
   645  	assert.Error(t, err, "dl_bad error should propagate")
   646  	assert.Contains(t, sr.degradedMetrics, "dl_bad", "dl_bad should be marked degraded")
   647  	assert.NotContains(t, sr.degradedMetrics, "dl_good", "dl_good must not be degraded (cascade-only victim)")
   648  	assert.NoError(t, mock.ExpectationsWereMet())
   649  }
