    86  func NewWriterFromPostgresConn(ctx context.Context, conn db.PgxPoolIface, opts *CmdOpts) (pgw *PostgresWriter, err error) {
    87  	l := log.GetLogger(ctx).WithField("sink", "postgres").WithField("db", conn.Config().ConnConfig.Database)
    88  	ctx = log.WithLogger(ctx, l)
    89  	pgw = &PostgresWriter{
    90  		ctx:                      ctx,
    91  		opts:                     opts,
    92  		input:                    make(chan metrics.MeasurementEnvelope, cacheLimit),
    93  		lastError:                make(chan error),
    94  		sinkDb:                   conn,
    95  		forceRecreatePartitions:  false,
    96  		partitionMapMetric:       make(map[string]ExistingPartitionInfo),
    97  		partitionMapMetricDbname: make(map[string]map[string]ExistingPartitionInfo),
    98  	}
    99  	l.Info("initialising measurements database...")
   100  	if err = pgw.init(); err != nil {
   101  		return nil, err
   102  	}
   103  	if needsMigration, e := pgw.NeedsMigration(); e != nil {
   104  		return nil, e
   105  	} else if needsMigration {
   106  		return nil, ErrNeedsMigration
   107  	}
   108  	if err = pgw.ReadMetricSchemaType(); err != nil {
   109  		return nil, err
   110  	}
   111  	if err = pgw.EnsureBuiltinMetricDummies(); err != nil {
   112  		return nil, err
   113  	}
   114  	pgw.scheduleJob(pgw.maintenanceInterval, func() {
   115  		pgw.DeleteOldPartitions()
   116  		pgw.MaintainUniqueSources()
   117  	})
   118  	go pgw.poll()
   119  	l.Info(`measurements sink is activated`)
   120  	return

   123  func (pgw *PostgresWriter) init() (err error) {
   124  	return db.Init(pgw.ctx, pgw.sinkDb, func(ctx context.Context, conn db.PgxIface) error {
   125  		var isValidPartitionInterval bool
   126  		if err = conn.QueryRow(ctx,
   127  			"SELECT extract(epoch from $1::interval), extract(epoch from $2::interval), $3::interval >= '1h'::interval",
   128  			pgw.opts.RetentionInterval, pgw.opts.MaintenanceInterval, pgw.opts.PartitionInterval,
   129  		).Scan(&pgw.retentionInterval, &pgw.maintenanceInterval, &isValidPartitionInterval); err != nil {
   130  			return err
   131  		}
   132  
   133  		// epoch returns seconds but time.Duration represents nanoseconds
   134  		pgw.retentionInterval *= time.Second
   135  		pgw.maintenanceInterval *= time.Second
   136  
   137  		if !isValidPartitionInterval {
   138  			return fmt.Errorf("--partition-interval must be at least 1 hour, got: %s", pgw.opts.PartitionInterval)
   139  		}
   140  		if pgw.maintenanceInterval < 0 {
   141  			return errors.New("--maintenance-interval must be a positive PostgreSQL interval or 0 to disable it")
   142  		}
   143  		if pgw.retentionInterval < time.Hour && pgw.retentionInterval != 0 {
   144  			return errors.New("--retention must be at least 1 hour PostgreSQL interval or 0 to disable it")
   145  		}
   146  
   147  		exists, err := db.DoesSchemaExist(ctx, conn, "admin")
   148  		if err != nil || exists {
   149  			return err
   150  		}
   151  		for _, sql := range metricSchemaSQLs {
   152  			if _, err = conn.Exec(ctx, sql); err != nil {
   153  				return err
   154  			}
   155  		}
   156  		return nil
   157  	})
   158  }

   249  // poll is the main loop that reads from the input channel and flushes the data to the database
   250  func (pgw *PostgresWriter) poll() {
   251  	cache := make([]metrics.MeasurementEnvelope, 0, cacheLimit)
   252  	cacheTimeout := pgw.opts.BatchingDelay
   253  	tick := time.NewTicker(cacheTimeout)
   254  	for {
   255  		select {
   256  		case <-pgw.ctx.Done(): //check context with high priority
   257  			return
   258  		default:
   259  			select {
   260  			case entry := <-pgw.input:
   261  				cache = append(cache, entry)
   262  				if len(cache) < cacheLimit {
   263  					break
   264  				}
   265  				tick.Stop()
   266  				pgw.flush(cache)
   267  				cache = cache[:0]
   268  				tick = time.NewTicker(cacheTimeout)
   269  			case <-tick.C:
   270  				pgw.flush(cache)
   271  				cache = cache[:0]
   272  			case <-pgw.ctx.Done():
   273  				return
   274  			}
   275  		}
   276  	}
   277  }

   296  func (c *copyFromMeasurements) Next() bool {
   297  	for {
   298  		// Check if we need to advance to the next envelope
   299  		if c.envelopeIdx < 0 || c.measurementIdx+1 >= len(c.envelopes[c.envelopeIdx].Data) {
   300  			// Advance to next envelope
   301  			if ok := c.NextEnvelope(); !ok {
   302  				return false // No more envelopes
   303  			}
   304  			// Set metric name from first envelope, or detect metric boundary
   305  			if c.metricName == "" {
   306  				c.metricName = c.envelopes[c.envelopeIdx].MetricName
   307  			} else if c.metricName != c.envelopes[c.envelopeIdx].MetricName {
   308  				// We've hit a different metric - we're done with current metric
   309  				// Reset position to process this envelope on next call
   310  				c.envelopeIdx--
   311  				c.measurementIdx = len(c.envelopes[c.envelopeIdx].Data) // Set to length so we've "finished" this envelope
   312  				c.metricName = ""                                       // Reset for next metric
   313  				return false
   314  			}
   315  		}
   316  
   317  		// Advance to next measurement in current envelope
   318  		c.measurementIdx++
   319  		if c.measurementIdx < len(c.envelopes[c.envelopeIdx].Data) {
   320  			return true // Found valid measurement
   321  		}
   322  		// If we reach here, we've exhausted current envelope, loop will advance to next envelope
   323  	}
   324  }

   330  func (c *copyFromMeasurements) Values() ([]any, error) {
   331  	row := maps.Clone(c.envelopes[c.envelopeIdx].Data[c.measurementIdx])
   332  	tagRow := maps.Clone(c.envelopes[c.envelopeIdx].CustomTags)
   333  	if tagRow == nil {
   334  		tagRow = make(map[string]string)
   335  	}
   336  	for k, v := range row {
   337  		if after, ok := strings.CutPrefix(k, metrics.TagPrefix); ok {
   338  			tagRow[after] = fmt.Sprintf("%v", v)
   339  			delete(row, k)
   340  		}
   341  	}
   342  	jsonTags, terr := jsoniter.ConfigFastest.MarshalToString(tagRow)
   343  	json, err := jsoniter.ConfigFastest.MarshalToString(row)
   344  	if err != nil || terr != nil {
   345  		return nil, errors.Join(err, terr)
   346  	}
   347  	return []any{time.Unix(0, c.envelopes[c.envelopeIdx].Data.GetEpoch()), c.envelopes[c.envelopeIdx].DBName, json, jsonTags}, nil
   348  }

   362  // flush sends the cached measurements to the database
   363  func (pgw *PostgresWriter) flush(msgs []metrics.MeasurementEnvelope) {
   364  	if len(msgs) == 0 {
   365  		return
   366  	}
   367  	logger := log.GetLogger(pgw.ctx)
   368  	pgPartBounds := make(map[string]ExistingPartitionInfo)                  // metric=min/max
   369  	pgPartBoundsDbName := make(map[string]map[string]ExistingPartitionInfo) // metric=[dbname=min/max]
   370  	var err error
   371  
   372  	slices.SortFunc(msgs, func(a, b metrics.MeasurementEnvelope) int {
   373  		if a.MetricName < b.MetricName {
   374  			return -1
   375  		} else if a.MetricName > b.MetricName {
   376  			return 1
   377  		}
   378  		return 0
   379  	})
   380  
   381  	for _, msg := range msgs {
   382  		for _, dataRow := range msg.Data {
   383  			epochTime := time.Unix(0, metrics.Measurement(dataRow).GetEpoch())
   384  			switch pgw.metricSchema {
   385  			case DbStorageSchemaTimescale:
   386  				// set min/max timestamps to check/create partitions
   387  				bounds, ok := pgPartBounds[msg.MetricName]
   388  				if !ok || (ok && epochTime.Before(bounds.StartTime)) {
   389  					bounds.StartTime = epochTime
   390  					pgPartBounds[msg.MetricName] = bounds
   391  				}
   392  				if !ok || (ok && epochTime.After(bounds.EndTime)) {
   393  					bounds.EndTime = epochTime
   394  					pgPartBounds[msg.MetricName] = bounds
   395  				}
   396  			case DbStorageSchemaPostgres:
   397  				_, ok := pgPartBoundsDbName[msg.MetricName]
   398  				if !ok {
   399  					pgPartBoundsDbName[msg.MetricName] = make(map[string]ExistingPartitionInfo)
   400  				}
   401  				bounds, ok := pgPartBoundsDbName[msg.MetricName][msg.DBName]
   402  				if !ok || (ok && epochTime.Before(bounds.StartTime)) {
   403  					bounds.StartTime = epochTime
   404  					pgPartBoundsDbName[msg.MetricName][msg.DBName] = bounds
   405  				}
   406  				if !ok || (ok && epochTime.After(bounds.EndTime)) {
   407  					bounds.EndTime = epochTime
   408  					pgPartBoundsDbName[msg.MetricName][msg.DBName] = bounds
   409  				}
   410  			default:
   411  				logger.Fatal("unknown storage schema...")
   412  			}
   413  		}
   414  	}
   415  
   416  	switch pgw.metricSchema {
   417  	case DbStorageSchemaPostgres:
   418  		err = pgw.EnsureMetricDbnameTime(pgPartBoundsDbName)
   419  	case DbStorageSchemaTimescale:
   420  		err = pgw.EnsureMetricTimescale(pgPartBounds)
   421  	default:
   422  		logger.Fatal("unknown storage schema...")
   423  	}
   424  	pgw.forceRecreatePartitions = false
   425  	if err != nil {
   426  		pgw.lastError <- err
   427  	}
   428  
   429  	var rowsBatched, n int64
   430  	t1 := time.Now()
   431  	cfm := newCopyFromMeasurements(msgs)
   432  	for !cfm.EOF() {
   433  		n, err = pgw.sinkDb.CopyFrom(context.Background(), cfm.MetricName(), targetColumns[:], cfm)
   434  		rowsBatched += n
   435  		if err != nil {
   436  			logger.Error(err)
   437  			if PgError, ok := err.(*pgconn.PgError); ok {
   438  				pgw.forceRecreatePartitions = PgError.Code == "23514"
   439  			}
   440  			if pgw.forceRecreatePartitions {
   441  				logger.Warning("Some metric partitions might have been removed, halting all metric storage. Trying to re-create all needed partitions on next run")
   442  			}
   443  		}
   444  	}
   445  	diff := time.Since(t1)
   446  	if err == nil {
   447  		logger.WithField("rows", rowsBatched).WithField("elapsed", diff).Info("measurements written")
   448  		return
   449  	}
   450  	pgw.lastError <- err
   451  }

   468  func (pgw *PostgresWriter) EnsureMetricDbnameTime(metricDbnamePartBounds map[string]map[string]ExistingPartitionInfo) (err error) {
   469  	var rows pgx.Rows
   470  	sqlEnsure := `select * from admin.ensure_partition_metric_dbname_time($1, $2, $3, $4)`
   471  	for metric, dbnameTimestampMap := range metricDbnamePartBounds {
   472  		_, ok := pgw.partitionMapMetricDbname[metric]
   473  		if !ok {
   474  			pgw.partitionMapMetricDbname[metric] = make(map[string]ExistingPartitionInfo)
   475  		}
   476  
   477  		for dbname, pb := range dbnameTimestampMap {
   478  			if pb.StartTime.IsZero() || pb.EndTime.IsZero() {
   479  				return fmt.Errorf("zero StartTime/EndTime in partitioning request: [%s:%v]", metric, pb)
   480  			}
   481  			partInfo, ok := pgw.partitionMapMetricDbname[metric][dbname]
   482  			if !ok || (ok && (pb.StartTime.Before(partInfo.StartTime))) || pgw.forceRecreatePartitions {
   483  				if rows, err = pgw.sinkDb.Query(pgw.ctx, sqlEnsure, metric, dbname, pb.StartTime, pgw.opts.PartitionInterval); err != nil {
   484  					return
   485  				}
   486  				if partInfo, err = pgx.CollectOneRow(rows, pgx.RowToStructByPos[ExistingPartitionInfo]); err != nil {
   487  					return err
   488  				}
   489  				pgw.partitionMapMetricDbname[metric][dbname] = partInfo
   490  			}
   491  			if pb.EndTime.After(partInfo.EndTime) || pb.EndTime.Equal(partInfo.EndTime) || pgw.forceRecreatePartitions {
   492  				if rows, err = pgw.sinkDb.Query(pgw.ctx, sqlEnsure, metric, dbname, pb.EndTime, pgw.opts.PartitionInterval); err != nil {
   493  					return
   494  				}
   495  				if partInfo, err = pgx.CollectOneRow(rows, pgx.RowToStructByPos[ExistingPartitionInfo]); err != nil {
   496  					return err
   497  				}
   498  				pgw.partitionMapMetricDbname[metric][dbname] = partInfo
   499  			}
   500  		}
   501  	}
   502  	return nil
   503  }

   519  // in each metric table in admin.all_distinct_dbname_metrics.
   520  // This is used to avoid listing the same source multiple times in Grafana dropdowns.
   521  func (pgw *PostgresWriter) MaintainUniqueSources() {
   522  	logger := log.GetLogger(pgw.ctx)
   523  	var rowsAffected int
