   164  func getFileWithLatestTimestamp(files []string) (string, error) {
   165  	var maxDate time.Time
   166  	var latest string
   167  
   168  	for _, f := range files {
   169  		fi, err := os.Stat(f)
   170  		if err != nil {
   171  			return "", err
   172  		}
   173  		if fi.ModTime().After(maxDate) {
   174  			latest = f
   175  			maxDate = fi.ModTime()
   176  		}
   177  	}
   178  	return latest, nil

   181  func getFileWithNextModTimestamp(logsGlobPath, currentFile string) (string, error) {
   182  	var nextFile string
   183  	var nextMod time.Time
   184  
   185  	files, err := filepath.Glob(logsGlobPath)
   186  	if err != nil {

   188  	}
   189  
   190  	fiCurrent, err := os.Stat(currentFile)
   191  	if err != nil {
   192  		return "", err
   193  	}
   194  
   195  	for _, f := range files {
   196  		if f == currentFile {
   197  			continue
   198  		}
   199  		fi, err := os.Stat(f)
   200  		if err != nil {
   201  			continue
   202  		}
   203  		if (nextMod.IsZero() || fi.ModTime().Before(nextMod)) && fi.ModTime().After(fiCurrent.ModTime()) {
   204  			nextMod = fi.ModTime()
   205  			nextFile = f
   206  		}
   207  	}
   208  	return nextFile, nil
