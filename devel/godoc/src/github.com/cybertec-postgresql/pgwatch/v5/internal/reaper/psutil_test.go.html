   129  
   130  	// Same path twice → same device ID → both rows reported with the same usage values
   131  	pgDirs := metrics.Measurements{
   132  		{"name": "data_directory", "path": tmpDir},
   133  		{"name": "pg_wal", "path": tmpDir},
   134  	}

   136  	a.NoError(err)
   137  	a.Len(result, 2)
   138  	a.Equal("data_directory", result[0]["tag_dir_or_tablespace"])
   139  	a.Equal("pg_wal", result[1]["tag_dir_or_tablespace"])
   140  	// Usage values must be identical since they share a device
