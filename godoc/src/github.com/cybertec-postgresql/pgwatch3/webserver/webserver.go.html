    59  func (Server *WebUIServer) handleStatic(w http.ResponseWriter, r *http.Request) {
    60  	if r.Method != "GET" {
    61  		http.Error(w, http.StatusText(http.StatusMethodNotAllowed), http.StatusMethodNotAllowed)
    62  		return
    63  	}
    64  	routes := []string{"/", "/sources", "/metrics", "/presets", "/logs"}
    65  	path := r.URL.Path
    66  	if slices.Contains(routes, path) {
    67  		path = "index.html"
    68  	} else {
    69  		path = strings.TrimPrefix(path, "/")
    70  	}
    71  
    72  	file, err := Server.uiFS.Open(path)
    73  	if err != nil {
    74  		if os.IsNotExist(err) {
    75  			Server.l.Println("file", path, "not found:", err)
    76  			http.NotFound(w, r)
    77  			return
    78  		}
    79  		Server.l.Println("file", path, "cannot be read:", err)
    80  		http.Error(w, http.StatusText(http.StatusInternalServerError), http.StatusInternalServerError)
    81  		return
    82  	}
    83  
    84  	contentType := mime.TypeByExtension(filepath.Ext(path))
    85  	w.Header().Set("Content-Type", contentType)
    86  	if strings.HasPrefix(path, "static/") {
    87  		w.Header().Set("Cache-Control", "public, max-age=31536000")
    88  	}
    89  	stat, err := file.Stat()
    90  	if err == nil && stat.Size() > 0 {
    91  		w.Header().Set("Content-Length", fmt.Sprintf("%d", stat.Size()))
    92  	}
    93  
    94  	n, _ := io.Copy(w, file)
    95  	Server.l.Debug("file", path, "copied", n, "bytes")
