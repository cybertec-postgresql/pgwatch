    96  func (md *MonitoredDatabase) Ping(ctx context.Context) error {

   152  // SyncFromReader will update the monitored databases with the latest configuration from the reader.
   153  // Any resolution errors will be returned, e.g. etcd unavailability.
   154  // It's up to the caller to proceed with the databases available or stop the execution due to errors.
   155  func (mds MonitoredDatabases) SyncFromReader(r Reader) (newmds MonitoredDatabases, err error) {
   156  	srcs, err := r.GetSources()
   157  	if err != nil {
   158  		return nil, err
   159  	}
   160  	newmds, err = srcs.ResolveDatabases()
   161  	for _, newMD := range newmds {
   162  		if md := mds.GetMonitoredDatabase(newMD.Name); md != nil {
   163  			newMD.Conn = md.Conn
   164  			newMD.ConnConfig = md.ConnConfig
   165  		}
   166  	}
   167  	return newmds, err
   168  }
