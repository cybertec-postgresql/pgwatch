   136  func (mds MonitoredDatabases) SyncFromReader(r Reader) (newmds MonitoredDatabases, err error) {
   137  	srcs, err := r.GetSources()
   138  	if err != nil {
   139  		return nil, err
   140  	}
   141  	if newmds, err = srcs.ResolveDatabases(); err != nil {
   142  		return nil, err
   143  	}
   144  	for _, newMD := range newmds {
   145  		if md := mds.GetMonitoredDatabase(newMD.DBUniqueName); md != nil {
   146  			newMD.Conn = md.Conn
   147  			newMD.ConnConfig = md.ConnConfig
   148  		}
   149  	}
   150  	return newmds, nil
   151  }
