    30  	resolvedDbs := make(MonitoredDatabases, 0, len(srcs))
    31  	for _, s := range srcs {
    32  		if !s.IsEnabled {
    33  			continue
    34  		}
    35  		dbs, err := s.ResolveDatabases()
    36  		if err != nil {
    37  			return nil, err
    38  		}
    39  		if len(dbs) == 0 {
    40  			resolvedDbs = append(resolvedDbs, &MonitoredDatabase{Source: *(&s).Clone()})
    41  			continue
    42  		}
    43  		resolvedDbs = append(resolvedDbs, dbs...)
    44  	}
    45  	return resolvedDbs, nil
    46  }

    79  func parseHostAndPortFromJdbcConnStr(connStr string) (string, string, error) {
    80  	r := regexp.MustCompile(`postgres://(.*)+:([0-9]+)/`)
    81  	matches := r.FindStringSubmatch(connStr)
    82  	if len(matches) != 3 {
    83  		logger.Errorf("Unexpected regex result groups:", matches)
    84  		return "", "", fmt.Errorf("unexpected regex result groups: %v", matches)
    85  	}
    86  	return matches[1], matches[2], nil
    87  }

   104  func getTransport(conf HostConfigAttrs) (*tls.Config, error) {
   105  	var caCertPool *x509.CertPool
   106  
   107  	// create valid CertPool only if the ca certificate file exists
   108  	if conf.CAFile != "" {
   109  		caCert, err := os.ReadFile(conf.CAFile)
   110  		if err != nil {
   111  			return nil, fmt.Errorf("cannot load CA file: %s", err)
   112  		}
   113  
   114  		caCertPool = x509.NewCertPool()
   115  		caCertPool.AppendCertsFromPEM(caCert)
   116  	}
   117  
   118  	var certificates []tls.Certificate
   119  
   120  	// create valid []Certificate only if the client cert and key files exists
   121  	if conf.CertFile != "" && conf.KeyFile != "" {
   122  		cert, err := tls.LoadX509KeyPair(conf.CertFile, conf.KeyFile)
   123  		if err != nil {
   124  			return nil, fmt.Errorf("cannot load client cert or key file: %s", err)
   125  		}
   126  
   127  		certificates = []tls.Certificate{cert}
   128  	}
   129  
   130  	tlsClientConfig := new(tls.Config)
   131  
   132  	if caCertPool != nil {
   133  		tlsClientConfig.RootCAs = caCertPool
   134  		if certificates != nil {
   135  			tlsClientConfig.Certificates = certificates
   136  		}
   137  	}
   138  
   139  	return tlsClientConfig, nil
   140  }

   142  func getEtcdClusterMembers(s Source) ([]PatroniClusterMember, error) {
   143  	var ret = make([]PatroniClusterMember, 0)
   144  	var cfg client.Config
   145  
   146  	if len(s.HostConfig.DcsEndpoints) == 0 {
   147  		return ret, errors.New("Missing ETCD connect info, make sure host config has a 'dcs_endpoints' key")
   148  	}
   149  
   150  	tlsConfig, err := getTransport(s.HostConfig)
   151  	if err != nil {
   152  		return nil, err
   153  	}
   154  	cfg = client.Config{
   155  		Endpoints:            s.HostConfig.DcsEndpoints,
   156  		TLS:                  tlsConfig,
   157  		DialKeepAliveTimeout: time.Second,
   158  		Username:             s.HostConfig.Username,
   159  		Password:             s.HostConfig.Password,
   160  	}
   161  
   162  	c, err := client.New(cfg)
   163  	if err != nil {
   164  		logger.Errorf("[%s ]Could not connect to ETCD: %v", s.DBUniqueName, err)
   165  		return ret, err
   166  	}
   167  	kapi := c.KV
   168  
   169  	if s.Kind == SourcePatroniNamespace { // all scopes, all DBs (regex filtering applies if defined)
   170  		if len(s.GetDatabaseName()) > 0 {
   171  			return ret, fmt.Errorf("Skipping Patroni entry %s - cannot specify a DB name when monitoring all scopes (regex patterns are supported though)", s.DBUniqueName)
   172  		}
   173  		if s.HostConfig.Namespace == "" {
   174  			return ret, fmt.Errorf("Skipping Patroni entry %s - search 'namespace' not specified", s.DBUniqueName)
   175  		}
   176  		resp, err := kapi.Get(context.Background(), s.HostConfig.Namespace)
   177  		if err != nil {
   178  			return ret, err
   179  		}
   180  
   181  		for _, node := range resp.Kvs {
   182  			scope := path.Base(string(node.Key)) // Key="/service/batman"
   183  			scopeMembers, err := extractEtcdScopeMembers(s, scope, kapi, true)
   184  			if err != nil {
   185  				continue
   186  			}
   187  			ret = append(ret, scopeMembers...)
   188  		}
   189  	} else {
   190  		ret, err = extractEtcdScopeMembers(s, s.HostConfig.Scope, kapi, false)
   191  		if err != nil {
   192  			return ret, err
   193  		}
   194  	}
   195  	lastFoundClusterMembers[s.DBUniqueName] = ret
   196  	return ret, nil
   197  }

   199  func extractEtcdScopeMembers(s Source, scope string, kapi client.KV, addScopeToName bool) ([]PatroniClusterMember, error) {
   200  	var ret = make([]PatroniClusterMember, 0)
   201  	var name string
   202  	membersPath := path.Join(s.HostConfig.Namespace, scope, "members")
   203  
   204  	resp, err := kapi.Get(context.Background(), membersPath)
   205  	if err != nil {
   206  		return nil, err
   207  	}
   208  	logger.Debugf("ETCD response for %s scope %s: %+v", s.DBUniqueName, scope, resp)
   209  
   210  	for _, node := range resp.Kvs {
   211  		logger.Debugf("Found a cluster member from etcd [%s:%s]: %+v", s.DBUniqueName, scope, node.Value)
   212  		nodeData, err := jsonTextToStringMap(string(node.Value))
   213  		if err != nil {
   214  			logger.Errorf("Could not parse ETCD node data for node \"%s\": %s", node, err)
   215  			continue
   216  		}
   217  		role := nodeData["role"]
   218  		connURL := nodeData["conn_url"]
   219  		if addScopeToName {
   220  			name = scope + "_" + path.Base(string(node.Key))
   221  		} else {
   222  			name = path.Base(string(node.Key))
   223  		}
   224  
   225  		ret = append(ret, PatroniClusterMember{Scope: scope, ConnURL: connURL, Role: role, Name: name})
   226  	}
   227  	return ret, nil
   228  }

   236  func ResolveDatabasesFromPatroni(ce Source) ([]*MonitoredDatabase, error) {
   237  	var mds []*MonitoredDatabase
   238  	var cm []PatroniClusterMember
   239  	var err error
   240  	var ok bool
   241  	var dbUnique string
   242  
   243  	switch ce.HostConfig.DcsType {
   244  	case dcsTypeEtcd:
   245  		cm, err = getEtcdClusterMembers(ce)
   246  	case dcsTypeZookeeper:
   247  		cm, err = getZookeeperClusterMembers(ce)
   248  	case dcsTypeConsul:
   249  		cm, err = getConsulClusterMembers(ce)
   250  	default:
   251  		return nil, errors.New("unknown DCS")
   252  	}
   253  	if err != nil {
   254  		logger.Warningf("Failed to get info from DCS for %s, using previous member info if any", ce.DBUniqueName)
   255  		cm, ok = lastFoundClusterMembers[ce.DBUniqueName]
   256  		if ok { // mask error from main loop not to remove monitored DBs due to "jitter"
   257  			err = nil
   258  		}
   259  	} else {
   260  		lastFoundClusterMembers[ce.DBUniqueName] = cm
   261  	}
   262  	if len(cm) == 0 {
   263  		logger.Warningf("No Patroni cluster members found for cluster [%s:%s]", ce.DBUniqueName, ce.HostConfig.Scope)
   264  		return mds, nil
   265  	}
   266  	logger.Infof("Found %d Patroni members for entry %s", len(cm), ce.DBUniqueName)
   267  
   268  	for _, m := range cm {
   269  		logger.Infof("Processing Patroni cluster member [%s:%s]", ce.DBUniqueName, m.Name)
   270  		if ce.OnlyIfMaster && m.Role != "master" {
   271  			logger.Infof("Skipping over Patroni cluster member [%s:%s] as not a master", ce.DBUniqueName, m.Name)
   272  			continue
   273  		}
   274  		host, port, err := parseHostAndPortFromJdbcConnStr(m.ConnURL)
   275  		if err != nil {
   276  			logger.Errorf("Could not parse Patroni conn str \"%s\" [%s:%s]: %v", m.ConnURL, ce.DBUniqueName, m.Scope, err)
   277  			continue
   278  		}
   279  		if ce.OnlyIfMaster {
   280  			dbUnique = ce.DBUniqueName
   281  			if ce.Kind == SourcePatroniNamespace {
   282  				dbUnique = ce.DBUniqueName + "_" + m.Scope
   283  			}
   284  		} else {
   285  			dbUnique = ce.DBUniqueName + "_" + m.Name
   286  		}
   287  		if ce.GetDatabaseName() != "" {
   288  			c := &MonitoredDatabase{Source: *ce.Clone()}
   289  			c.DBUniqueName = dbUnique
   290  			mds = append(mds, c)
   291  			continue
   292  		}
   293  		c, err := db.New(context.TODO(), ce.ConnStr,
   294  			func(c *pgxpool.Config) error {
   295  				c.ConnConfig.Host = host
   296  				c.ConnConfig.Database = "template1"
   297  				i, err := strconv.ParseUint(port, 10, 16)
   298  				c.ConnConfig.Port = uint16(i)
   299  				mds[len(mds)].ConnStr = c.ConnString()
   300  				return err
   301  			})
   302  		if err != nil {
   303  			logger.Errorf("Could not contact Patroni member [%s:%s]: %v", ce.DBUniqueName, m.Scope, err)
   304  			continue
   305  		}
   306  		defer c.Close()
   307  		sql := `select datname::text as datname,
   308  					quote_ident(datname)::text as datname_escaped
   309  					from pg_database
   310  					where not datistemplate
   311  					and datallowconn
   312  					and has_database_privilege (datname, 'CONNECT')
   313  					and case when length(trim($1)) > 0 then datname ~ $1 else true end
   314  					and case when length(trim($2)) > 0 then not datname ~ $2 else true end`
   315  
   316  		rows, err := c.Query(context.TODO(), sql, ce.IncludePattern, ce.ExcludePattern)
   317  		if err != nil {
   318  			return nil, err
   319  		}
   320  		data, err := pgx.CollectRows(rows, pgx.RowToMap)
   321  		if err != nil {
   322  			logger.Errorf("Could not get DB name listing from Patroni member [%s:%s]: %v", ce.DBUniqueName, m.Scope, err)
   323  			continue
   324  		}
   325  
   326  		for _, d := range data {
   327  			connURL, err := url.Parse(ce.ConnStr)
   328  			if err != nil {
   329  				continue
   330  			}
   331  			connURL.Host = host + ":" + port
   332  			connURL.Path = d["datname"].(string)
   333  			c := ce.Clone()
   334  			c.DBUniqueName = dbUnique + "_" + d["datname_escaped"].(string)
   335  			c.ConnStr = connURL.String()
   336  			mds = append(mds, &MonitoredDatabase{Source: *c})
   337  		}
   338  
   339  	}
   340  
   341  	return mds, err
   342  }

   345  func ResolveDatabasesFromPostgres(s Source) (resolvedDbs MonitoredDatabases, err error) {
   346  	var (
   347  		c      db.PgxPoolIface
   348  		dbname string
   349  		rows   pgx.Rows
   350  	)
   351  	c, err = db.New(context.TODO(), s.ConnStr)
   352  	if err != nil {
   353  		return
   354  	}
   355  	defer c.Close()
   356  
   357  	sql := `select /* pgwatch3_generated */
   358  		quote_ident(datname)::text as datname_escaped
   359  		from pg_database
   360  		where not datistemplate
   361  		and datallowconn
   362  		and has_database_privilege (datname, 'CONNECT')
   363  		and case when length(trim($1)) > 0 then datname ~ $1 else true end
   364  		and case when length(trim($2)) > 0 then not datname ~ $2 else true end`
   365  
   366  	if rows, err = c.Query(context.TODO(), sql, s.IncludePattern, s.ExcludePattern); err != nil {
   367  		return nil, err
   368  	}
   369  	for rows.Next() {
   370  		if err = rows.Scan(&dbname); err != nil {
   371  			return nil, err
   372  		}
   373  		rdb := &MonitoredDatabase{Source: *s.Clone()}
   374  		rdb.DBUniqueName += "_" + dbname
   375  		rdb.SetDatabaseName(dbname)
   376  		resolvedDbs = append(resolvedDbs, rdb)
   377  	}
   378  
   379  	if err := rows.Err(); err != nil {
   380  		return nil, err
   381  	}
   382  	return
   383  }
