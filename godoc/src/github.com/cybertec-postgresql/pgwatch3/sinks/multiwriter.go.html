    41  			w, err = NewJSONWriter(ctx, path)

    46  		case "rpc":
    47  			w, err = NewRPCWriter(ctx, path)
    48  		default:
    49  			return nil, fmt.Errorf("unknown schema %s in sink URI %s", scheme, s)
    50  		}
    51  		if err != nil {
    52  			return nil, err
    53  		}
    54  		mw.AddWriter(w)
    55  		l.Info(`measurements sink actviated`)
