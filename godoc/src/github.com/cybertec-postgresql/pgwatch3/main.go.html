    40  // SetupCloseHandler creates a 'listener' on a new goroutine which will notify the
    41  // program if it receives an interrupt from the OS. We then handle this by calling
    42  // our clean up procedure and exiting the program.
    43  func SetupCloseHandler(cancel context.CancelFunc) {
    44  	c := make(chan os.Signal, 2)
    45  	signal.Notify(c, os.Interrupt, syscall.SIGTERM)
    46  	go func() {
    47  		<-c
    48  		logger.Debug("SetupCloseHandler received an interrupt from OS. Closing session...")
    49  		cancel()
    50  		exitCode.Store(ExitCodeUserCancel)
    51  	}()
    52  }

    72  // NewConfigurationReaders creates the configuration readers based on the configuration kind from the options.
    73  func NewConfigurationReaders(opts *config.Options) (sourcesReader sources.ReaderWriter, metricsReader metrics.ReaderWriter, err error) {
    74  	var configKind config.Kind
    75  	configKind, err = opts.GetConfigKind()
    76  	switch {
    77  	case err != nil:
    78  		return
    79  	case configKind != config.ConfigPgURL:
    80  		ctx := log.WithLogger(mainContext, logger.WithField("config", "files"))
    81  		if sourcesReader, err = sources.NewYAMLSourcesReaderWriter(ctx, opts.Sources.Config); err != nil {
    82  			return
    83  		}
    84  		metricsReader, err = metrics.NewYAMLMetricReaderWriter(ctx, opts.Metrics.Metrics)
    85  	default:
    86  		var configDb db.PgxPoolIface
    87  		ctx := log.WithLogger(mainContext, logger.WithField("config", "postgres"))
    88  		if configDb, err = db.New(ctx, opts.Sources.Config); err != nil {
    89  			return
    90  		}
    91  		if metricsReader, err = metrics.NewPostgresMetricReaderWriter(ctx, configDb); err != nil {
    92  			return
    93  		}
    94  		sourcesReader, err = sources.NewPostgresSourcesReaderWriter(ctx, configDb)
    95  	}
    96  	return
    97  }

   100  func UpgradeConfiguration(m metrics.Migrator) (err error) {
   101  	if opts.Upgrade {

   105  		upgrade, err = m.NeedsMigration()
   106  		if upgrade {
   107  			err = errors.Join(err, errors.New("configuration needs upgrade, use --upgrade option"))

   109  	}
   110  	return

   113  func main() {
   114  	var (
   115  		err    error
   116  		cancel context.CancelFunc
   117  	)
   118  	exitCode.Store(ExitCodeOK)
   119  	defer func() {
   120  		if err := recover(); err != nil {
   121  			exitCode.Store(ExitCodeFatalError)
   122  			log.GetLogger(mainContext).WithField("callstack", string(debug.Stack())).Error(err)
   123  		}
   124  		os.Exit(int(exitCode.Load()))
   125  	}()
   126  
   127  	mainContext, cancel = context.WithCancel(context.Background())
   128  	SetupCloseHandler(cancel)
   129  	defer cancel()
   130  
   131  	if opts, err = config.New(os.Stdout); err != nil {
   132  		printVersion()
   133  		fmt.Println(err)
   134  		if !opts.Help {
   135  			exitCode.Store(ExitCodeConfigError)
   136  		}
   137  		return
   138  	}
   139  	logger = log.Init(opts.Logging)
   140  	mainContext = log.WithLogger(mainContext, logger)
   141  
   142  	logger.Debugf("opts: %+v", opts)
   143  
   144  	// sourcesReaderWriter reads/writes the monitored sources (databases, patroni clusters, pgpools, etc.) information
   145  	// metricsReaderWriter reads/writes the metric and preset definitions
   146  	sourcesReaderWriter, metricsReaderWriter, err := NewConfigurationReaders(opts)
   147  	if err != nil {
   148  		exitCode.Store(ExitCodeInitError)
   149  		logger.Error(err)
   150  		return
   151  	}
   152  
   153  	//check if we want to upgrade the configuration, which can be one of the following:
   154  	// - PostgreSQL database schema
   155  	// - YAML file schema
   156  	if m, ok := metricsReaderWriter.(metrics.Migrator); ok {
   157  		if err = UpgradeConfiguration(m); err != nil {
   158  			exitCode.Store(ExitCodeUpgradeError)
   159  			logger.Error(err)
   160  			return
   161  		}
   162  	}
   163  
   164  	if opts.Init {
   165  		// At this point we have initialised the sources, metrics and presets configurations.
   166  		// Any fatal errors are handled by the configuration readers. So we may exit gracefully.
   167  		return
   168  	}
   169  
   170  	if !opts.Ping {
   171  		uifs, _ := fs.Sub(webuifs, "webui/build")
   172  		ui := webserver.Init(opts.WebUI, uifs, metricsReaderWriter, sourcesReaderWriter, logger)
   173  		if ui == nil {
   174  			os.Exit(int(ExitCodeWebUIError))
   175  		}
   176  	}
   177  
   178  	reaper := reaper.NewReaper(opts, sourcesReaderWriter, metricsReaderWriter)
   179  	if err = reaper.Reap(mainContext); err != nil {
   180  		logger.Error(err)
   181  		exitCode.Store(ExitCodeFatalError)
   182  	}
   183  }
