    52  func writeMetricsToPostgres(ctx context.Context, conn db.PgxIface, metricDefs *Metrics) error {
    53  	logger := log.GetLogger(ctx)
    54  	logger.Info("writing metrics definitions to configuration database...")
    55  	tx, err := conn.Begin(ctx)
    56  	if err != nil {
    57  		return err
    58  	}
    59  	defer func() { _ = tx.Rollback(ctx) }()
    60  	for metricName, metric := range metricDefs.MetricDefs {
    61  		_, err = tx.Exec(ctx, `INSERT INTO pgwatch3.metric (name, sqls, init_sql, description, node_status, gauges, is_instance_level, storage_name)
    62  		values ($1, $2, $3, $4, $5, $6, $7, $8) ON CONFLICT (name) 
    63  		DO UPDATE SET sqls = $2, init_sql = $3, description = $4, node_status = $5, 
    64  		gauges = $6, is_instance_level = $7, storage_name = $8`,
    65  			metricName, metric.SQLs, metric.InitSQL, metric.Description, metric.NodeStatus, metric.Gauges, metric.IsInstanceLevel, metric.StorageName)
    66  		if err != nil {
    67  			return err
    68  		}
    69  	}
    70  	for presetName, preset := range metricDefs.PresetDefs {
    71  		_, err = tx.Exec(ctx, `INSERT INTO pgwatch3.preset (name, description, metrics) 
    72  		VALUES ($1, $2, $3) ON CONFLICT (name) DO UPDATE SET description = $2, metrics = $3;`,
    73  			presetName, preset.Description, preset.Metrics)
    74  		if err != nil {
    75  			return err
    76  		}
    77  	}
    78  	return tx.Commit(ctx)
