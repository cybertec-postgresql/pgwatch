   114  func ParseLogs(ctx context.Context, conn db.PgxIface, mdb *sources.MonitoredDatabase, realDbname, metricName string, configMap map[string]float64, storeCh chan<- []MeasurementMessage) {
   115  
   116  	var latest, previous, serverMessagesLang string
   117  	var latestHandle *os.File
   118  	var reader *bufio.Reader
   119  	var linesRead = 0 // to skip over already parsed lines on Postgres server restart for example
   120  	var logsMatchRegex, logsMatchRegexPrev, logsGlobPath string
   121  	var lastSendTime time.Time                    // to storage channel
   122  	var eventCounts = make(map[string]int64)      // for the specific DB. [WARNING: 34, ERROR: 10, ...], zeroed on storage send
   123  	var eventCountsTotal = make(map[string]int64) // for the whole instance
   124  	var hostConfig sources.HostConfigAttrs
   125  	var config = configMap
   126  	var interval float64
   127  	var err error
   128  	var firstRun = true
   129  	var csvlogRegex *regexp.Regexp
   130  	dbUniqueName := realDbname
   131  	logger := log.GetLogger(ctx)
   132  	for { // re-try loop. re-start in case of FS errors or just to refresh host config
   133  		select {
   134  		case <-ctx.Done():
   135  			return
   136  		default:
   137  			if interval == 0 {
   138  				interval = config[metricName]
   139  			}
   140  		}
   141  
   142  		if hostConfig.LogsMatchRegex != "" {
   143  			logsMatchRegex = hostConfig.LogsMatchRegex
   144  		}
   145  		if logsMatchRegex == "" {
   146  			logger.Debugf("[%s] Log parsing enabled with default CSVLOG regex", dbUniqueName)
   147  			logsMatchRegex = CSVLogDefaultRegEx
   148  		}
   149  		if hostConfig.LogsGlobPath != "" {
   150  			logsGlobPath = hostConfig.LogsGlobPath
   151  		}
   152  		if logsGlobPath == "" {
   153  			logsGlobPath, err = tryDetermineLogFolder(ctx, conn)
   154  			if err != nil {
   155  				logger.WithError(err).Printf("[%s] Could not determine Postgres logs parsing folder. Configured logs_glob_path = %s", dbUniqueName, logsGlobPath)
   156  				time.Sleep(60 * time.Second)
   157  				continue
   158  			}
   159  		}
   160  		serverMessagesLang, err = tryDetermineLogMessagesLanguage(ctx, conn)
   161  		if err != nil {
   162  			logger.WithError(err).Warningf("[%s] Could not determine language (lc_collate) used for server logs, cannot parse logs...", dbUniqueName)
   163  			time.Sleep(60 * time.Second)
   164  			continue
   165  		}
   166  
   167  		if logsMatchRegexPrev != logsMatchRegex { // avoid regex recompile if no changes
   168  			csvlogRegex, err = regexp.Compile(logsMatchRegex)
   169  			if err != nil {
   170  				logger.Errorf("[%s] Invalid regex: %s", dbUniqueName, logsMatchRegex)
   171  				time.Sleep(60 * time.Second)
   172  				continue
   173  			}
   174  			logger.Infof("[%s] Changing logs parsing regex to: %s", dbUniqueName, logsMatchRegex)
   175  			logsMatchRegexPrev = logsMatchRegex
   176  		}
   177  
   178  		logger.Debugf("[%s] Considering log files determined by glob pattern: %s", dbUniqueName, logsGlobPath)
   179  
   180  		if latest == "" || firstRun {
   181  
   182  			globMatches, err := filepath.Glob(logsGlobPath)
   183  			if err != nil || len(globMatches) == 0 {
   184  				logger.Infof("[%s] No logfiles found to parse from glob '%s'. Sleeping 60s...", dbUniqueName, logsGlobPath)
   185  				time.Sleep(60 * time.Second)
   186  				continue
   187  			}
   188  
   189  			logger.Debugf("[%s] Found %v logfiles from glob pattern, picking the latest", dbUniqueName, len(globMatches))
   190  			if len(globMatches) > 1 {
   191  				// find latest timestamp
   192  				latest, _ = getFileWithLatestTimestamp(globMatches)
   193  				if latest == "" {
   194  					logger.Warningf("[%s] Could not determine the latest logfile. Sleeping 60s...", dbUniqueName)
   195  					time.Sleep(60 * time.Second)
   196  					continue
   197  				}
   198  
   199  			} else if len(globMatches) == 1 {
   200  				latest = globMatches[0]
   201  			}
   202  			logger.Infof("[%s] Starting to parse logfile: %s ", dbUniqueName, latest)
   203  		}
   204  
   205  		if latestHandle == nil {
   206  			latestHandle, err = os.Open(latest)
   207  			if err != nil {
   208  				logger.Warningf("[%s] Failed to open logfile %s: %s. Sleeping 60s...", dbUniqueName, latest, err)
   209  				time.Sleep(60 * time.Second)
   210  				continue
   211  			}
   212  			reader = bufio.NewReader(latestHandle)
   213  			if previous == latest && linesRead > 0 { // handle postmaster restarts
   214  				i := 1
   215  				for i <= linesRead {
   216  					_, err = reader.ReadString('\n')
   217  					if err == io.EOF && i < linesRead {
   218  						logger.Warningf("[%s] Failed to open logfile %s: %s. Sleeping 60s...", dbUniqueName, latest, err)
   219  						linesRead = 0
   220  						break
   221  					} else if err != nil {
   222  						logger.Warningf("[%s] Failed to skip %d logfile lines for %s, there might be duplicates reported. Error: %s", dbUniqueName, linesRead, latest, err)
   223  						time.Sleep(60 * time.Second)
   224  						linesRead = i
   225  						break
   226  					}
   227  					i++
   228  				}
   229  				logger.Debug("[%s] Skipped %d already processed lines from %s", dbUniqueName, linesRead, latest)
   230  			} else if firstRun { // seek to end
   231  				_, _ = latestHandle.Seek(0, 2)
   232  				firstRun = false
   233  			}
   234  		}
   235  
   236  		var eofSleepMillis = 0
   237  		// readLoopStart := time.Now()
   238  
   239  		for {
   240  			// if readLoopStart.Add(time.Second * time.Duration(opts.Source.Refresh)).Before(time.Now()) {
   241  			// 	break // refresh config
   242  			// }
   243  			line, err := reader.ReadString('\n')
   244  			if err != nil && err != io.EOF {
   245  				logger.Warningf("[%s] Failed to read logfile %s: %s. Sleeping 60s...", dbUniqueName, latest, err)
   246  				err = latestHandle.Close()
   247  				if err != nil {
   248  					logger.Warningf("[%s] Failed to close logfile %s properly: %s", dbUniqueName, latest, err)
   249  				}
   250  				latestHandle = nil
   251  				time.Sleep(60 * time.Second)
   252  				break
   253  			}
   254  
   255  			if err == io.EOF {
   256  				//log.Debugf("[%s] EOF reached for logfile %s", dbUniqueName, latest)
   257  				if eofSleepMillis < 5000 && float64(eofSleepMillis) < interval*1000 {
   258  					eofSleepMillis += 100 // progressively sleep more if nothing going on but not more that 5s or metric interval
   259  				}
   260  				time.Sleep(time.Millisecond * time.Duration(eofSleepMillis))
   261  
   262  				// check for newly opened logfiles
   263  				file, _ := getFileWithNextModTimestamp(logsGlobPath, latest)
   264  				if file != "" {
   265  					previous = latest
   266  					latest = file
   267  					err = latestHandle.Close()
   268  					latestHandle = nil
   269  					if err != nil {
   270  						logger.Warningf("[%s] Failed to close logfile %s properly: %s", dbUniqueName, latest, err)
   271  					}
   272  					logger.Infof("[%s] Switching to new logfile: %s", dbUniqueName, file)
   273  					linesRead = 0
   274  					break
   275  				}
   276  			} else {
   277  				eofSleepMillis = 0
   278  				linesRead++
   279  			}
   280  
   281  			if err == nil && line != "" {
   282  
   283  				matches := csvlogRegex.FindStringSubmatch(line)
   284  				if len(matches) == 0 {
   285  					//log.Debugf("[%s] No logline regex match for line:", dbUniqueName) // normal case actually for queries spanning multiple loglines
   286  					//log.Debugf(line)
   287  					goto send_to_storage_if_needed
   288  				}
   289  
   290  				result := regexMatchesToMap(csvlogRegex, matches)
   291  				//log.Debugf("RegexMatchesToMap: %+v", result)
   292  				errorSeverity, ok := result["error_severity"]
   293  				if !ok {
   294  					logger.Error("error_severity group must be defined in parse regex:", csvlogRegex)
   295  					time.Sleep(time.Minute)
   296  					break
   297  				}
   298  				if serverMessagesLang != "en" {
   299  					errorSeverity = severityToEnglish(serverMessagesLang, errorSeverity)
   300  				}
   301  
   302  				databaseName, ok := result["database_name"]
   303  				if !ok {
   304  					logger.Error("database_name group must be defined in parse regex:", csvlogRegex)
   305  					time.Sleep(time.Minute)
   306  					break
   307  				}
   308  				if realDbname == databaseName {
   309  					eventCounts[errorSeverity]++
   310  				}
   311  				eventCountsTotal[errorSeverity]++
   312  			}
   313  
   314  		send_to_storage_if_needed:
   315  			if lastSendTime.IsZero() || lastSendTime.Before(time.Now().Add(-1*time.Second*time.Duration(interval))) {
   316  				logger.Debugf("[%s] Sending log event counts for last interval to storage channel. Local eventcounts: %+v, global eventcounts: %+v", dbUniqueName, eventCounts, eventCountsTotal)
   317  				metricStoreMessages := eventCountsToMetricStoreMessages(eventCounts, eventCountsTotal, mdb)
   318  				storeCh <- metricStoreMessages
   319  				zeroEventCounts(eventCounts)
   320  				zeroEventCounts(eventCountsTotal)
   321  				lastSendTime = time.Now()
   322  			}
   323  
   324  		} // file read loop
   325  	} // config loop

   348  func tryDetermineLogFolder(ctx context.Context, conn db.PgxIface) (string, error) {

   375  func regexMatchesToMap(csvlogRegex *regexp.Regexp, matches []string) map[string]string {
   376  	result := make(map[string]string)
   377  	if len(matches) == 0 || csvlogRegex == nil {
   378  		return result
   379  	}
   380  	for i, name := range csvlogRegex.SubexpNames() {
   381  		if i != 0 && name != "" {
   382  			result[name] = matches[i]
   383  		}
   384  	}
   385  	return result
