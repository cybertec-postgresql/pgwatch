    82  	cmdOpts := new(Options)

    84  	var err error
    85  	if _, err = parser.Parse(); err != nil {
    86  		if flagsErr, ok := err.(*flags.Error); ok && flagsErr.Type == flags.ErrHelp {
    87  			cmdOpts.Help = true
    88  		}
    89  		if !flags.WroteHelp(err) {
    90  			parser.WriteHelp(writer)
    91  		}
    92  	}
    93  	if err == nil {
    94  		err = validateConfig(cmdOpts)
    95  	}
    96  	return cmdOpts, err
    97  }
