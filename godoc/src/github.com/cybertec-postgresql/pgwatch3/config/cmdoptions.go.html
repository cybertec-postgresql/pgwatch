    72  func New(writer io.Writer) (*Options, error) {
    73  	cmdOpts := new(Options)
    74  	parser := flags.NewParser(cmdOpts, flags.HelpFlag)
    75  	parser.SubcommandsOptional = true // if not command specified, start monitoring
    76  	addCommands(parser, cmdOpts)
    77  	nonParsedArgs, err := parser.Parse()
    78  	if err != nil { //subcommands executed as part of parsing
    79  		if flagsErr, ok := err.(*flags.Error); ok && flagsErr.Type == flags.ErrHelp {
    80  			cmdOpts.Help = true
    81  		}
    82  		if !flags.WroteHelp(err) {
    83  			parser.WriteHelp(writer)
    84  		}
    85  	} else {
    86  		if !cmdOpts.CommandCompleted && len(nonParsedArgs) > 0 {
    87  			err = fmt.Errorf("unknown argument(s): %v", nonParsedArgs)
    88  		}
    89  	}
    90  	if err == nil {
    91  		err = validateConfig(cmdOpts)
    92  	}
    93  	return cmdOpts, err
    94  }

   106  func (c *Options) GetConfigKind(arg string) (_ Kind, err error) {
   107  	if arg == "" {
   108  		return Kind(ConfigError), errors.New("no configuration provided")
   109  	}
   110  	if _, err := pgx.ParseConfig(arg); err == nil {
   111  		return Kind(ConfigPgURL), nil
   112  	}
   113  	var fi os.FileInfo
   114  	if fi, err = os.Stat(arg); err == nil {
   115  		if fi.IsDir() {
   116  			return Kind(ConfigFolder), nil
   117  		}
   118  		return Kind(ConfigFile), nil
   119  	}
   120  	return Kind(ConfigError), err
   121  }

   123  // InitMetricReader creates a new source reader based on the configuration kind from the options.
   124  func (c *Options) InitMetricReader(ctx context.Context) (err error) {
   125  	var configKind Kind
   126  	if c.Metrics.Metrics == "" { //if config database is configured, use it for metrics as well
   127  		if k, err := c.GetConfigKind(c.Sources.Sources); err == nil && k == ConfigPgURL {
   128  			c.Metrics.Metrics = c.Sources.Sources
   129  		} else { // otherwise use built-in metrics
   130  			c.MetricsReaderWriter, err = metrics.NewYAMLMetricReaderWriter(ctx, "")
   131  			return err
   132  		}
   133  	}
   134  	if configKind, err = c.GetConfigKind(c.Metrics.Metrics); err != nil {
   135  		return
   136  	}
   137  	switch configKind {
   138  	case ConfigPgURL:
   139  		var configDb db.PgxPoolIface
   140  		if configDb, err = db.New(ctx, c.Sources.Sources); err != nil {
   141  			return err
   142  		}
   143  		c.MetricsReaderWriter, err = metrics.NewPostgresMetricReaderWriter(ctx, configDb)
   144  	default:
   145  		c.MetricsReaderWriter, err = metrics.NewYAMLMetricReaderWriter(ctx, c.Metrics.Metrics)
   146  	}
   147  	return err
   148  }

   151  func (c *Options) InitSourceReader(ctx context.Context) (err error) {
   152  	var configKind Kind
   153  	if configKind, err = c.GetConfigKind(c.Sources.Sources); err != nil {

   155  	}
   156  	switch configKind {
   157  	case ConfigPgURL:

   160  			return err

   162  		c.SourcesReaderWriter, err = sources.NewPostgresSourcesReaderWriter(ctx, configDb)
   163  	default:

   165  	}
