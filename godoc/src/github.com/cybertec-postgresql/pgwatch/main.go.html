    37  // SetupCloseHandler creates a 'listener' on a new goroutine which will notify the
    38  // program if it receives an interrupt from the OS. We then handle this by calling
    39  // our clean up procedure and exiting the program.
    40  func SetupCloseHandler(cancel context.CancelFunc) {
    41  	c := make(chan os.Signal, 2)
    42  	signal.Notify(c, os.Interrupt, syscall.SIGTERM)
    43  	go func() {
    44  		<-c
    45  		log.GetLogger(mainCtx).Debug("SetupCloseHandler received an interrupt from OS. Closing session...")
    46  		cancel()
    47  		exitCode.Store(cmdopts.ExitCodeUserCancel)
    48  	}()
    49  }
