    11  // JSONWriter is a sink that writes metric measurements to a file in JSON format.
    12  // It supports compression and rotation of output files. The default rotation is based on the file size (100Mb).
    13  // JSONWriter is useful for debugging and testing purposes, as well as for integration with other systems,

    21  	jw := &JSONWriter{
    22  		ctx: ctx,

    29  func (jw *JSONWriter) Write(msgs []metrics.MeasurementMessage) error {
    30  	if jw.ctx.Err() != nil {
    31  		return jw.ctx.Err()
    32  	}
    33  	if len(msgs) == 0 {
    34  		return nil
    35  	}
    36  	enc := json.NewEncoder(jw.lw)
    37  	for _, msg := range msgs {
    38  		dataRow := map[string]any{
    39  			"metric":      msg.MetricName,
    40  			"data":        msg.Data,
    41  			"dbname":      msg.DBName,
    42  			"custom_tags": msg.CustomTags,
    43  		}
    44  		if err := enc.Encode(dataRow); err != nil {
    45  			return err
    46  		}
    47  	}
    48  	return nil
    49  }
