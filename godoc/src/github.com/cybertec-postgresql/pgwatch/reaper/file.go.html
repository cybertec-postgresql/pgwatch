    11  func DoesEmergencyTriggerfileExist(fname string) bool {
    12  	// Main idea of the feature is to be able to quickly free monitored DBs / network of any extra "monitoring effect" load.
    13  	// In highly automated K8s / IaC environments such a temporary change might involve pull requests, peer reviews, CI/CD etc
    14  	// which can all take too long vs "exec -it pgwatch-pod -- touch /tmp/pgwatch-emergency-pause".
    15  	// After creating the file it can still take up to --servers-refresh-loop-seconds (2min def.) for change to take effect!
    16  	if fname == "" {
    17  		return false
    18  	}
    19  	_, err := os.Stat(fname)
    20  	return err == nil
    21  }
